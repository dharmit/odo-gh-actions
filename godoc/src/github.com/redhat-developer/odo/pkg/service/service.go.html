    41  func DeleteOperatorService(client kclient.ClientInterface, serviceName string) error {
    42  	kind, name, err := SplitServiceKindName(serviceName)
    43  	if err != nil {
    44  		return fmt.Errorf("refer %q to see list of running services: %w", serviceName, err)
    45  	}
    46  
    47  	csv, err := client.GetCSVWithCR(kind)
    48  	if err != nil {
    49  		return err
    50  	}
    51  
    52  	if csv == nil {
    53  		return fmt.Errorf("unable to find any Operator providing the service %q", kind)
    54  	}
    55  
    56  	crs := client.GetCustomResourcesFromCSV(csv)
    57  	var cr *olm.CRDDescription
    58  
    59  	for _, c := range *crs {
    60  		customResource := c
    61  		if customResource.Kind == kind {
    62  			cr = &customResource
    63  			break
    64  		}
    65  	}
    66  
    67  	return client.DeleteDynamicResource(name, kclient.GetGVRFromCR(cr), false)
    68  }

    70  // ListOperatorServices lists all operator backed services.
    71  // It returns list of services, slice of services that it failed (if any) to list and error (if any)
    72  func ListOperatorServices(client kclient.ClientInterface) ([]unstructured.Unstructured, []string, error) {
    73  	klog.V(4).Info("Getting list of services")
    74  
    75  	// First let's get the list of all the operators in the namespace
    76  	csvs, err := client.ListClusterServiceVersions()
    77  	if err != nil {
    78  		return nil, nil, err
    79  	}
    80  
    81  	if err != nil {
    82  		return nil, nil, fmt.Errorf("unable to list operator backed services: %w", err)
    83  	}
    84  
    85  	var allCRInstances []unstructured.Unstructured
    86  	var failedListingCR []string
    87  
    88  	// let's get the Services a.k.a Custom Resources (CR) defined by each operator, one by one
    89  	for _, csv := range csvs.Items {
    90  		clusterServiceVersion := csv
    91  		klog.V(4).Infof("Getting services started from operator: %s", clusterServiceVersion.Name)
    92  		customResources := client.GetCustomResourcesFromCSV(&clusterServiceVersion)
    93  
    94  		// list and write active instances of each service/CR
    95  		var instances []unstructured.Unstructured
    96  		for _, cr := range *customResources {
    97  			customResource := cr
    98  
    99  			list, err := GetCRInstances(client, &customResource)
   100  			if err != nil {
   101  				crName := strings.Join([]string{csv.Name, cr.Kind}, "/")
   102  				klog.V(4).Infof("Failed to list instances of %q with error: %s", crName, err.Error())
   103  				failedListingCR = append(failedListingCR, crName)
   104  				continue
   105  			}
   106  
   107  			if len(list.Items) > 0 {
   108  				instances = append(instances, list.Items...)
   109  			}
   110  		}
   111  
   112  		// assuming there are more than one instances of a CR
   113  		allCRInstances = append(allCRInstances, instances...)
   114  	}
   115  
   116  	return allCRInstances, failedListingCR, nil
   117  }

   132  // SplitServiceKindName splits the service name provided for deletion by the
   133  // user. It has to be of the format <service-kind>/<service-name>. Example: EtcdCluster/myetcd
   134  func SplitServiceKindName(serviceName string) (string, string, error) {
   135  	sn := strings.SplitN(serviceName, "/", 2)
   136  	if len(sn) != 2 || sn[0] == "" || sn[1] == "" {
   137  		return "", "", fmt.Errorf("couldn't split %q into exactly two", serviceName)
   138  	}
   139  
   140  	kind := sn[0]
   141  	name := sn[1]
   142  
   143  	return kind, name, nil
   144  }

   147  	if devfileObj.Data == nil {

   188  // PushKubernetesResources updates service(s) from Kubernetes Inlined component in a devfile by creating new ones or removing old ones
   189  func PushKubernetesResources(client kclient.ClientInterface, devfileObj parser.DevfileObj, k8sComponents []devfile.Component, labels map[string]string, annotations map[string]string, context string) error {
   190  	// check csv support before proceeding
   191  	csvSupported, err := client.IsCSVSupported()
   192  	if err != nil {
   193  		return err
   194  	}
   195  
   196  	var deployed map[string]DeployedInfo
   197  
   198  	if csvSupported {
   199  		deployed, err = ListDeployedServices(client, labels)
   200  		if err != nil {
   201  			return err
   202  		}
   203  
   204  		for key, deployedResource := range deployed {
   205  			if deployedResource.isLinkResource {
   206  				delete(deployed, key)
   207  			}
   208  		}
   209  	}
   210  
   211  	// create an object on the kubernetes cluster for all the Kubernetes Inlined components
   212  	for _, c := range k8sComponents {
   213  		u, er := libdevfile.GetK8sComponentAsUnstructured(devfileObj, c.Name, context, devfilefs.DefaultFs{})
   214  		if er != nil {
   215  			return er
   216  		}
   217  		_, er = PushKubernetesResource(client, u, labels, annotations)
   218  		if er != nil {
   219  			return er
   220  		}
   221  		if csvSupported {
   222  			delete(deployed, u.GetKind()+"/"+u.GetName())
   223  		}
   224  	}
   225  
   226  	if csvSupported {
   227  		for key, val := range deployed {
   228  			if isLinkResource(val.Kind) {
   229  				continue
   230  			}
   231  			err = DeleteOperatorService(client, key)
   232  			if err != nil {
   233  				return err
   234  
   235  			}
   236  		}
   237  	}
   238  
   239  	return nil
   240  }

   242  // PushKubernetesResource pushes a Kubernetes resource (u) to the cluster using client
   243  // adding labels to the resource
   244  func PushKubernetesResource(client kclient.ClientInterface, u unstructured.Unstructured, labels map[string]string, annotations map[string]string) (bool, error) {
   245  	if isLinkResource(u.GetKind()) {
   246  		// it's a service binding related resource
   247  		return false, nil
   248  	}
   249  
   250  	isOp, err := isOperatorBackedService(client, u)
   251  	if err != nil {
   252  		return false, err
   253  	}
   254  
   255  	// Add all passed in labels to the k8s resource regardless if it's an operator or not
   256  	u.SetLabels(mergeMaps(u.GetLabels(), labels))
   257  
   258  	// Pass in all annotations to the k8s resource
   259  	u.SetAnnotations(mergeMaps(u.GetAnnotations(), annotations))
   260  
   261  	_, err = updateOperatorService(client, u)
   262  	return isOp, err
   263  }

   265  func isOperatorBackedService(client kclient.ClientInterface, u unstructured.Unstructured) (bool, error) {
   266  	restMapping, err := client.GetRestMappingFromUnstructured(u)
   267  	if err != nil {
   268  		return false, err
   269  	}
   270  	// check if the GVR of the CRD belongs to any of the CRs provided by any of the Operators
   271  	// if yes, it is an Operator backed service.
   272  	// if no, it is likely a Kubernetes built-in resource.
   273  	operatorGVRList, err := client.GetOperatorGVRList()
   274  	if err != nil {
   275  		return false, err
   276  	}
   277  
   278  	for _, i := range operatorGVRList {
   279  		if i.Resource == restMapping.Resource {
   280  			return true, nil
   281  		}
   282  	}
   283  	return false, nil
   284  }

   305  func ListDeployedServices(client kclient.ClientInterface, labels map[string]string) (map[string]DeployedInfo, error) {
   306  	deployed := map[string]DeployedInfo{}
   307  
   308  	deployedServices, _, err := ListOperatorServices(client)
   309  	if err != nil {
   310  		// We ignore ErrNoSuchOperator error as we can deduce Operator Services are not installed
   311  		return nil, err
   312  	}
   313  	for _, svc := range deployedServices {
   314  		name := svc.GetName()
   315  		kind := svc.GetKind()
   316  		deployedLabels := svc.GetLabels()
   317  		if odolabels.IsManagedByOdo(deployedLabels) && odolabels.GetComponentName(deployedLabels) == odolabels.GetComponentName(labels) {
   318  			deployed[kind+"/"+name] = DeployedInfo{
   319  				Kind:           kind,
   320  				Name:           name,
   321  				isLinkResource: isLinkResource(kind),
   322  			}
   323  		}
   324  	}
   325  
   326  	return deployed, nil
   327  }

   329  // UpdateServicesWithOwnerReferences adds an owner reference to an inlined Kubernetes resource (except service binding objects)
   330  // if not already present in the list of owner references
   331  func UpdateServicesWithOwnerReferences(client kclient.ClientInterface, devfileObj parser.DevfileObj, k8sComponents []devfile.Component, ownerReference metav1.OwnerReference, context string) error {
   332  	for _, c := range k8sComponents {
   333  		// get the string representation of the YAML definition of a CRD
   334  		u, err := libdevfile.GetK8sComponentAsUnstructured(devfileObj, c.Name, context, devfilefs.DefaultFs{})
   335  		if err != nil {
   336  			return err
   337  		}
   338  
   339  		if isLinkResource(u.GetKind()) {
   340  			// ignore service binding resources
   341  			continue
   342  		}
   343  
   344  		restMapping, err := client.GetRestMappingFromUnstructured(u)
   345  		if err != nil {
   346  			return err
   347  		}
   348  
   349  		d, err := client.GetDynamicResource(restMapping.Resource, u.GetName())
   350  		if err != nil {
   351  			return err
   352  		}
   353  
   354  		found := false
   355  		for _, ownerRef := range d.GetOwnerReferences() {
   356  			if ownerRef.UID == ownerReference.UID {
   357  				found = true
   358  				break
   359  			}
   360  		}
   361  		if found {
   362  			continue
   363  		}
   364  		d.SetOwnerReferences(append(d.GetOwnerReferences(), ownerReference))
   365  
   366  		err = client.UpdateDynamicResource(restMapping.Resource, u.GetName(), d)
   367  		if err != nil {
   368  			return err
   369  		}
   370  	}
   371  	return nil
   372  }

   378  // updateOperatorService creates the given operator on the cluster

   380  func updateOperatorService(client kclient.ClientInterface, u unstructured.Unstructured) (bool, error) {
   381  
   382  	// Create the service on cluster

   384  	if err != nil {
   385  		return false, err
   386  	}

   389  		createSpinner := log.Spinnerf("Creating kind %s", u.GetKind())

   391  	}
   392  	return updated, err
