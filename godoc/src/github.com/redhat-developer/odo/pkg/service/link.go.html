    35  // PushLinks updates Link(s) from Kubernetes Inlined component in a devfile by creating new ones or removing old ones
    36  // if service binding operator is not present, it will call pushLinksWithoutOperator to create the links without it.
    37  func PushLinks(client kclient.ClientInterface, devfileObj parser.DevfileObj, k8sComponents []devfile.Component, labels map[string]string, deployment *v1.Deployment, context string) error {
    38  	serviceBindingSupport, err := client.IsServiceBindingSupported()
    39  	if err != nil {
    40  		return err
    41  	}
    42  
    43  	if !serviceBindingSupport {
    44  		klog.V(4).Info("Service Binding Operator is not installed on cluster. Service Binding will be created by odo using SB library.")
    45  		return pushLinksWithoutOperator(client, devfileObj, k8sComponents, labels, deployment, context)
    46  	}
    47  
    48  	return pushLinksWithOperator(client, devfileObj, k8sComponents, labels, deployment, context)
    49  }

    51  // pushLinksWithOperator creates links or deletes links (if service binding operator is installed) between components and services
    52  func pushLinksWithOperator(client kclient.ClientInterface, devfileObj parser.DevfileObj, k8sComponents []devfile.Component, labels map[string]string, deployment *v1.Deployment, context string) error {
    53  
    54  	ownerReference := generator.GetOwnerReference(deployment)
    55  	deployed, err := ListDeployedServices(client, labels)
    56  	if err != nil {
    57  		return err
    58  	}
    59  
    60  	for key, deployedResource := range deployed {
    61  		if !deployedResource.isLinkResource {
    62  			delete(deployed, key)
    63  		}
    64  	}
    65  
    66  	// create an object on the kubernetes cluster for all the Kubernetes Inlined components
    67  	var strCRD string
    68  	for _, c := range k8sComponents {
    69  		// get the string representation of the YAML definition of a CRD
    70  		strCRD, err = libdevfile.GetK8sManifestWithVariablesSubstituted(devfileObj, c.Name, context, devfilefs.DefaultFs{})
    71  		if err != nil {
    72  			return err
    73  		}
    74  
    75  		// convert the YAML definition into map[string]interface{} since it's needed to create dynamic resource
    76  		u := unstructured.Unstructured{}
    77  		if e := yaml.Unmarshal([]byte(strCRD), &u.Object); e != nil {
    78  			return e
    79  		}
    80  
    81  		if !isLinkResource(u.GetKind()) {
    82  			// operator hub is not installed on the cluster
    83  			// or it's a service binding related resource
    84  			continue
    85  		}
    86  
    87  		crdName := u.GetName()
    88  		u.SetOwnerReferences([]metav1.OwnerReference{ownerReference})
    89  		u.SetLabels(labels)
    90  
    91  		_, err = updateOperatorService(client, u)
    92  		delete(deployed, u.GetKind()+"/"+crdName)
    93  		if err != nil {
    94  			if strings.Contains(err.Error(), "already exists") {
    95  				// TODO: better way to handle this might be introduced by https://github.com/redhat-developer/odo/issues/4553
    96  				continue // this ensures that services slice is not updated
    97  			} else {
    98  				return err
    99  			}
   100  		}
   101  	}
   102  
   103  	for key, val := range deployed {
   104  		if !isLinkResource(val.Kind) {
   105  			continue
   106  		}
   107  		err = DeleteOperatorService(client, key)
   108  		if err != nil {
   109  			return err
   110  
   111  		}
   112  	}
   113  
   114  	return nil
   115  }

   117  // pushLinksWithoutOperator creates links or deletes links (if service binding operator is not installed) between components and services
   118  // returns true if the component needs to be restarted (a secret was generated and added to the deployment)
   119  func pushLinksWithoutOperator(client kclient.ClientInterface, devfileObj parser.DevfileObj, k8sComponents []devfile.Component, labels map[string]string, deployment *v1.Deployment, context string) error {
   120  
   121  	// check csv support before proceeding
   122  	csvSupport, err := client.IsCSVSupported()
   123  	if err != nil {
   124  		return err
   125  	}
   126  
   127  	secrets, err := client.ListSecrets(odolabels.GetSelector(odolabels.GetComponentName(labels), odolabels.GetAppName(labels), odolabels.ComponentAnyMode, false))
   128  	if err != nil {
   129  		return err
   130  	}
   131  
   132  	ownerReferences := generator.GetOwnerReference(deployment)
   133  
   134  	clusterLinksMap := make(map[string]string)
   135  	for _, secret := range secrets {
   136  		if value, ok := secret.GetLabels()[LinkLabel]; ok {
   137  			clusterLinksMap[value] = secret.Name
   138  		}
   139  	}
   140  
   141  	localLinksMap := make(map[string]string)
   142  	// create an object on the kubernetes cluster for all the Kubernetes Inlined components
   143  	var strCRD string
   144  	for _, c := range k8sComponents {
   145  		// get the string representation of the YAML definition of a CRD
   146  		strCRD, err = libdevfile.GetK8sManifestWithVariablesSubstituted(devfileObj, c.Name, context, devfilefs.DefaultFs{})
   147  		if err != nil {
   148  			return err
   149  		}
   150  
   151  		// convert the YAML definition into map[string]interface{} since it's needed to create dynamic resource
   152  		u := unstructured.Unstructured{}
   153  		if e := yaml.Unmarshal([]byte(strCRD), &u.Object); e != nil {
   154  			return e
   155  		}
   156  
   157  		if !isLinkResource(u.GetKind()) {
   158  			// not a service binding object, thus continue
   159  			continue
   160  		}
   161  		localLinksMap[c.Name] = strCRD
   162  	}
   163  
   164  	var processingPipeline sboPipeline.Pipeline
   165  
   166  	deploymentGVK, err := client.GetDeploymentAPIVersion()
   167  	if err != nil {
   168  		return err
   169  	}
   170  
   171  	// delete the links not present on the devfile
   172  	for linkName, secretName := range clusterLinksMap {
   173  		if _, ok := localLinksMap[linkName]; !ok {
   174  
   175  			// recreate parts of the service binding request for deletion
   176  			var newServiceBinding sboApi.ServiceBinding
   177  			newServiceBinding.Name = linkName
   178  			newServiceBinding.Namespace = client.GetCurrentNamespace()
   179  			newServiceBinding.Spec.Application = sboApi.Application{
   180  				Ref: sboApi.Ref{
   181  					Name:    deployment.Name,
   182  					Group:   deploymentGVK.Group,
   183  					Version: deploymentGVK.Version,
   184  					Kind:    deploymentGVK.Kind,
   185  				},
   186  			}
   187  			newServiceBinding.Status.Secret = secretName
   188  
   189  			// set the deletion time stamp to trigger deletion
   190  			timeNow := metav1.Now()
   191  			newServiceBinding.DeletionTimestamp = &timeNow
   192  
   193  			// if the pipeline was created before
   194  			// skip deletion
   195  			if processingPipeline == nil {
   196  				processingPipeline, err = getPipeline(client)
   197  				if err != nil {
   198  					return err
   199  				}
   200  			}
   201  			_, err = processingPipeline.Process(&newServiceBinding)
   202  			if err != nil {
   203  				return err
   204  			}
   205  
   206  			// since the library currently doesn't delete the secret after unbinding
   207  			// delete the secret manually
   208  			err = client.DeleteSecret(secretName, client.GetCurrentNamespace())
   209  			if err != nil {
   210  				return err
   211  			}
   212  		}
   213  	}
   214  
   215  	var serviceCompMap map[string]string
   216  
   217  	// create the links
   218  	for linkName, strCRD := range localLinksMap {
   219  		if _, ok := clusterLinksMap[linkName]; !ok {
   220  			if serviceCompMap == nil {
   221  				// prevent listing of services unless required
   222  				services, e := client.ListServices("")
   223  				if e != nil {
   224  					return e
   225  				}
   226  
   227  				// get the services and get match them against the component
   228  				serviceCompMap = make(map[string]string)
   229  				for _, service := range services {
   230  					serviceCompMap[service.Name] = odolabels.GetComponentName(service.Labels)
   231  				}
   232  			}
   233  
   234  			// get the string representation of the YAML definition of a CRD
   235  			var serviceBinding sboApi.ServiceBinding
   236  			err = yaml.Unmarshal([]byte(strCRD), &serviceBinding)
   237  			if err != nil {
   238  				return err
   239  			}
   240  
   241  			if len(serviceBinding.Spec.Services) != 1 {
   242  				continue
   243  			}
   244  
   245  			if !csvSupport && !isLinkResource(serviceBinding.Spec.Services[0].Kind) {
   246  				// ignore service binding objects linked to services if csv support is not present on the cluster
   247  				continue
   248  			}
   249  
   250  			// set the labels and namespace
   251  			serviceBinding.SetLabels(labels)
   252  			serviceBinding.Namespace = client.GetCurrentNamespace()
   253  			ns := client.GetCurrentNamespace()
   254  			serviceBinding.Spec.Services[0].Namespace = &ns
   255  
   256  			_, err = json.MarshalIndent(serviceBinding, " ", " ")
   257  			if err != nil {
   258  				return err
   259  			}
   260  
   261  			if processingPipeline == nil {
   262  				processingPipeline, err = getPipeline(client)
   263  				if err != nil {
   264  					return err
   265  				}
   266  			}
   267  
   268  			_, err = processingPipeline.Process(&serviceBinding)
   269  			if err != nil {
   270  				if kerrors.IsForbidden(err) {
   271  					// due to https://github.com/redhat-developer/service-binding-operator/issues/1003
   272  					return fmt.Errorf("please install the service binding operator")
   273  				}
   274  				return err
   275  			}
   276  
   277  			if len(serviceBinding.Status.Secret) == 0 {
   278  				return fmt.Errorf("no secret was provided by service binding's pipleine")
   279  			}
   280  
   281  			// get the generated secret and update it with the labels and owner reference
   282  			secret, err := client.GetSecret(serviceBinding.Status.Secret, client.GetCurrentNamespace())
   283  			if err != nil {
   284  				return err
   285  			}
   286  			secret.Labels = labels
   287  			secret.Labels[LinkLabel] = linkName
   288  			if _, ok := serviceCompMap[serviceBinding.Spec.Services[0].Name]; ok {
   289  				secret.Labels[ServiceLabel] = serviceCompMap[serviceBinding.Spec.Services[0].Name]
   290  			} else {
   291  				secret.Labels[ServiceLabel] = serviceBinding.Spec.Services[0].Name
   292  			}
   293  			secret.Labels[ServiceKind] = serviceBinding.Spec.Services[0].Kind
   294  			if serviceBinding.Spec.Services[0].Kind != "Service" {
   295  				// the service name is stored as kind-name as `/` is not a valid char for labels of kubernetes secrets
   296  				secret.Labels[ServiceLabel] = fmt.Sprintf("%v-%v", serviceBinding.Spec.Services[0].Kind, serviceBinding.Spec.Services[0].Name)
   297  			}
   298  			secret.SetOwnerReferences([]metav1.OwnerReference{ownerReferences})
   299  			_, err = client.UpdateSecret(secret, client.GetCurrentNamespace())
   300  			if err != nil {
   301  				return err
   302  			}
   303  		}
   304  	}
   305  
   306  	return nil
   307  }

   309  // getPipeline gets the pipeline to process service binding requests
   310  func getPipeline(client kclient.ClientInterface) (sboPipeline.Pipeline, error) {
   311  	mgr, err := ctrl.NewManager(client.GetClientConfig(), ctrl.Options{
   312  		Scheme: runtime.NewScheme(),
   313  		// disable the health probes to prevent binding to them
   314  		HealthProbeBindAddress: "0",
   315  		// disable the prometheus metrics
   316  		MetricsBindAddress: "0",
   317  	})
   318  	if err != nil {
   319  		return nil, err
   320  	}
   321  
   322  	authClient, err := authv1.NewForConfig(client.GetClientConfig())
   323  	if err != nil {
   324  		return nil, err
   325  	}
   326  
   327  	return OdoDefaultBuilder.WithContextProvider(
   328  		sboContext.Provider(
   329  			client.GetDynamicClient(),
   330  			authClient.SubjectAccessReviews(),
   331  			sboKubernetes.ResourceLookup(mgr.GetRESTMapper()),
   332  		),
   333  	).Build(), nil
   334  }
