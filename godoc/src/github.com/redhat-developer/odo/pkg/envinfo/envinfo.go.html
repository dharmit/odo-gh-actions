    70  // getEnvInfoFile first checks for the ENVINFO variable
    71  // then we check for directory and eventually the file (which we return as a string)
    72  func getEnvInfoFile(envDir string) (string, string, error) {
    73  	if env, ok := os.LookupEnv(envInfoEnvName); ok {
    74  		return env, filepath.Join(env, "..", "..", "..", "devfile.yaml"), nil
    75  	}
    76  
    77  	if envDir == "" {
    78  		var err error
    79  		envDir, err = os.Getwd()
    80  		if err != nil {
    81  			return "", "", err
    82  		}
    83  	}
    84  
    85  	return filepath.Join(envDir, ".odo", "env", envInfoFileName), location.DevfileLocation(envDir), nil
    86  }

    93  // newEnvSpecificInfo retrieves the env.yaml file, if it does not exist, we return a *BLANK* environment file.
    94  func newEnvSpecificInfo(envDir string, fs filesystem.Filesystem) (*EnvSpecificInfo, error) {
    95  	// Get the path of the environment file
    96  	envInfoFile, devfilePath, err := getEnvInfoFile(envDir)
    97  	if err != nil {
    98  		return nil, fmt.Errorf("failed to get the path of the environment file: %w", err)
    99  	}
   100  
   101  	// Organize that information into a struct
   102  	e := EnvSpecificInfo{
   103  		EnvInfo:           NewEnvInfo(),
   104  		devfilePath:       devfilePath,
   105  		Filename:          envInfoFile,
   106  		envinfoFileExists: true,
   107  		fs:                fs,
   108  	}
   109  
   110  	// If the env.yaml file does not exist then we simply return and set e.envinfoFileExists as false
   111  	if _, err = e.fs.Stat(envInfoFile); os.IsNotExist(err) {
   112  		e.envinfoFileExists = false
   113  		return &e, nil
   114  	}
   115  
   116  	// Retrieve the environment file
   117  	err = getFromFile(&e.EnvInfo, e.Filename)
   118  	if err != nil {
   119  		return nil, err
   120  	}
   121  
   122  	return &e, nil
   123  }

   146  // SetConfiguration sets the environment specific info such as the Cluster Host, Name, etc.
   147  // we then **write** this data to the environment yaml file (see envInfoFileName const)
   148  func (esi *EnvSpecificInfo) SetConfiguration(parameter string, value interface{}) (err error) {
   149  	if p, ok := asLocallySupportedParameter(parameter); ok {
   150  		switch p {
   151  		case "name":
   152  			val := value.(string)
   153  			esi.componentSettings.Name = val
   154  		case "project":
   155  			val := value.(string)
   156  			esi.componentSettings.Project = val
   157  		case "debugport":
   158  			val, err := strconv.Atoi(value.(string))
   159  			if err != nil {
   160  				return fmt.Errorf("failed to set debug port: %w", err)
   161  			}
   162  			esi.componentSettings.DebugPort = &val
   163  		}
   164  
   165  		return esi.writeToFile()
   166  	}
   167  	return fmt.Errorf("unknown parameter: %q is not a parameter in the odo environment file, please refer `odo env set --help` to see valid parameters", parameter)
   168  
   169  }

   171  // DeleteEnvDirIfEmpty Deletes the env directory if its empty
   172  func (esi *EnvSpecificInfo) DeleteEnvDirIfEmpty() error {
   173  	envDir := filepath.Dir(esi.Filename)
   174  	_, err := esi.fs.Stat(envDir)
   175  	if os.IsNotExist(err) {
   176  		// If the Env dir doesn't exist then we dont mind
   177  		return nil
   178  	} else if err != nil {
   179  		// Possible to not have permission to the dir
   180  		return err
   181  	}
   182  	f, err := esi.fs.Open(envDir)
   183  	if err != nil {
   184  		return err
   185  	}
   186  	defer f.Close()
   187  	_, err = f.Readdir(1)
   188  
   189  	// If directory is empty we can remove it
   190  	if err == io.EOF {
   191  		klog.V(4).Info("Deleting the env directory as well because its empty")
   192  
   193  		return esi.fs.Remove(envDir)
   194  	}
   195  	return err
   196  }

   222  // DeleteConfiguration is used to delete environment specific info from local odo envinfo
   223  func (esi *EnvSpecificInfo) DeleteConfiguration(parameter string) error {
   224  	for _, manParam := range manParams {
   225  		if parameter == manParam {
   226  			return fmt.Errorf("failed to unset %q: %q is mandatory parameter", parameter, parameter)
   227  		}
   228  	}
   229  
   230  	if p, ok := asLocallySupportedParameter(parameter); ok {
   231  		if err := util.DeleteConfiguration(&esi.componentSettings, p); err != nil {
   232  			return err
   233  		}
   234  		return esi.writeToFile()
   235  	}
   236  	return fmt.Errorf("unknown parameter: %q is not a parameter in the odo environment file, please refer `odo env unset --help` to unset a valid parameter", parameter)

   271  // GetContainers returns the Container components from the devfile
   272  // returns empty list if nil
   273  func (ei *EnvInfo) GetContainers() ([]localConfigProvider.LocalContainer, error) {
   274  	var localContainers []localConfigProvider.LocalContainer
   275  
   276  	devfileComponents, err := ei.devfileObj.Data.GetComponents(common.DevfileOptions{})
   277  	if err != nil {
   278  		return localContainers, err
   279  	}
   280  	for _, component := range devfileComponents {
   281  		if component.Container == nil {
   282  			continue
   283  		}
   284  		localContainers = append(localContainers, localConfigProvider.LocalContainer{
   285  			Name: component.Name,
   286  		})
   287  	}
   288  	return localContainers, nil
   289  }
