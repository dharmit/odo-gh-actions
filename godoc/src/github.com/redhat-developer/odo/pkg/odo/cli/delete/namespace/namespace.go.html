    84  // Run contains the logic for the 'odo delete namespace' command
    85  func (do *DeleteOptions) Run(ctx context.Context) error {
    86  	if do.forceFlag || ui.Proceed(fmt.Sprintf("Are you sure you want to delete namespace %q?", do.namespaceName)) {
    87  		// Create the "spinner"
    88  		s := &log.Status{}
    89  
    90  		// If the --wait parameter has been passed, we add a spinner..
    91  		if do.waitFlag {
    92  			s = log.Spinnerf("Waiting for %s %q to be deleted", do.commandName, do.namespaceName)
    93  			defer s.End(false)
    94  		}
    95  
    96  		err := do.clientset.ProjectClient.Delete(do.namespaceName, do.waitFlag)
    97  		if err != nil {
    98  			return err
    99  		}
   100  		s.End(true)
   101  
   102  		successMessage := fmt.Sprintf(`%s %q deleted`,
   103  			strings.Title(do.commandName), do.namespaceName)
   104  		log.Successf(successMessage)
   105  
   106  		return nil
   107  	}
   108  	log.Error("Aborting namespace deletion")
   109  	return nil
   110  }
