   109  func (o *DevOptions) Complete(cmdline cmdline.Cmdline, args []string) error {
   110  	var err error
   111  
   112  	// Define this first so that if user hits Ctrl+c very soon after running odo dev, odo doesn't panic
   113  	o.ctx, o.cancel = context.WithCancel(context.Background())
   114  
   115  	o.contextDir, err = os.Getwd()
   116  	if err != nil {
   117  		return err
   118  	}
   119  
   120  	isEmptyDir, err := location.DirIsEmpty(o.clientset.FS, o.contextDir)
   121  	if err != nil {
   122  		return err
   123  	}
   124  	if isEmptyDir {
   125  		return errors.New("this command cannot run in an empty directory, run the command in a directory containing source code or initialize using 'odo init'")
   126  	}
   127  	initFlags := o.clientset.InitClient.GetFlags(cmdline.GetFlags())
   128  	err = o.clientset.InitClient.InitDevfile(initFlags, o.contextDir,
   129  		func(interactiveMode bool) {
   130  			scontext.SetInteractive(cmdline.Context(), interactiveMode)
   131  			if interactiveMode {
   132  				fmt.Println("The current directory already contains source code. " +
   133  					"odo will try to autodetect the language and project type in order to select the best suited Devfile for your project.")
   134  			}
   135  		},
   136  		func(newDevfileObj parser.DevfileObj) error {
   137  			return newDevfileObj.WriteYamlDevfile()
   138  		})
   139  	if err != nil {
   140  		return err
   141  	}
   142  
   143  	o.variables, err = vars.GetVariables(o.clientset.FS, o.varFileFlag, o.varsFlag, os.LookupEnv)
   144  	if err != nil {
   145  		return err
   146  	}
   147  
   148  	o.Context, err = genericclioptions.New(genericclioptions.NewCreateParameters(cmdline).NeedDevfile("").WithVariables(o.variables))
   149  	if err != nil {
   150  		return fmt.Errorf("unable to create context: %v", err)
   151  	}
   152  
   153  	o.initialDevfileObj = o.Context.EnvSpecificInfo.GetDevfileObj()
   154  
   155  	// ENV.YAML
   156  	//
   157  	// Set the appropriate variables in the env.yaml file
   158  	//
   159  	// TODO: Eventually refactor with code in deploy/deploy.go
   160  	envfileinfo, err := envinfo.NewEnvSpecificInfo("")
   161  	if err != nil {
   162  		return fmt.Errorf("unable to retrieve configuration information: %v", err)
   163  	}
   164  
   165  	// If the env.yaml does not exist, we will save the project name
   166  	if !envfileinfo.Exists() {
   167  		err = envfileinfo.SetComponentSettings(envinfo.ComponentSettings{Project: o.GetProject()})
   168  		if err != nil {
   169  			return fmt.Errorf("failed to write new env.yaml file: %w", err)
   170  		}
   171  	} else if envfileinfo.Exists() && envfileinfo.GetComponentSettings().Project != o.GetProject() {
   172  		// If the env.yaml exists and the project is set incorrectly, we'll override it.
   173  		klog.V(4).Info("Overriding project name in env.yaml as it's set incorrectly, new project name: ", o.GetProject())
   174  		err = envfileinfo.SetConfiguration("project", o.GetProject())
   175  		if err != nil {
   176  			return fmt.Errorf("failed to update project in env.yaml file: %w", err)
   177  		}
   178  	}
   179  
   180  	// END ENV.YAML
   181  
   182  	o.clientset.KubernetesClient.SetNamespace(o.GetProject())
   183  
   184  	return nil
   185  }

   187  func (o *DevOptions) Validate() error {
   188  	if !o.debugFlag && !libdevfile.HasRunCommand(o.initialDevfileObj.Data) {
   189  		return clierrors.NewNoCommandInDevfileError("run")
   190  	}
   191  	if o.debugFlag && !libdevfile.HasDebugCommand(o.initialDevfileObj.Data) {
   192  		return clierrors.NewNoCommandInDevfileError("debug")
   193  	}
   194  	return nil
   195  }

   197  func (o *DevOptions) Run(ctx context.Context) (err error) {
   198  	var (
   199  		devFileObj  = o.Context.EnvSpecificInfo.GetDevfileObj()
   200  		path        = filepath.Dir(o.Context.EnvSpecificInfo.GetDevfilePath())
   201  		devfileName = devFileObj.GetMetadataName()
   202  		namespace   = o.GetProject()
   203  	)
   204  
   205  	// Output what the command is doing / information
   206  	log.Title("Developing using the "+devfileName+" Devfile",
   207  		"Namespace: "+namespace,
   208  		"odo version: "+version.VERSION)
   209  
   210  	// check for .gitignore file and add odo-file-index.json to .gitignore
   211  	gitIgnoreFile, err := util.TouchGitIgnoreFile(path)
   212  	if err != nil {
   213  		return err
   214  	}
   215  
   216  	// add odo-file-index.json path to .gitignore
   217  	err = util.AddOdoFileIndex(gitIgnoreFile)
   218  	if err != nil {
   219  		return err
   220  	}
   221  
   222  	// add devstate.json path to .gitignore
   223  	err = util.AddOdoDevState(gitIgnoreFile)
   224  	if err != nil {
   225  		return err
   226  	}
   227  
   228  	var ignores []string
   229  	err = genericclioptions.ApplyIgnore(&ignores, "")
   230  	if err != nil {
   231  		return err
   232  	}
   233  	// Ignore the devfile, as it will be handled independently
   234  	o.ignorePaths = ignores
   235  
   236  	log.Section("Deploying to the cluster in developer mode")
   237  	componentStatus, err := o.clientset.DevClient.Start(devFileObj, namespace, o.ignorePaths, path, o.debugFlag, o.buildCommandFlag, o.runCommandFlag, o.randomPortsFlag, o.errOut, o.clientset.FS)
   238  	if err != nil {
   239  		return err
   240  	}
   241  
   242  	scontext.SetComponentType(ctx, component.GetComponentTypeFromDevfileMetadata(devFileObj.Data.GetMetadata()))
   243  	scontext.SetLanguage(ctx, devFileObj.Data.GetMetadata().Language)
   244  	scontext.SetProjectType(ctx, devFileObj.Data.GetMetadata().ProjectType)
   245  	scontext.SetDevfileName(ctx, devFileObj.GetMetadataName())
   246  
   247  	d := Handler{
   248  		clientset:   *o.clientset,
   249  		randomPorts: o.randomPortsFlag,
   250  		errOut:      o.errOut,
   251  	}
   252  	err = o.clientset.DevClient.Watch(
   253  		o.GetDevfilePath(),
   254  		devFileObj,
   255  		path,
   256  		o.ignorePaths,
   257  		o.out,
   258  		&d,
   259  		o.ctx,
   260  		o.debugFlag,
   261  		o.buildCommandFlag,
   262  		o.runCommandFlag,
   263  		o.variables,
   264  		o.randomPortsFlag,
   265  		!o.noWatchFlag,
   266  		o.errOut,
   267  		componentStatus,
   268  	)
   269  	return err
   270  }

   272  // RegenerateAdapterAndPush regenerates the adapter and pushes the files to remote pod
   273  func (o *Handler) RegenerateAdapterAndPush(pushParams adapters.PushParameters, watchParams watch.WatchParameters, componentStatus *watch.ComponentStatus) error {
   274  	var adapter kcomponent.ComponentAdapter
   275  
   276  	adapter, err := o.regenerateComponentAdapterFromWatchParams(watchParams)
   277  	if err != nil {
   278  		return fmt.Errorf("unable to generate component from watch parameters: %w", err)
   279  	}
   280  
   281  	err = adapter.Push(pushParams, componentStatus)
   282  	if err != nil {
   283  		return fmt.Errorf("watch command was unable to push component: %w", err)
   284  	}
   285  
   286  	return nil
   287  }

   289  func (o *Handler) regenerateComponentAdapterFromWatchParams(parameters watch.WatchParameters) (kcomponent.ComponentAdapter, error) {
   290  	devObj, err := ododevfile.ParseAndValidateFromFileWithVariables(location.DevfileLocation(""), parameters.Variables)
   291  	if err != nil {
   292  		return nil, err
   293  	}
   294  
   295  	return kcomponent.NewKubernetesAdapter(
   296  		o.clientset.KubernetesClient,
   297  		o.clientset.PreferenceClient,
   298  		o.clientset.PortForwardClient,
   299  		o.clientset.BindingClient,
   300  		kcomponent.AdapterContext{
   301  			ComponentName: parameters.ComponentName,
   302  			Context:       parameters.Path,
   303  			AppName:       parameters.ApplicationName,
   304  			Devfile:       devObj,
   305  			FS:            o.clientset.FS,
   306  		},
   307  		parameters.EnvSpecificInfo.GetNamespace(),
   308  	), nil
   309  }

   326  func NewCmdDev(name, fullName string) *cobra.Command {
   327  	o := NewDevOptions()
   328  	devCmd := &cobra.Command{
   329  		Use:   name,
   330  		Short: "Deploy component to development cluster",
   331  		Long: `odo dev is a long running command that will automatically sync your source to the cluster.
   332  It forwards endpoints with exposure values 'public' or 'internal' to a port on localhost.`,
   333  		Example: fmt.Sprintf(devExample, fullName),
   334  		Args:    cobra.MaximumNArgs(0),
   335  		Run: func(cmd *cobra.Command, args []string) {
   336  			genericclioptions.GenericRun(o, cmd, args)
   337  		},
   338  	}
   339  	devCmd.Flags().BoolVar(&o.noWatchFlag, "no-watch", false, "Do not watch for file changes")
   340  	devCmd.Flags().BoolVar(&o.randomPortsFlag, "random-ports", false, "Assign random ports to redirected ports")
   341  	devCmd.Flags().BoolVar(&o.debugFlag, "debug", false, "Execute the debug command within the component")
   342  	devCmd.Flags().StringArrayVar(&o.varsFlag, "var", []string{}, "Variable to override Devfile variable and variables in var-file")
   343  	devCmd.Flags().StringVar(&o.varFileFlag, "var-file", "", "File containing variables to override Devfile variables")
   344  	devCmd.Flags().StringVar(&o.buildCommandFlag, "build-command", "",
   345  		"Alternative build command. The default one will be used if this flag is not set.")
   346  	devCmd.Flags().StringVar(&o.runCommandFlag, "run-command", "",
   347  		"Alternative run command to execute. The default one will be used if this flag is not set.")
   348  	clientset.Add(devCmd,
   349  		clientset.BINDING,
   350  		clientset.DEV,
   351  		clientset.FILESYSTEM,
   352  		clientset.INIT,
   353  		clientset.KUBERNETES,
   354  		clientset.PORT_FORWARD,
   355  		clientset.PREFERENCE,
   356  		clientset.WATCH,
   357  	)
   358  	// Add a defined annotation in order to appear in the help menu
   359  	devCmd.Annotations["command"] = "main"
   360  	devCmd.SetUsageTemplate(odoutil.CmdUsageTemplate)
   361  
   362  	return devCmd
   363  }
