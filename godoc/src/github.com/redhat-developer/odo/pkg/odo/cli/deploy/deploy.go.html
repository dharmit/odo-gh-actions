    67  // Complete DeployOptions after they've been created
    68  func (o *DeployOptions) Complete(cmdline cmdline.Cmdline, args []string) (err error) {
    69  	o.contextDir, err = os.Getwd()
    70  	if err != nil {
    71  		return err
    72  	}
    73  	isEmptyDir, err := location.DirIsEmpty(o.clientset.FS, o.contextDir)
    74  	if err != nil {
    75  		return err
    76  	}
    77  	if isEmptyDir {
    78  		return errors.New("this command cannot run in an empty directory, run the command in a directory containing source code or initialize using 'odo init'")
    79  	}
    80  
    81  	initFlags := o.clientset.InitClient.GetFlags(cmdline.GetFlags())
    82  
    83  	err = o.clientset.InitClient.InitDevfile(initFlags, o.contextDir,
    84  		func(interactiveMode bool) {
    85  			scontext.SetInteractive(cmdline.Context(), interactiveMode)
    86  			if interactiveMode {
    87  				fmt.Println("The current directory already contains source code. " +
    88  					"odo will try to autodetect the language and project type in order to select the best suited Devfile for your project.")
    89  			}
    90  		},
    91  		func(newDevfileObj parser.DevfileObj) error {
    92  			return newDevfileObj.WriteYamlDevfile()
    93  		})
    94  	if err != nil {
    95  		return err
    96  	}
    97  
    98  	o.variables, err = vars.GetVariables(o.clientset.FS, o.varFileFlag, o.varsFlag, os.LookupEnv)
    99  	if err != nil {
   100  		return err
   101  	}
   102  
   103  	o.Context, err = genericclioptions.New(genericclioptions.NewCreateParameters(cmdline).NeedDevfile(o.contextDir).WithVariables(o.variables).CreateAppIfNeeded())
   104  	if err != nil {
   105  		return err
   106  	}
   107  
   108  	// ENV.YAML
   109  	//
   110  	// Set the appropriate variables in the env.yaml file
   111  	//
   112  	// TODO: Eventually refactor with code in dev/dev.go
   113  
   114  	envfileinfo, err := envinfo.NewEnvSpecificInfo("")
   115  	if err != nil {
   116  		return fmt.Errorf("unable to retrieve configuration information: %w", err)
   117  	}
   118  
   119  	// If the env.yaml does not exist, we will save the project name
   120  	if !envfileinfo.Exists() {
   121  		err = envfileinfo.SetComponentSettings(envinfo.ComponentSettings{Project: o.GetProject()})
   122  		if err != nil {
   123  			return fmt.Errorf("failed to write new env.yaml file: %w", err)
   124  		}
   125  	} else if envfileinfo.Exists() && envfileinfo.GetComponentSettings().Project != o.GetProject() {
   126  		// If the env.yaml exists and the project is set incorrectly, we'll override it.
   127  		klog.V(4).Info("Overriding project name in env.yaml as it's set incorrectly, new project name: ", o.GetProject())
   128  		err = envfileinfo.SetConfiguration("project", o.GetProject())
   129  		if err != nil {
   130  			return fmt.Errorf("failed to update project in env.yaml file: %w", err)
   131  		}
   132  	}
   133  
   134  	// END ENV.YAML
   135  
   136  	// this ensures that odo deploy uses the namespace set in env.yaml
   137  	o.clientset.KubernetesClient.SetNamespace(o.GetProject())
   138  	return
   139  }

   146  // Run contains the logic for the odo command
   147  func (o *DeployOptions) Run(ctx context.Context) error {
   148  	devfileObj := o.EnvSpecificInfo.GetDevfileObj()
   149  	devfileName := devfileObj.GetMetadataName()
   150  	path := filepath.Dir(o.EnvSpecificInfo.GetDevfilePath())
   151  	appName := o.GetApplication()
   152  	namespace := o.GetProject()
   153  	scontext.SetComponentType(ctx, component.GetComponentTypeFromDevfileMetadata(devfileObj.Data.GetMetadata()))
   154  	scontext.SetLanguage(ctx, devfileObj.Data.GetMetadata().Language)
   155  	scontext.SetProjectType(ctx, devfileObj.Data.GetMetadata().ProjectType)
   156  	scontext.SetDevfileName(ctx, devfileName)
   157  	// Output what the command is doing / information
   158  	log.Title("Deploying the application using "+devfileName+" Devfile",
   159  		"Namespace: "+namespace,
   160  		"odo version: "+version.VERSION)
   161  
   162  	// Run actual deploy command to be used
   163  	err := o.clientset.DeployClient.Deploy(o.clientset.FS, devfileObj, path, appName)
   164  
   165  	if err == nil {
   166  		log.Info("\nYour Devfile has been successfully deployed")
   167  	}
   168  
   169  	return err
   170  }
