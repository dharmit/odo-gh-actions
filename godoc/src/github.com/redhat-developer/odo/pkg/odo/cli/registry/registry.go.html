    78  func (o *ListOptions) Validate() error {
    79  	if o.devfileList.DevfileRegistries == nil {
    80  		if len(o.registryFlag) > 0 {
    81  			return fmt.Errorf("the registry %q is not in preferences", o.registryFlag)

    83  		return fmt.Errorf("no registry in preferences, please add a registry using 'odo preference add registry' command")

    85  
    86  	if len(o.devfileList.Items) == 0 {
    87  		return fmt.Errorf("no deployable components found")

   121  	listCmd.Flags().StringVar(&o.registryFlag, "devfile-registry", "", "Only show components from the specific Devfile registry")
   122  	listCmd.Flags().BoolVar(&o.detailsFlag, "details", false, "Show details of each component")
   123  
   124  	// Add a defined annotation in order to appear in the help menu
   125  	listCmd.Annotations["command"] = "main"
   126  	listCmd.SetUsageTemplate(odoutil.CmdUsageTemplate)

   132  func (o *ListOptions) printDevfileList(DevfileList []api.DevfileStack) {
   133  
   134  	// Create the table and use our own style
   135  	t := table.NewWriter()
   136  
   137  	t.SetStyle(table.Style{
   138  		Box: table.BoxStyle{
   139  			PaddingLeft:  " ",
   140  			PaddingRight: " ",
   141  		},
   142  		Color: table.ColorOptions{
   143  			Header: text.Colors{text.FgHiGreen, text.Underline},
   144  		},
   145  		Format: table.FormatOptions{
   146  			Footer: text.FormatUpper,
   147  			Header: text.FormatUpper,
   148  			Row:    text.FormatDefault,
   149  		},
   150  		Options: table.Options{
   151  			DrawBorder:      false,
   152  			SeparateColumns: false,
   153  			SeparateFooter:  false,
   154  			SeparateHeader:  false,
   155  			SeparateRows:    false,
   156  		},
   157  	})
   158  	t.SetOutputMirror(log.GetStdout())
   159  
   160  	t.AppendHeader(table.Row{"NAME", "REGISTRY", "DESCRIPTION"})
   161  
   162  	for _, devfileComponent := range DevfileList {
   163  		// Mark the name as yellow in the index so it's easier to see.
   164  		name := text.Colors{text.FgHiYellow}.Sprint(devfileComponent.Name)
   165  
   166  		if o.detailsFlag {
   167  
   168  			// Output the details of the component
   169  			fmt.Printf(`%s: %s
   170  %s: %s
   171  %s: %s
   172  %s: %s
   173  %s: %s
   174  %s: %s 
   175  %s: %s
   176  %s: %s
   177  %s: %s
   178  %s:
   179    - %s
   180  %s:
   181    - Dev: %s
   182    - Deploy: %s
   183    - Debug: %s
   184  %s`,
   185  				log.Sbold("Name"), name,
   186  				log.Sbold("Display Name"), devfileComponent.DisplayName,
   187  				log.Sbold("Registry"), devfileComponent.Registry.Name,
   188  				log.Sbold("Registry URL"), devfileComponent.Registry.URL,
   189  				log.Sbold("Version"), devfileComponent.Version,
   190  				log.Sbold("Description"), devfileComponent.Description,
   191  				log.Sbold("Tags"), strings.Join(devfileComponent.Tags[:], ", "),
   192  				log.Sbold("Project Type"), devfileComponent.ProjectType,
   193  				log.Sbold("Language"), devfileComponent.Language,
   194  				log.Sbold("Starter Projects"), strings.Join(devfileComponent.StarterProjects, "\n  - "),
   195  				log.Sbold("Supported odo Features"),
   196  				boolToYesNo(devfileComponent.DevfileData.SupportedOdoFeatures.Dev),
   197  				boolToYesNo(devfileComponent.DevfileData.SupportedOdoFeatures.Deploy),
   198  				boolToYesNo(devfileComponent.DevfileData.SupportedOdoFeatures.Debug),
   199  				"\n")
   200  		} else {
   201  			// Create a simplified row only showing the name, registry and description.
   202  			t.AppendRow(table.Row{name, devfileComponent.Registry.Name, util.TruncateString(devfileComponent.Description, 40, "...")})
   203  		}
   204  
   205  	}
   206  
   207  	// Exit with an error if there are no components to show, so we don't render the table / continue
   208  	if len(DevfileList) == 0 {
   209  		log.Error("There are no Devfiles available to show")
   210  		return
   211  	}
   212  
   213  	// Render the table
   214  	if !o.detailsFlag {
   215  		t.Render()
   216  	}
   217  
   218  }
