    84  func (o *InitOptions) Complete(cmdline cmdline.Cmdline, args []string) (err error) {
    85  
    86  	o.ctx = cmdline.Context()
    87  
    88  	o.contextDir, err = o.clientset.FS.Getwd()

    90  		return err

    96  
    97  	return nil

   100  // Validate validates the InitOptions based on completed values
   101  func (o *InitOptions) Validate() error {
   102  
   103  	devfilePresent, err := location.DirectoryContainsDevfile(o.clientset.FS, o.contextDir)
   104  	if err != nil {
   105  		return err
   106  	}
   107  	if devfilePresent {
   108  		return errors.New("a devfile already exists in the current directory")
   109  	}
   110  
   111  	err = o.clientset.InitClient.Validate(o.flags, o.clientset.FS, o.contextDir)
   112  	if err != nil {
   113  		return err
   114  	}
   115  
   116  	if len(o.flags) == 0 && log.IsJSON() {
   117  		return errors.New("parameters are expected to select a devfile")
   118  	}
   119  	return nil
   120  }

   158  // run downloads the devfile and starter project and returns the content and the path of the devfile
   159  func (o *InitOptions) run(ctx context.Context) (devfileObj parser.DevfileObj, path string, err error) {
   160  	var starterDownloaded bool
   161  
   162  	defer func() {
   163  		if err == nil {
   164  			return
   165  		}
   166  		if starterDownloaded {
   167  			err = fmt.Errorf("%w\nthe command failed after downloading the starter project. By security, the directory is not cleaned up", err)
   168  		} else {
   169  			_ = o.clientset.FS.Remove("devfile.yaml")
   170  			err = fmt.Errorf("%w\nthe command failed, the devfile has been removed from current directory", err)
   171  		}
   172  	}()
   173  
   174  	isEmptyDir, err := location.DirIsEmpty(o.clientset.FS, o.contextDir)
   175  	if err != nil {
   176  		return parser.DevfileObj{}, "", err
   177  	}
   178  
   179  	// Show a welcome message for when you initially run `odo init`.
   180  
   181  	var infoOutput string
   182  	if isEmptyDir && len(o.flags) == 0 {
   183  		infoOutput = messages.NoSourceCodeDetected
   184  	} else if len(o.flags) == 0 {
   185  		infoOutput = messages.SourceCodeDetected
   186  	}
   187  	log.Title(messages.InitializingNewComponent, infoOutput, "odo version: "+version.VERSION)
   188  	log.Info("\nInteractive mode enabled, please answer the following questions:")
   189  
   190  	devfileObj, devfilePath, err := o.clientset.InitClient.SelectAndPersonalizeDevfile(o.flags, o.contextDir)
   191  	if err != nil {
   192  		return parser.DevfileObj{}, "", err
   193  	}
   194  
   195  	starterInfo, err := o.clientset.InitClient.SelectStarterProject(devfileObj, o.flags, o.clientset.FS, o.contextDir)
   196  	if err != nil {
   197  		return parser.DevfileObj{}, "", err
   198  	}
   199  
   200  	// Set the name in the devfile but do not write it yet to disk,
   201  	// because the starter project downloaded at the end might come bundled with a specific Devfile.
   202  	name, err := o.clientset.InitClient.PersonalizeName(devfileObj, o.flags)
   203  	if err != nil {
   204  		return parser.DevfileObj{}, "", fmt.Errorf("failed to update the devfile's name: %w", err)
   205  	}
   206  
   207  	if starterInfo != nil {
   208  		// WARNING: this will remove all the content of the destination directory, ie the devfile.yaml file
   209  		err = o.clientset.InitClient.DownloadStarterProject(starterInfo, o.contextDir)
   210  		if err != nil {
   211  			return parser.DevfileObj{}, "", fmt.Errorf("unable to download starter project %q: %w", starterInfo.Name, err)
   212  		}
   213  		starterDownloaded = true
   214  
   215  		// in case the starter project contains a devfile, read it again
   216  		if _, err = o.clientset.FS.Stat(devfilePath); err == nil {
   217  			devfileObj, _, err = devfile.ParseDevfileAndValidate(parser.ParserArgs{Path: devfilePath, FlattenedDevfile: pointer.BoolPtr(false)})
   218  			if err != nil {
   219  				return parser.DevfileObj{}, "", err
   220  			}
   221  		}
   222  	}
   223  	// WARNING: SetMetadataName writes the Devfile to disk
   224  	if err = devfileObj.SetMetadataName(name); err != nil {
   225  		return parser.DevfileObj{}, "", err
   226  	}
   227  	scontext.SetComponentType(ctx, component.GetComponentTypeFromDevfileMetadata(devfileObj.Data.GetMetadata()))
   228  	scontext.SetLanguage(ctx, devfileObj.Data.GetMetadata().Language)
   229  	scontext.SetProjectType(ctx, devfileObj.Data.GetMetadata().ProjectType)
   230  	scontext.SetDevfileName(ctx, devfileObj.GetMetadataName())
   231  
   232  	return devfileObj, devfilePath, nil
   233  }
