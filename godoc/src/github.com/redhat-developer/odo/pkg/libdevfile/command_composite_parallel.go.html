    42  // Execute loops over each command and executes them in parallel
    43  func (o *parallelCompositeCommand) Execute(handler Handler) error {
    44  	allCommands, err := allCommandsMap(o.devfileObj)
    45  	if err != nil {
    46  		return err
    47  	}
    48  	commandExecs := util.NewConcurrentTasks(len(o.command.Composite.Commands))
    49  	for _, devfileCmd := range o.command.Composite.Commands {
    50  		cmd, err2 := newCommand(o.devfileObj, allCommands[devfileCmd])
    51  		if err2 != nil {
    52  			return err2
    53  		}
    54  		commandExecs.Add(util.ConcurrentTask{
    55  			ToRun: func(errChannel chan error) {
    56  				err3 := cmd.Execute(handler)
    57  				if err3 != nil {
    58  					errChannel <- err3
    59  				}
    60  			},
    61  		})
    62  	}
    63  	err = commandExecs.Run()
    64  	if err != nil {
    65  		return fmt.Errorf("parallel command execution failed: %w", err)
    66  	}
    67  	return nil
    68  }
