    36  // ListClusterResourcesToDelete lists Kubernetes resources from cluster in namespace for a given odo component
    37  // It only returns resources not owned by another resource of the component, letting the garbage collector do its job
    38  func (do *DeleteComponentClient) ListClusterResourcesToDelete(componentName string, namespace string) ([]unstructured.Unstructured, error) {
    39  	var result []unstructured.Unstructured
    40  	selector := odolabels.GetSelector(componentName, "app", odolabels.ComponentAnyMode, false)
    41  	list, err := do.kubeClient.GetAllResourcesFromSelector(selector, namespace)
    42  	if err != nil {
    43  		return nil, err
    44  	}
    45  	for _, resource := range list {
    46  		// If the resource is Terminating, there is no sense in displaying it.
    47  		if resource.GetDeletionTimestamp() != nil {
    48  			continue
    49  		}
    50  		referenced := false
    51  		for _, ownerRef := range resource.GetOwnerReferences() {
    52  			if references(list, ownerRef) {
    53  				referenced = true
    54  				break
    55  			}
    56  		}
    57  		if !referenced {
    58  			result = append(result, resource)
    59  		}
    60  	}
    61  
    62  	return result, nil
    63  }

    65  func (do *DeleteComponentClient) DeleteResources(resources []unstructured.Unstructured, wait bool) []unstructured.Unstructured {
    66  	var failed []unstructured.Unstructured
    67  	for _, resource := range resources {
    68  		gvr, err := do.kubeClient.GetRestMappingFromUnstructured(resource)
    69  		if err != nil {
    70  			failed = append(failed, resource)
    71  			continue
    72  		}
    73  		err = do.kubeClient.DeleteDynamicResource(resource.GetName(), gvr.Resource, wait)
    74  		if err != nil {
    75  			klog.V(3).Infof("failed to delete resource %q (%s.%s.%s): %v", resource.GetName(), gvr.Resource.Group, gvr.Resource.Version, gvr.Resource.Resource, err)
    76  			failed = append(failed, resource)
    77  		}
    78  	}
    79  	return failed
    80  }

    82  // references returns true if ownerRef references a resource in the list
    83  func references(list []unstructured.Unstructured, ownerRef metav1.OwnerReference) bool {
    84  	for _, resource := range list {
    85  		if ownerRef.APIVersion == resource.GetAPIVersion() && ownerRef.Kind == resource.GetKind() && ownerRef.Name == resource.GetName() {
    86  			return true
    87  		}
    88  	}
    89  	return false
    90  }

    92  // ListResourcesToDeleteFromDevfile parses all the devfile components and returns a list of resources that are present on the cluster and can be deleted
    93  func (do DeleteComponentClient) ListResourcesToDeleteFromDevfile(devfileObj parser.DevfileObj, appName string, mode string) (isInnerLoopDeployed bool, resources []unstructured.Unstructured, err error) {
    94  	var deployment *v1.Deployment
    95  	if mode == odolabels.ComponentDevMode || mode == odolabels.ComponentAnyMode {
    96  		// Inner Loop
    97  		// Fetch the deployment of the devfile component
    98  		componentName := devfileObj.GetMetadataName()
    99  		var deploymentName string
   100  		deploymentName, err = util.NamespaceKubernetesObject(componentName, appName)
   101  		if err != nil {
   102  			return isInnerLoopDeployed, resources, fmt.Errorf("failed to get the resource %q name for component %q; cause: %w", kclient.DeploymentKind, componentName, err)
   103  		}
   104  
   105  		deployment, err = do.kubeClient.GetDeploymentByName(deploymentName)
   106  		if err != nil && !kerrors.IsNotFound(err) {
   107  			return isInnerLoopDeployed, resources, err
   108  		}
   109  
   110  		// if the deployment is found on the cluster,
   111  		// then convert it to unstructured.Unstructured object so that it can be appended to resources;
   112  		// else continue to outer loop
   113  		if deployment.Name != "" {
   114  			isInnerLoopDeployed = true
   115  			var unstructuredDeploy unstructured.Unstructured
   116  			unstructuredDeploy, err = kclient.ConvertK8sResourceToUnstructured(deployment)
   117  			if err != nil {
   118  				return isInnerLoopDeployed, resources, fmt.Errorf("failed to parse the resource %q: %q; cause: %w", kclient.DeploymentKind, deploymentName, err)
   119  			}
   120  			resources = append(resources, unstructuredDeploy)
   121  		}
   122  	}
   123  
   124  	// Parse the devfile for K8s resources; these may belong to either innerloop or outerloop
   125  	localResources, err := libdevfile.ListKubernetesComponents(devfileObj, filepath.Dir(devfileObj.Ctx.GetAbsPath()))
   126  	if err != nil {
   127  		return isInnerLoopDeployed, resources, fmt.Errorf("failed to gather resources for deletion: %w", err)
   128  	}
   129  	for _, lr := range localResources {
   130  		var gvr *meta.RESTMapping
   131  		gvr, err = do.kubeClient.GetRestMappingFromUnstructured(lr)
   132  		if err != nil {
   133  			continue
   134  		}
   135  		// Try to fetch the resource from the cluster; if it exists, append it to the resources list
   136  		var cr *unstructured.Unstructured
   137  		cr, err = do.kubeClient.GetDynamicResource(gvr.Resource, lr.GetName())
   138  		// If a specific mode is asked for, then make sure it matches with the cr's mode.
   139  		if err != nil || (mode != odolabels.ComponentAnyMode && odolabels.GetMode(cr.GetLabels()) != mode) {
   140  			if cr != nil {
   141  				klog.V(4).Infof("Ignoring resource: %s/%s; its mode(%s) does not match with the given mode(%s)", gvr.Resource.Resource, lr.GetName(), odolabels.GetMode(cr.GetLabels()), mode)
   142  			} else {
   143  				klog.V(4).Infof("Ignoring resource: %s/%s; it does not exist on the cluster", gvr.Resource.Resource, lr.GetName())
   144  			}
   145  			continue
   146  		}
   147  		resources = append(resources, *cr)
   148  	}
   149  
   150  	return isInnerLoopDeployed, resources, nil
   151  }

   153  // ExecutePreStopEvents executes preStop events if any, as a precondition to deleting a devfile component deployment
   154  func (do *DeleteComponentClient) ExecutePreStopEvents(devfileObj parser.DevfileObj, appName string) error {
   155  	if !libdevfile.HasPreStopEvents(devfileObj) {
   156  		return nil
   157  	}
   158  	componentName := devfileObj.GetMetadataName()
   159  	klog.V(4).Infof("Gathering information for component: %q", componentName)
   160  
   161  	klog.V(3).Infof("Checking component status for %q", componentName)
   162  	selector := odolabels.GetSelector(componentName, appName, odolabels.ComponentDevMode, false)
   163  	pod, err := do.kubeClient.GetRunningPodFromSelector(selector)
   164  	if err != nil {
   165  		klog.V(1).Info("Component not found on the cluster.")
   166  
   167  		if kerrors.IsForbidden(err) {
   168  			klog.V(3).Infof("Resource for %q forbidden", componentName)
   169  			log.Warningf("You are forbidden from accessing the resource. Please check if you the right permissions and try again.")
   170  			return nil
   171  		}
   172  
   173  		if e, ok := err.(*kclient.PodNotFoundError); ok {
   174  			klog.V(3).Infof("Resource for %q not found; cause: %v", componentName, e)
   175  			log.Warningf("Resources not found on the cluster. Run `odo delete component -v <DEBUG_LEVEL_0-9>` to know more.")
   176  			return nil
   177  		}
   178  
   179  		return fmt.Errorf("unable to determine if component %s exists; cause: %v", componentName, err.Error())
   180  	}
   181  
   182  	// do not fail Delete operation if if the pod is not running or if the event execution fails
   183  	if pod.Status.Phase != corev1.PodRunning {
   184  		klog.V(4).Infof("unable to execute preStop events, pod for component %q is not running", componentName)
   185  		return nil
   186  	}
   187  
   188  	klog.V(4).Infof("Executing %q event commands for component %q", libdevfile.PreStop, componentName)
   189  	// ignore the failures if any; delete should not fail because preStop events failed to execute
   190  	err = libdevfile.ExecPreStopEvents(devfileObj, component.NewExecHandler(do.kubeClient, appName, componentName, pod.Name, "", false))
   191  	if err != nil {
   192  		klog.V(4).Infof("Failed to execute %q event commands for component %q, cause: %v", libdevfile.PreStop, componentName, err.Error())
   193  	}
   194  
   195  	return nil
   196  }
