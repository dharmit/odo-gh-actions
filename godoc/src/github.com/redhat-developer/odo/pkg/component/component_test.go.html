   194  func getUnstructured(name, kind, apiVersion, managed, managedByVersion, componentType, namespace string) (u unstructured.Unstructured) {
   195  	u.SetName(name)
   196  	u.SetKind(kind)
   197  	u.SetAPIVersion(apiVersion)
   198  	u.SetNamespace(namespace)
   199  	u.SetLabels(labels.Builder().
   200  		WithComponentName(name).
   201  		WithManager(managed).
   202  		WithManagedByVersion(managedByVersion).
   203  		Labels())
   204  	u.SetAnnotations(labels.Builder().
   205  		WithProjectType(componentType).
   206  		Labels())
   207  	return
   208  }
