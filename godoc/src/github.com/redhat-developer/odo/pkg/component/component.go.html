    45  // GatherName parses the Devfile and retrieves an appropriate name in two ways.
    46  // 1. If metadata.name exists, we use it
    47  // 2. If metadata.name does NOT exist, we use the folder name where the devfile.yaml is located
    48  func GatherName(devObj parser.DevfileObj, devfilePath string) (string, error) {
    49  
    50  	metadata := devObj.Data.GetMetadata()
    51  
    52  	klog.V(4).Infof("metadata.Name: %s", metadata.Name)
    53  
    54  	// 1. Use metadata.name if it exists
    55  	if metadata.Name != "" {
    56  
    57  		// Remove any suffix's that end with `-`. This is because many Devfile's use the original v1 Devfile pattern of
    58  		// having names such as "foo-bar-" in order to prepend container names such as "foo-bar-container1"
    59  		return strings.TrimSuffix(metadata.Name, "-"), nil
    60  	}
    61  
    62  	// 2. Use the folder name as a last resort if nothing else exists
    63  	sourcePath, err := dfutil.GetAbsPath(devfilePath)
    64  	if err != nil {
    65  		return "", fmt.Errorf("unable to get source path: %w", err)
    66  	}
    67  	klog.V(4).Infof("Source path: %s", sourcePath)
    68  	klog.V(4).Infof("devfile dir: %s", filepath.Dir(sourcePath))
    69  
    70  	return filepath.Base(filepath.Dir(sourcePath)), nil
    71  }

   111  // ListAllClusterComponents returns a list of all "components" on a cluster
   112  // that are both odo and non-odo components.
   113  //
   114  // We then return a list of "components" intended for listing / output purposes specifically for commands such as:
   115  // `odo list`
   116  // that are both odo and non-odo components.
   117  func ListAllClusterComponents(client kclient.ClientInterface, namespace string) ([]api.ComponentAbstract, error) {
   118  
   119  	// Get all the dynamic resources available
   120  	resourceList, err := client.GetAllResourcesFromSelector("", namespace)
   121  	if err != nil {
   122  		return nil, fmt.Errorf("unable to list all dynamic resources required to find components: %w", err)
   123  	}
   124  
   125  	var components []api.ComponentAbstract
   126  
   127  	for _, resource := range resourceList {
   128  
   129  		// ignore "PackageManifest" as they are not components, it is just a record in OpenShift catalog.
   130  		if resource.GetKind() == "PackageManifest" {
   131  			continue
   132  		}
   133  
   134  		var labels, annotations map[string]string
   135  
   136  		// Retrieve the labels and annotations from the unstructured resource output
   137  		if resource.GetLabels() != nil {
   138  			labels = resource.GetLabels()
   139  		}
   140  		if resource.GetAnnotations() != nil {
   141  			annotations = resource.GetAnnotations()
   142  		}
   143  
   144  		// Figure out the correct name to use
   145  		// if there is no instance label (app.kubernetes.io/instance),
   146  		// we SKIP the resource as it is not a component essential for Kubernetes.
   147  		name := odolabels.GetComponentName(labels)
   148  		if name == "" {
   149  			continue
   150  		}
   151  
   152  		// Get the component type (if there is any..)
   153  		componentType, err := odolabels.GetProjectType(nil, annotations)
   154  		if err != nil || componentType == "" {
   155  			componentType = api.TypeUnknown
   156  		}
   157  
   158  		// Get the managedBy label
   159  		// IMPORTANT. If both "managed-by" and "instance" labels are BLANK, it is most likely an operator
   160  		// or a non-component. We do not want to show these in the list of components
   161  		// so we skip them if there is no "managed-by" label.
   162  
   163  		managedBy := odolabels.GetManagedBy(labels)
   164  		if managedBy == "" && name == "" {
   165  			continue
   166  		}
   167  
   168  		managedByVersion := odolabels.GetManagedByVersion(labels)
   169  
   170  		// Generate the appropriate "component" with all necessary information
   171  		component := api.ComponentAbstract{
   172  			Name:             name,
   173  			ManagedBy:        managedBy,
   174  			Type:             componentType,
   175  			ManagedByVersion: managedByVersion,
   176  		}
   177  		mode := odolabels.GetMode(labels)
   178  		componentFound := false
   179  		for v, otherCompo := range components {
   180  			if component.Name == otherCompo.Name {
   181  				componentFound = true
   182  				if mode != "" {
   183  					modeFound := false
   184  					for _, m := range components[v].RunningIn {
   185  						if m == api.RunningMode(mode) {
   186  							modeFound = true
   187  							break
   188  						}
   189  					}
   190  					if !modeFound {
   191  						components[v].RunningIn = append(components[v].RunningIn, api.RunningMode(mode))
   192  						sort.Sort(components[v].RunningIn)
   193  					}
   194  				}
   195  				if otherCompo.Type == api.TypeUnknown && component.Type != api.TypeUnknown {
   196  					components[v].Type = component.Type
   197  				}
   198  				if otherCompo.ManagedBy == api.TypeUnknown && component.ManagedBy != api.TypeUnknown {
   199  					components[v].ManagedBy = component.ManagedBy
   200  				}
   201  			}
   202  		}
   203  		if !componentFound {
   204  			if mode != "" {
   205  				component.RunningIn = []api.RunningMode{api.RunningMode(mode)}
   206  			}
   207  			components = append(components, component)
   208  		}
   209  	}
   210  
   211  	return components, nil
   212  }

   214  func getResourcesForComponent(client kclient.ClientInterface, name string, namespace string) ([]unstructured.Unstructured, error) {
   215  	selector := labels.GetSelector(name, "app", labels.ComponentAnyMode, false)
   216  	resourceList, err := client.GetAllResourcesFromSelector(selector, namespace)
   217  	if err != nil {
   218  		return nil, err
   219  	}
   220  	filteredList := []unstructured.Unstructured{}
   221  	for _, resource := range resourceList {
   222  		// ignore "PackageManifest" as they are not components, it is just a record in OpenShift catalog.
   223  		if resource.GetKind() == "PackageManifest" {
   224  			continue
   225  		}
   226  		filteredList = append(filteredList, resource)
   227  	}
   228  	return filteredList, nil
   229  }

   231  // GetRunningModes returns the list of modes on which a "name" component is deployed, by looking into namespace
   232  // the resources deployed with matching labels, based on the "odo.dev/mode" label
   233  func GetRunningModes(client kclient.ClientInterface, name string) ([]api.RunningMode, error) {
   234  	list, err := getResourcesForComponent(client, name, client.GetCurrentNamespace())
   235  	if err != nil {
   236  		return []api.RunningMode{api.RunningModeUnknown}, nil
   237  	}
   238  
   239  	if len(list) == 0 {
   240  		return nil, NewNoComponentFoundError(name, client.GetCurrentNamespace())
   241  	}
   242  
   243  	mapResult := map[string]bool{}
   244  	for _, resource := range list {
   245  		resourceLabels := resource.GetLabels()
   246  		mode := labels.GetMode(resourceLabels)
   247  		if mode != "" {
   248  			mapResult[mode] = true
   249  		}
   250  	}
   251  	keys := make(api.RunningModeList, 0, len(mapResult))
   252  	for k := range mapResult {
   253  		keys = append(keys, api.RunningMode(k))
   254  	}
   255  	sort.Sort(keys)
   256  	return keys, nil
   257  }

   270  // GetDevfileInfoFromCluster extracts information from the labels and annotations of resources to rebuild a Devfile
   271  func GetDevfileInfoFromCluster(client kclient.ClientInterface, name string) (parser.DevfileObj, error) {
   272  	list, err := getResourcesForComponent(client, name, client.GetCurrentNamespace())
   273  	if err != nil {
   274  		return parser.DevfileObj{}, nil
   275  	}
   276  
   277  	if len(list) == 0 {
   278  		return parser.DevfileObj{}, nil
   279  	}
   280  
   281  	devfileData, err := data.NewDevfileData(string(data.APISchemaVersion200))
   282  	if err != nil {
   283  		return parser.DevfileObj{}, err
   284  	}
   285  	metadata := devfileData.GetMetadata()
   286  	metadata.Name = UnknownValue
   287  	metadata.DisplayName = UnknownValue
   288  	metadata.ProjectType = UnknownValue
   289  	metadata.Language = UnknownValue
   290  	metadata.Version = UnknownValue
   291  	metadata.Description = UnknownValue
   292  
   293  	for _, resource := range list {
   294  		labels := resource.GetLabels()
   295  		annotations := resource.GetAnnotations()
   296  		name := odolabels.GetComponentName(labels)
   297  		if len(name) > 0 && metadata.Name == UnknownValue {
   298  			metadata.Name = name
   299  		}
   300  		typ, err := odolabels.GetProjectType(labels, annotations)
   301  		if err != nil {
   302  			continue
   303  		}
   304  		if len(typ) > 0 && metadata.ProjectType == UnknownValue {
   305  			metadata.ProjectType = typ
   306  		}
   307  	}
   308  	devfileData.SetMetadata(metadata)
   309  	return parser.DevfileObj{
   310  		Data: devfileData,
   311  	}, nil
   312  }
