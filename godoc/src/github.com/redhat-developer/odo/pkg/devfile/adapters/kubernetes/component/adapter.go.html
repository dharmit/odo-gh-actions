    95  func (a Adapter) Push(parameters adapters.PushParameters, componentStatus *watch.ComponentStatus) (err error) {

    98  	err = dfutil.ValidateK8sResourceName("component name", a.ComponentName)
    99  	if err != nil {
   100  		return err
   101  	}
   102  
   103  	err = dfutil.ValidateK8sResourceName("component namespace", parameters.EnvSpecificInfo.GetNamespace())

   105  		return err
   106  	}
   107  
   108  	deployment, componentExists, err := a.getComponentDeployment()
   109  	if err != nil {
   110  		return err
   111  	}
   112  
   113  	if componentStatus.State != watch.StateWaitDeployment && componentStatus.State != watch.StateReady {
   114  		log.SpinnerNoSpin("Waiting for Kubernetes resources")
   115  	}
   116  
   117  	// Set the mode to Dev since we are using "odo dev" here
   118  	labels := odolabels.GetLabels(a.ComponentName, a.AppName, odolabels.ComponentDevMode, false)
   119  
   120  	k8sComponents, err := a.pushDevfileKubernetesComponents(labels)
   121  	if err != nil {
   122  		return err
   123  	}
   124  
   125  	var updated bool
   126  	deployment, updated, err = a.createOrUpdateComponent(componentExists, parameters.EnvSpecificInfo, libdevfile.DevfileCommands{
   127  		BuildCmd: parameters.DevfileBuildCmd,
   128  		RunCmd:   parameters.DevfileRunCmd,
   129  		DebugCmd: parameters.DevfileDebugCmd,
   130  	}, parameters.DebugPort, deployment)
   131  	if err != nil {
   132  		return fmt.Errorf("unable to create or update component: %w", err)
   133  	}
   134  
   135  	ownerReference := generator.GetOwnerReference(deployment)
   136  	err = a.updatePVCsOwnerReferences(ownerReference)
   137  	if err != nil {
   138  		return err
   139  	}
   140  
   141  	// Update all services with owner references
   142  	err = a.kubeClient.TryWithBlockOwnerDeletion(ownerReference, func(ownerRef metav1.OwnerReference) error {
   143  		return service.UpdateServicesWithOwnerReferences(a.kubeClient, a.Devfile, k8sComponents, ownerRef, a.Context)
   144  	})
   145  	if err != nil {
   146  		return err
   147  	}
   148  
   149  	// create the Kubernetes objects from the manifest and delete the ones not in the devfile
   150  	err = service.PushLinks(a.kubeClient, a.Devfile, k8sComponents, labels, deployment, a.Context)
   151  	if err != nil {
   152  		return fmt.Errorf("failed to create service bindings associated with the component: %w", err)
   153  	}
   154  
   155  	if updated {
   156  		klog.V(4).Infof("Deployment has been updated to generation %d. Waiting new event...\n", deployment.GetGeneration())
   157  		componentStatus.State = watch.StateWaitDeployment
   158  		return nil
   159  	}
   160  
   161  	numberReplicas := deployment.Status.ReadyReplicas
   162  	if numberReplicas != 1 {
   163  		klog.V(4).Infof("Deployment has %d ready replicas. Waiting new event...\n", numberReplicas)
   164  		componentStatus.State = watch.StateWaitDeployment
   165  		return nil
   166  	}
   167  
   168  	injected, err := a.bindingClient.CheckServiceBindingsInjectionDone(a.ComponentName, a.AppName)
   169  	if err != nil {
   170  		return err
   171  	}
   172  
   173  	if !injected {
   174  		klog.V(4).Infof("Waiting for all service bindings to be injected...\n")
   175  		return errors.New("some servicebindings are not injected")
   176  	}
   177  
   178  	// Check if endpoints changed in Devfile
   179  	portsToForward, err := a.portForwardClient.GetPortsToForward(a.Devfile)
   180  	if err != nil {
   181  		return err
   182  	}
   183  	portsChanged := !reflect.DeepEqual(portsToForward, a.portForwardClient.GetForwardedPorts())
   184  
   185  	if componentStatus.State == watch.StateReady && !portsChanged {
   186  		// If the deployment is already in Ready State, no need to continue
   187  		return nil
   188  	}
   189  
   190  	// Now the Deployment has a Ready replica, we can get the Pod to work inside it
   191  	pod, err := a.kubeClient.GetPodUsingComponentName(a.ComponentName)
   192  	if err != nil {
   193  		return fmt.Errorf("unable to get pod for component %s: %w", a.ComponentName, err)
   194  	}
   195  	parameters.EnvSpecificInfo.SetDevfileObj(a.Devfile)
   196  
   197  	// Find at least one pod with the source volume mounted, error out if none can be found
   198  	containerName, syncFolder, err := getFirstContainerWithSourceVolume(pod.Spec.Containers)
   199  	if err != nil {
   200  		return fmt.Errorf("error while retrieving container from pod %s with a mounted project volume: %w", pod.GetName(), err)
   201  	}
   202  
   203  	s := log.Spinner("Syncing files into the container")
   204  	defer s.End(false)
   205  
   206  	// Get commands
   207  	pushDevfileCommands, err := a.getPushDevfileCommands(parameters)
   208  	if err != nil {
   209  		return fmt.Errorf("failed to validate devfile build and run commands: %w", err)
   210  	}
   211  
   212  	podChanged := componentStatus.State == watch.StateWaitDeployment
   213  
   214  	// Get a sync adapter. Check if project files have changed and sync accordingly
   215  	syncAdapter := sync.New(&a, a.kubeClient, a.ComponentName)
   216  	compInfo := adapters.ComponentInfo{
   217  		ContainerName: containerName,
   218  		PodName:       pod.GetName(),
   219  		SyncFolder:    syncFolder,
   220  	}
   221  	syncParams := adapters.SyncParameters{
   222  		PushParams:      parameters,
   223  		CompInfo:        compInfo,
   224  		ComponentExists: componentExists,
   225  		PodChanged:      podChanged,
   226  		Files:           getSyncFilesFromAttributes(pushDevfileCommands),
   227  	}
   228  
   229  	execRequired, err := syncAdapter.SyncFiles(syncParams)
   230  	if err != nil {
   231  		componentStatus.State = watch.StateReady
   232  		return fmt.Errorf("failed to sync to component with name %s: %w", a.ComponentName, err)
   233  	}
   234  	s.End(true)
   235  
   236  	// PostStart events from the devfile will only be executed when the component
   237  	// didn't previously exist
   238  	if !componentStatus.PostStartEventsDone && libdevfile.HasPostStartEvents(a.Devfile) {
   239  		err = libdevfile.ExecPostStartEvents(a.Devfile,
   240  			component.NewExecHandler(a.kubeClient, a.AppName, a.ComponentName, pod.Name, "", parameters.Show))
   241  		if err != nil {
   242  			return err
   243  		}
   244  	}
   245  	componentStatus.PostStartEventsDone = true
   246  
   247  	cmdKind := devfilev1.RunCommandGroupKind
   248  	cmdName := parameters.DevfileRunCmd
   249  	if parameters.Debug {
   250  		cmdKind = devfilev1.DebugCommandGroupKind
   251  		cmdName = parameters.DevfileDebugCmd
   252  	}
   253  
   254  	cmd, err := libdevfile.ValidateAndGetCommand(a.Devfile, cmdName, cmdKind)
   255  	if err != nil {
   256  		return err
   257  	}
   258  
   259  	commandType, err := parsercommon.GetCommandType(cmd)
   260  	if err != nil {
   261  		return err
   262  	}
   263  	var running bool
   264  	var isComposite bool
   265  	cmdHandler := adapterHandler{
   266  		Adapter:         a,
   267  		parameters:      parameters,
   268  		componentExists: componentExists,
   269  		podName:         pod.GetName(),
   270  	}
   271  
   272  	if commandType == devfilev1.ExecCommandType {
   273  		running, err = cmdHandler.isRemoteProcessForCommandRunning(cmd, pod.Name)
   274  		if err != nil {
   275  			return err
   276  		}
   277  	} else if commandType == devfilev1.CompositeCommandType {
   278  		// this handler will run each command in this composite command individually,
   279  		// and will determine whether each command is running or not.
   280  		isComposite = true
   281  	} else {
   282  		return fmt.Errorf("unsupported type %q for Devfile command %s, only exec and composite are handled",
   283  			commandType, cmd.Id)
   284  	}
   285  
   286  	klog.V(4).Infof("running=%v, execRequired=%v",
   287  		running, execRequired)
   288  
   289  	if isComposite || !running || execRequired {
   290  		// Invoke the build command once (before calling libdevfile.ExecuteCommandByNameAndKind), as, if cmd is a composite command,
   291  		// the handler we pass will be called for each command in that composite command.
   292  		doExecuteBuildCommand := func() error {
   293  			execHandler := component.NewExecHandler(a.kubeClient, a.AppName, a.ComponentName, pod.Name,
   294  				"Building your application in container on cluster", parameters.Show)
   295  			return libdevfile.Build(a.Devfile, parameters.DevfileBuildCmd, execHandler)
   296  		}
   297  		if componentExists {
   298  			if cmd.Exec == nil || !util.SafeGetBool(cmd.Exec.HotReloadCapable) {
   299  				if err = doExecuteBuildCommand(); err != nil {
   300  					return err
   301  				}
   302  			}
   303  		} else {
   304  			if err = doExecuteBuildCommand(); err != nil {
   305  				return err
   306  			}
   307  		}
   308  		err = libdevfile.ExecuteCommandByNameAndKind(a.Devfile, cmdName, cmdKind, &cmdHandler, false)
   309  		if err != nil {
   310  			return err
   311  		}
   312  	}
   313  
   314  	if podChanged || portsChanged {
   315  		a.portForwardClient.StopPortForwarding()
   316  	}
   317  
   318  	err = a.portForwardClient.StartPortForwarding(a.Devfile, a.ComponentName, parameters.RandomPorts, parameters.ErrOut)
   319  	if err != nil {
   320  		return adapters.NewErrPortForward(err)
   321  	}
   322  	componentStatus.EndpointsForwarded = a.portForwardClient.GetForwardedPorts()
   323  
   324  	componentStatus.State = watch.StateReady
   325  	return nil
   326  }

   332  	componentExists bool,
   333  	ei envinfo.EnvSpecificInfo,
   334  	commands libdevfile.DevfileCommands,
   335  	devfileDebugPort int,
   336  	deployment *appsv1.Deployment,

   342  	componentName := a.ComponentName

   345  		Client:              a.kubeClient,

   365  	odolabels.SetProjectType(annotations, component.GetComponentTypeFromDevfileMetadata(a.AdapterContext.Devfile.Data.GetMetadata()))
   366  	klog.V(4).Infof("We are deploying these annotations: %s", annotations)

   410  	if err != nil {

   416  	if err != nil {

   422  	allVolumes = append(allVolumes, odoMandatoryVolumes...)

   436  		InitContainers:    initContainers,
   437  		Containers:        containers,

   464  	serviceAnnotations["service.binding/backend_port"] = "path={.spec.ports},elementType=sliceOfMaps,sourceKey=name,sourceValue=port"

   501  	} else {

   503  			deployment, err = a.kubeClient.ApplyDeployment(*deployment)
   504  		} else {
   505  			deployment, err = a.kubeClient.CreateDeployment(*deployment)
   506  		}
   507  
   508  		if err != nil {
   509  			return nil, false, err
   510  		}

   519  				return err
   520  			})
   521  			if err != nil {
   522  				return nil, false, err
   523  			}
   524  			klog.V(2).Infof("Successfully created Service for component %s", componentName)
   525  		}
   526  
   527  	}
   528  	newGeneration := deployment.GetGeneration()
   529  
   530  	return deployment, newGeneration != originalGeneration, nil

   533  func (a *Adapter) createOrUpdateServiceForComponent(svc *corev1.Service, componentName string, ownerReference metav1.OwnerReference) error {
   534  	oldSvc, err := a.kubeClient.GetOneService(a.ComponentName, a.AppName)
   535  	originOwnerReferences := svc.OwnerReferences
   536  	if err != nil {
   537  		// no old service was found, create a new one
   538  		if len(svc.Spec.Ports) > 0 {
   539  			err = a.kubeClient.TryWithBlockOwnerDeletion(ownerReference, func(ownerRef metav1.OwnerReference) error {
   540  				svc.OwnerReferences = append(originOwnerReferences, ownerReference)
   541  				_, err = a.kubeClient.CreateService(*svc)
   542  				return err
   543  			})
   544  			if err != nil {
   545  				return err
   546  			}
   547  			klog.V(2).Infof("Successfully created Service for component %s", componentName)
   548  		}
   549  		return nil
   550  	}
   551  	if len(svc.Spec.Ports) > 0 {
   552  		svc.Spec.ClusterIP = oldSvc.Spec.ClusterIP
   553  		svc.ResourceVersion = oldSvc.GetResourceVersion()
   554  		err = a.kubeClient.TryWithBlockOwnerDeletion(ownerReference, func(ownerRef metav1.OwnerReference) error {
   555  			svc.OwnerReferences = append(originOwnerReferences, ownerRef)
   556  			_, err = a.kubeClient.UpdateService(*svc)
   557  			return err
   558  		})
   559  		if err != nil {
   560  			return err
   561  		}
   562  		klog.V(2).Infof("Successfully update Service for component %s", componentName)
   563  		return nil
   564  	}
   565  	// delete the old existing service if the component currently doesn't expose any ports
   566  	return a.kubeClient.DeleteService(oldSvc.Name)
   567  }
