    26  	var sourceVolume corev1.Volume

    28  	if sourcePVCName != "" {

    67  // AddOdoProjectVolume adds the odo project volume to the containers
    68  func AddOdoProjectVolume(containers *[]corev1.Container) {
    69  	if containers == nil {
    70  		return
    71  	}
    72  	for i, container := range *containers {
    73  		for _, env := range container.Env {
    74  			if env.Name == _envProjectsRoot {
    75  				container.VolumeMounts = append(container.VolumeMounts, corev1.VolumeMount{
    76  					Name:      storage.OdoSourceVolume,
    77  					MountPath: env.Value,
    78  				})
    79  				(*containers)[i] = container
    80  			}
    81  		}
    82  	}

   100  // UpdateContainersEntrypointsIfNeeded updates the run components entrypoint
   101  // if no entrypoint has been specified for the component in the devfile
   102  func UpdateContainersEntrypointsIfNeeded(
   103  	devfileObj devfileParser.DevfileObj,
   104  	containers []corev1.Container,
   105  	devfileBuildCmd string,
   106  	devfileRunCmd string,
   107  	devfileDebugCmd string,
   108  ) ([]corev1.Container, error) {
   109  	buildCommand, hasBuildCmd, err := libdevfile.GetCommand(devfileObj, devfileBuildCmd, v1alpha2.BuildCommandGroupKind)
   110  	if err != nil {
   111  		return nil, err
   112  	}
   113  	runCommand, hasRunCmd, err := libdevfile.GetCommand(devfileObj, devfileRunCmd, v1alpha2.RunCommandGroupKind)
   114  	if err != nil {
   115  		return nil, err
   116  	}
   117  	debugCommand, hasDebugCmd, err := libdevfile.GetCommand(devfileObj, devfileDebugCmd, v1alpha2.DebugCommandGroupKind)
   118  	if err != nil {
   119  		return nil, err
   120  	}
   121  
   122  	var cmdsToHandle []v1alpha2.Command
   123  	if hasBuildCmd {
   124  		cmdsToHandle = append(cmdsToHandle, buildCommand)
   125  	}
   126  	if hasRunCmd {
   127  		cmdsToHandle = append(cmdsToHandle, runCommand)
   128  	}
   129  	if hasDebugCmd {
   130  		cmdsToHandle = append(cmdsToHandle, debugCommand)
   131  	}
   132  
   133  	var components []string
   134  	var containerComps []string
   135  	for _, cmd := range cmdsToHandle {
   136  		containerComps, err = libdevfile.GetContainerComponentsForCommand(devfileObj, cmd)
   137  		if err != nil {
   138  			return nil, err
   139  		}
   140  		components = append(components, containerComps...)
   141  	}
   142  
   143  	for _, c := range components {
   144  		for i := range containers {
   145  			container := &containers[i]
   146  			if container.Name == c {
   147  				overrideContainerCommandAndArgsIfNeeded(container)
   148  			}
   149  		}
   150  	}
   151  
   152  	return containers, nil
   153  
   154  }

   156  // UpdateContainerEnvVars updates the container environment variables
   157  func UpdateContainerEnvVars(
   158  	devfileObj devfileParser.DevfileObj,
   159  	containers []corev1.Container,
   160  	devfileDebugCmd string,
   161  	devfileDebugPort int,
   162  ) ([]corev1.Container, error) {
   163  
   164  	debugCommand, hasDebugCmd, err := libdevfile.GetCommand(devfileObj, devfileDebugCmd, v1alpha2.DebugCommandGroupKind)
   165  	if err != nil {
   166  		return nil, err
   167  	}
   168  	if !hasDebugCmd {
   169  		return containers, nil
   170  	}
   171  
   172  	debugContainers, err := libdevfile.GetContainerComponentsForCommand(devfileObj, debugCommand)
   173  	if err != nil {
   174  		return nil, err
   175  	}
   176  
   177  	for i := range containers {
   178  		container := &containers[i]
   179  
   180  		// Check if the container belongs to a debug command component
   181  		for _, c := range debugContainers {
   182  			if container.Name == c && !isEnvPresent(container.Env, _envDebugPort) {
   183  				klog.V(2).Infof("Updating container %v env with debug command's debugPort", container.Name)
   184  				container.Env = append(container.Env,
   185  					corev1.EnvVar{
   186  						Name:  _envDebugPort,
   187  						Value: strconv.Itoa(devfileDebugPort),
   188  					})
   189  				break
   190  			}
   191  		}
   192  	}
   193  
   194  	return containers, nil
   195  }

   197  // overrideContainerCommandAndArgsIfNeeded overrides the container's entrypoint
   198  // if the corresponding component does not have any command and/or args in the Devfile.
   199  // This is a workaround until the default Devfile registry exposes stacks with non-terminating containers.
   200  func overrideContainerCommandAndArgsIfNeeded(container *corev1.Container) {
   201  	if len(container.Command) != 0 || len(container.Args) != 0 {
   202  		return
   203  	}
   204  
   205  	klog.V(2).Infof("No entrypoint defined for the component, setting container %v entrypoint to 'tail -f /dev/null'. You can set a 'command' and/or 'args' for the component to override this default entrypoint.", container.Name)
   206  	// #5768: overriding command and args if the container had no Command or Args defined in it.
   207  	// This is a workaround for us to quickly switch to running without Supervisord,
   208  	// while waiting for the Devfile registry to expose stacks with non-terminating containers.
   209  	// TODO(rm3l): Remove this once https://github.com/devfile/registry/pull/102 is merged on the Devfile side
   210  	container.Command = []string{"tail"}
   211  	container.Args = []string{"-f", "/dev/null"}
   212  }
