    44  // ConvertLabelsToSelector converts the given labels to selector
    45  // To pass operands such as !=, append a ! prefix to the value.
    46  // For E.g. map[string]string{"app.kubernetes.io/managed-by": "!odo"}
    47  // Using != operators also means that resource will be filtered even if it doesn't have the key.
    48  // So a resource not labelled with key "app.kubernetes.io/managed-by" will also be returned.
    49  // TODO(feloy) sync with devfile library?
    50  func ConvertLabelsToSelector(labels map[string]string) string {
    51  	var selector string
    52  	isFirst := true
    53  	keys := make([]string, 0, len(labels))
    54  	for k := range labels {
    55  		keys = append(keys, k)
    56  	}
    57  	sort.Strings(keys)
    58  	for _, k := range keys {
    59  		v := labels[k]
    60  		if isFirst {
    61  			isFirst = false
    62  			if v == "" {
    63  				selector = selector + fmt.Sprintf("%v", k)
    64  			} else {
    65  				if strings.HasPrefix(v, "!") {
    66  					v = strings.Replace(v, "!", "", -1)
    67  					selector = fmt.Sprintf("%v!=%v", k, v)
    68  				} else {
    69  					selector = fmt.Sprintf("%v=%v", k, v)
    70  				}
    71  			}
    72  		} else {
    73  			if v == "" {
    74  				selector = selector + fmt.Sprintf(",%v", k)
    75  			} else {
    76  				if strings.HasPrefix(v, "!") {
    77  					v = strings.Replace(v, "!", "", -1)
    78  					selector = selector + fmt.Sprintf(",%v!=%v", k, v)
    79  				} else {
    80  					selector = selector + fmt.Sprintf(",%v=%v", k, v)
    81  				}
    82  			}
    83  		}
    84  	}
    85  	return selector
    86  }

   151  func GetDNS1123Name(str string) string {
   152  	nonAllowedCharsRegex := regexp.MustCompile(`[^a-zA-Z0-9_-]+`)
   153  	withReplacedChars := strings.Replace(
   154  		nonAllowedCharsRegex.ReplaceAllString(str, "-"),
   155  		"--", "-", -1)
   156  	name := strings.ToLower(removeNonAlphaSuffix(removeNonAlphaPrefix(withReplacedChars)))
   157  	// if the name is all numeric
   158  	if len(str) != 0 && len(name) == 0 {
   159  		name = strings.ToLower(removeNonAlphaSuffix(removeNonAlphaPrefix("x" + withReplacedChars)))
   160  	}
   161  	return name

   169  func removeNonAlphaSuffix(input string) string {
   170  	suffixRegex := regexp.MustCompile("^(.*?)[^a-zA-Z0-9]+$") // regex that strips all trailing non alpha-numeric chars
   171  	matches := suffixRegex.FindStringSubmatch(input)
   172  	matchesLength := len(matches)
   173  	if matchesLength == 0 {
   174  		// in this case the string does not contain a non-alphanumeric suffix
   175  		return input
   176  	}
   177  	// in this case we return the smallest match which in the last element in the array
   178  	return matches[matchesLength-1]
   179  }

   215  // GetAndExtractZip downloads a zip file from a URL with a http prefix or
   216  // takes an absolute path prefixed with file:// and extracts it to a destination.
   217  // pathToUnzip specifies the path within the zip folder to extract
   218  // TODO(feloy) sync with devfile library?
   219  func GetAndExtractZip(zipURL string, destination string, pathToUnzip string, starterToken string) error {
   220  	if zipURL == "" {
   221  		return fmt.Errorf("empty zip url: %s", zipURL)
   222  	}
   223  
   224  	var pathToZip string
   225  	if strings.HasPrefix(zipURL, "file://") {
   226  		pathToZip = strings.TrimPrefix(zipURL, "file:/")
   227  		if runtime.GOOS == "windows" {
   228  			pathToZip = strings.Replace(pathToZip, "\\", "/", -1)
   229  		}
   230  	} else if strings.HasPrefix(zipURL, "http://") || strings.HasPrefix(zipURL, "https://") {
   231  		// Generate temporary zip file location
   232  		time := time.Now().Format(time.RFC3339)
   233  		time = strings.Replace(time, ":", "-", -1) // ":" is illegal char in windows
   234  		pathToZip = path.Join(os.TempDir(), "_"+time+".zip")
   235  
   236  		params := dfutil.DownloadParams{
   237  			Request: dfutil.HTTPRequestParams{
   238  				URL:   zipURL,
   239  				Token: starterToken,
   240  			},
   241  			Filepath: pathToZip,
   242  		}
   243  		err := dfutil.DownloadFile(params)
   244  		if err != nil {
   245  			return err
   246  		}
   247  
   248  		defer func() {
   249  			if err := dfutil.DeletePath(pathToZip); err != nil {
   250  				klog.Errorf("Could not delete temporary directory for zip file. Error: %s", err)
   251  			}
   252  		}()
   253  	} else {
   254  		return fmt.Errorf("invalid Zip URL: %s . Should either be prefixed with file://, http:// or https://", zipURL)
   255  	}
   256  
   257  	filenames, err := Unzip(pathToZip, destination, pathToUnzip)
   258  	if err != nil {
   259  		return err
   260  	}
   261  
   262  	if len(filenames) == 0 {
   263  		return errors.New("no files were unzipped, ensure that the project repo is not empty or that subDir has a valid path")
   264  	}
   265  
   266  	return nil
   267  }

   269  // Unzip will decompress a zip archive, moving specified files and folders
   270  // within the zip file (parameter 1) to an output directory (parameter 2)
   271  // Source: https://golangcode.com/unzip-files-in-go/
   272  // pathToUnzip (parameter 3) is the path within the zip folder to extract
   273  // TODO(feloy) sync with devfile library?
   274  func Unzip(src, dest, pathToUnzip string) ([]string, error) {
   275  	var filenames []string
   276  
   277  	r, err := zip.OpenReader(src)
   278  	if err != nil {
   279  		return filenames, err
   280  	}
   281  	defer r.Close()
   282  
   283  	// change path separator to correct character
   284  	pathToUnzip = filepath.FromSlash(pathToUnzip)
   285  
   286  	// removes first slash of pathToUnzip if present
   287  	pathToUnzip = strings.TrimPrefix(pathToUnzip, string(os.PathSeparator))
   288  
   289  	for _, f := range r.File {
   290  		// Store filename/path for returning and using later on
   291  		index := strings.Index(f.Name, "/")
   292  		filename := filepath.FromSlash(f.Name[index+1:])
   293  		if filename == "" {
   294  			continue
   295  		}
   296  
   297  		// if subDir has a pattern
   298  		match, err := filepath.Match(pathToUnzip, filename)
   299  		if err != nil {
   300  			return filenames, err
   301  		}
   302  
   303  		// destination filepath before trim
   304  		fpath := filepath.Join(dest, filename)
   305  
   306  		// used for pattern matching
   307  		fpathDir := filepath.Dir(fpath)
   308  
   309  		// check for prefix or match
   310  		if strings.HasPrefix(filename, pathToUnzip) {
   311  			filename = strings.TrimPrefix(filename, pathToUnzip)
   312  		} else if !strings.HasPrefix(filename, pathToUnzip) && !match && !sliceContainsString(fpathDir, filenames) {
   313  			continue
   314  		}
   315  		// adds trailing slash to destination if needed as filepath.Join removes it
   316  		if (len(filename) == 1 && os.IsPathSeparator(filename[0])) || filename == "" {
   317  			fpath = dest + string(os.PathSeparator)
   318  		} else {
   319  			fpath = filepath.Join(dest, filename)
   320  		}
   321  		// Check for ZipSlip. More Info: http://bit.ly/2MsjAWE
   322  		if !strings.HasPrefix(fpath, filepath.Clean(dest)+string(os.PathSeparator)) {
   323  			return filenames, fmt.Errorf("%s: illegal file path", fpath)
   324  		}
   325  
   326  		filenames = append(filenames, fpath)
   327  
   328  		if f.FileInfo().IsDir() {
   329  			// Make Folder
   330  			if err = os.MkdirAll(fpath, os.ModePerm); err != nil {
   331  				return filenames, err
   332  			}
   333  			continue
   334  		}
   335  
   336  		// Make File
   337  		if err = os.MkdirAll(filepath.Dir(fpath), os.ModePerm); err != nil {
   338  			return filenames, err
   339  		}
   340  
   341  		outFile, err := os.OpenFile(fpath, os.O_WRONLY|os.O_CREATE|os.O_TRUNC, f.Mode())
   342  		if err != nil {
   343  			return filenames, err
   344  		}
   345  
   346  		rc, err := f.Open()
   347  		if err != nil {
   348  			return filenames, err
   349  		}
   350  
   351  		// limit the number of bytes copied from a file
   352  		// This is set to the limit of file size in Github
   353  		// which is 100MB
   354  		limited := io.LimitReader(rc, 100*1024*1024)
   355  
   356  		_, err = io.Copy(outFile, limited)
   357  
   358  		// Close the file without defer to close before next iteration of loop
   359  		outFile.Close()
   360  		rc.Close()
   361  
   362  		if err != nil {
   363  			return filenames, err
   364  		}
   365  	}
   366  	return filenames, nil
   367  }

   468  // DisplayLog displays logs to user stdout with some color formatting
   469  // numberOfLastLines limits the number of lines from the output when we are not following it
   470  // TODO(feloy) sync with devfile library?
   471  func DisplayLog(followLog bool, rd io.ReadCloser, writer io.Writer, compName string, numberOfLastLines int) (err error) {
   472  
   473  	defer rd.Close()
   474  
   475  	// Copy to stdout (in yellow)
   476  	color.Set(color.FgYellow)
   477  	defer color.Unset()
   478  
   479  	// If we are going to followLog, we'll be copying it to stdout
   480  	// else, we copy it to a buffer
   481  	if followLog {
   482  
   483  		c := make(chan os.Signal, 1)
   484  		signal.Notify(c, os.Interrupt, syscall.SIGTERM)
   485  		go func() {
   486  			<-c
   487  			color.Unset()
   488  			os.Exit(1)
   489  		}()
   490  
   491  		if _, err = io.Copy(writer, rd); err != nil {
   492  			return fmt.Errorf("error followLoging logs for %s: %w", compName, err)
   493  		}
   494  
   495  	} else if numberOfLastLines == -1 {
   496  		// Copy to buffer (we aren't going to be followLoging the logs..)
   497  		buf := new(bytes.Buffer)
   498  		_, err = io.Copy(buf, rd)
   499  		if err != nil {
   500  			return fmt.Errorf("unable to copy followLog to buffer: %w", err)
   501  		}
   502  
   503  		// Copy to stdout
   504  		if _, err = io.Copy(writer, buf); err != nil {
   505  			return fmt.Errorf("error copying logs to stdout: %w", err)
   506  		}
   507  	} else {
   508  		reader := bufio.NewReader(rd)
   509  		var lines []string
   510  		for {
   511  			line, err := reader.ReadString('\n')
   512  			if err != nil {
   513  				if err != io.EOF {
   514  					return err
   515  				} else {
   516  					break
   517  				}
   518  			}
   519  
   520  			lines = append(lines, line)
   521  		}
   522  
   523  		index := len(lines) - numberOfLastLines
   524  		if index < 0 {
   525  			index = 0
   526  		}
   527  
   528  		for i := index; i < len(lines); i++ {
   529  			_, err := fmt.Fprintf(writer, lines[i])
   530  			if err != nil {
   531  				return err
   532  			}
   533  		}
   534  	}
   535  	return
   536

   565  		return err
   566  	}
   567  	if srcinfo, err = fs.Stat(src); err != nil {
   568  		return err
   569  	}

   640  		if err != nil {

   652  // gitSubDir handles subDir for git components
   653  func gitSubDir(srcPath, destinationPath, subDir string, fs filesystem.Filesystem) error {
   654  	go StartSignalWatcher([]os.Signal{syscall.SIGHUP, syscall.SIGINT, syscall.SIGTERM, syscall.SIGQUIT, os.Interrupt}, func(_ os.Signal) {
   655  		err := cleanDir(destinationPath, map[string]bool{
   656  			"devfile.yaml": true,
   657  		}, fs)
   658  		if err != nil {
   659  			klog.V(4).Infof("error %v occurred while calling handleInterruptedSubDir", err)
   660  		}
   661  		err = fs.RemoveAll(srcPath)
   662  		if err != nil {
   663  			klog.V(4).Infof("error %v occurred during temp folder clean up", err)
   664  		}
   665  	})
   666  
   667  	err := func() error {
   668  		// Open the directory.
   669  		outputDirRead, err := fs.Open(filepath.Join(srcPath, subDir))
   670  		if err != nil {
   671  			return err
   672  		}
   673  		defer func() {
   674  			if err1 := outputDirRead.Close(); err1 != nil {
   675  				klog.V(4).Infof("err occurred while closing temp dir: %v", err1)
   676  
   677  			}
   678  		}()
   679  		// Call Readdir to get all files.
   680  		outputDirFiles, err := outputDirRead.Readdir(0)
   681  		if err != nil {
   682  			return err
   683  		}
   684  
   685  		// Loop over files.
   686  		for outputIndex := range outputDirFiles {
   687  			outputFileHere := outputDirFiles[outputIndex]
   688  
   689  			// Get name of file.
   690  			fileName := outputFileHere.Name()
   691  
   692  			oldPath := filepath.Join(srcPath, subDir, fileName)
   693  
   694  			if outputFileHere.IsDir() {
   695  				err = copyDirWithFS(oldPath, filepath.Join(destinationPath, fileName), fs)
   696  			} else {
   697  				err = copyFileWithFs(oldPath, filepath.Join(destinationPath, fileName), fs)
   698  			}
   699  
   700  			if err != nil {
   701  				return err
   702  			}
   703  		}
   704  		return nil
   705  	}()
   706  	if err != nil {
   707  		return err
   708  	}
   709  	return fs.RemoveAll(srcPath)
   710  }
