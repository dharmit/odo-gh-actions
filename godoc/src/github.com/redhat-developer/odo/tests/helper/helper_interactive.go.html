    21  // InteractiveContext represents the context of an interactive command to be run.
    22  type InteractiveContext struct {
    23  
    24  	//Command represents the original command ran

    39  // Tester represents the function that contains all steps to test the given interactive command.
    40  // The InteractiveContext argument needs to be passed to the various helper.SendLine and helper.ExpectString methods.
    41  type Tester func(InteractiveContext)

    43  // RunInteractive runs the command in interactive mode and returns the output, and error.
    44  // It takes command as array of strings, and a function `tester` that contains steps to run the test as an argument.
    45  // The command is executed as a separate process, the environment of which is controlled via the `env` argument.
    46  // The initial value of the sub-process environment is a copy of the environment of the current process.
    47  // If `env` is not `nil`, it will be appended to the end of the sub-process environment.
    48  // If there are duplicate environment keys, only the last value in the slice for each duplicate key is used.
    49  func RunInteractive(command []string, env []string, tester Tester) (string, error) {
    50  
    51  	fmt.Fprintln(GinkgoWriter, "running command", command, "with env", env)
    52  
    53  	ptm, pts, err := pty.Open()
    54  	if err != nil {
    55  		log.Fatal(err)
    56  	}
    57  
    58  	term := vt10x.New(vt10x.WithWriter(pts))
    59  
    60  	// Set to 1 MINUTE timeout, since the command may take a while to start
    61  	c, err := expect.NewConsole(expect.WithStdin(ptm), expect.WithStdout(term), expect.WithCloser(pts, ptm), expect.WithDefaultTimeout(3*time.Minute))
    62  	if err != nil {
    63  		log.Fatal(err)
    64  	}
    65  	defer c.Close()
    66  
    67  	// execute the command
    68  	cmd := exec.Command(command[0], command[1:]...)
    69  	// setup stdin, stdout and stderr
    70  	cmd.Stdin = c.Tty()
    71  	cmd.Stdout = c.Tty()
    72  	cmd.Stderr = c.Tty()
    73  	if env != nil {
    74  		cmd.Env = append(os.Environ(), env...)
    75  	}
    76  	err = cmd.Start()
    77  	if err != nil {
    78  		log.Fatal(err)
    79  	}
    80  
    81  	buf := new(bytes.Buffer)
    82  	ctx := InteractiveContext{
    83  		Command: command,
    84  		console: c,
    85  		buffer:  buf,
    86  		StopCommand: func() {
    87  			_ = cmd.Process.Kill()
    88  		},
    89  	}
    90  	tester(ctx)
    91  
    92  	err = cmd.Wait()
    93  
    94  	// Close the slave end of the pty, and read the remaining bytes from the master end.
    95  	c.Tty().Close()
    96  
    97  	return buf.String(), err
    98  }

   100  // expectDescriptionSupplier returns a function intended to be used as description supplier
   101  // when checking errors do not occur in ExpectString and SendLine.
   102  // Note that the function returned is evaluated lazily, only in case an error occurs.
   103  func expectDescriptionSupplier(ctx InteractiveContext, line string) func() string {
   104  	return func() string {
   105  		return fmt.Sprintf("error while sending or expecting line: \"%s\"\n"+
   106  			"=== output of command '%+q' read so far ===\n%v\n======================",
   107  			line,
   108  			ctx.Command,
   109  			ctx.buffer)
   110  	}
   111  }

   113  func SendLine(ctx InteractiveContext, line string) {
   114  	_, err := ctx.console.SendLine(line)
   115  	Expect(err).ShouldNot(HaveOccurred(), expectDescriptionSupplier(ctx, line))
   116  }

   118  func ExpectString(ctx InteractiveContext, line string) {
