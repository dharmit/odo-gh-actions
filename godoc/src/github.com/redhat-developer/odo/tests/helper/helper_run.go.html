    22  func CmdRunner(program string, args ...string) *gexec.Session {
    23  	// prefix ginkgo verbose output with program name
    24  	prefix := fmt.Sprintf("[%s] ", filepath.Base(program))
    25  	prefixWriter := gexec.NewPrefixedWriter(prefix, GinkgoWriter)
    26  	command := exec.Command(program, args...)
    27  	setSysProcAttr(command)
    28  	fmt.Fprintln(GinkgoWriter, runningCmd(command))
    29  	session, err := gexec.Start(command, prefixWriter, prefixWriter)
    30  	Expect(err).NotTo(HaveOccurred())
    31  	return session
    32  }

    66  // WaitAndCheckForTerminatingState waits for the given interval
    67  // and checks if the given resource type has been deleted on the cluster or is in the terminating state
    68  // path is the path to the program's binary
    69  func WaitAndCheckForTerminatingState(path, resourceType, namespace string, timeoutMinutes int) bool {
    70  	pingTimeout := time.After(time.Duration(timeoutMinutes) * time.Minute)
    71  	// this is a test package so time.Tick() is acceptable
    72  	// nolint
    73  	tick := time.Tick(time.Second)
    74  	for {
    75  		select {
    76  		case <-pingTimeout:
    77  			Fail(fmt.Sprintf("Timeout after %d minutes", timeoutMinutes))
    78  
    79  		case <-tick:
    80  			session := CmdRunner(path, "get", resourceType, "--namespace", namespace)
    81  			Eventually(session).Should(gexec.Exit(0))
    82  			// https://github.com/kubernetes/kubectl/issues/847
    83  			outputStdErr := string(session.Wait().Err.Contents())
    84  			outputStdOut := string(session.Wait().Out.Contents())
    85  
    86  			// if the resource gets deleted before the check, we won't get the `terminating` state output
    87  			// thus we also check and exit when the resource has been deleted on the cluster.
    88  			if strings.Contains(strings.ToLower(outputStdErr), "no resources found") || strings.Contains(strings.ToLower(outputStdOut), "terminating") {
    89  				return true
    90  			}
    91  		}
    92  	}
    93  }

    96  // belonging to the given component, app and project
    97  func GetAnnotationsDeployment(path, componentName, appName, projectName string) map[string]string {
    98  	var mapOutput = make(map[string]string)
    99  	selector := labels.Builder().WithComponentName(componentName).WithAppName(appName).SelectorFlag()
   100  	output := Cmd(path, "get", "deployment", selector, "--namespace", projectName,
   101  		"-o", "go-template='{{ range $k, $v := (index .items 0).metadata.annotations}}{{$k}}:{{$v}}{{\"\\n\"}}{{end}}'").ShouldPass().Out()
   102  
   103  	for _, line := range strings.Split(output, "\n") {
   104  		line = strings.TrimPrefix(line, "'")
   105  		splits := strings.Split(line, ":")
   106  		if len(splits) < 2 {
   107  			continue
   108  		}
   109  		name := splits[0]
   110  		value := strings.Join(splits[1:], ":")
   111  		mapOutput[name] = value
   112  	}
   113  	return mapOutput
   114  }
