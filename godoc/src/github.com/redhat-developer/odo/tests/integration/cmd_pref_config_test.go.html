    30  	Context("check that help works", func() {
    31  		It("should display help info", func() {
    32  			helpArgs := []string{"-h", "help", "--help"}
    33  			for _, helpArg := range helpArgs {
    34  				appHelp := helper.Cmd("odo", helpArg).ShouldPass().Out()

    36  			}

    44  		})
    45  	})

    47  	Context("When viewing global config", func() {
    48  		var newContext string
    49  		// ConsentTelemetry is set to false in helper.CommonBeforeEach so that it does not prompt to set a value
    50  		// during the tests, but we want to check preference values as they would be in real time and hence
    51  		// we set the GLOBALODOCONFIG variable to a value in new context
    52  		var _ = JustBeforeEach(func() {
    53  			newContext = helper.CreateNewContext()
    54  			os.Setenv("GLOBALODOCONFIG", filepath.Join(newContext, "preference.yaml"))
    55  		})
    56  		var _ = JustAfterEach(func() {
    57  			helper.DeleteDir(newContext)
    58  		})
    59  		It("should get the default global config keys", func() {
    60  			configOutput := helper.Cmd("odo", "preference", "view").ShouldPass().Out()
    61  			preferences := []string{"UpdateNotification", "Timeout", "PushTimeout", "RegistryCacheTime", "Ephemeral", "ConsentTelemetry"}
    62  			helper.MatchAllInOutput(configOutput, preferences)
    63  			for _, key := range preferences {
    64  				value := helper.GetPreferenceValue(key)
    65  				Expect(value).To(BeEmpty())
    66  			}
    67  		})
    68  	})
    69  
    70  	Context("When configuring global config values", func() {
    71  		preferences := []struct {
    72  			name              string
    73  			value             string
    74  			updateValue       string
    75  			invalidValue      string
    76  			firstSetWithForce bool
    77  		}{
    78  			{"UpdateNotification", "false", "true", "foo", false},
    79  			{"Timeout", "5s", "6s", "foo", false},
    80  			// !! Do not test ConsentTelemetry with true because it sends out the telemetry data and messes up the statistics !!
    81  			{"ConsentTelemetry", "false", "false", "foo", false},
    82  			{"PushTimeout", "4s", "6s", "foo", false},
    83  			{"RegistryCacheTime", "4m", "6m", "foo", false},
    84  			{"Ephemeral", "false", "true", "foo", true},
    85  		}
    86  
    87  		It("should successfully updated", func() {
    88  			for _, pref := range preferences {
    89  				// construct arguments for the first command
    90  				firstCmdArgs := []string{"preference", "set"}
    91  				if pref.firstSetWithForce {
    92  					firstCmdArgs = append(firstCmdArgs, "-f")
    93  				}
    94  				firstCmdArgs = append(firstCmdArgs, pref.name, pref.value)
    95  
    96  				helper.Cmd("odo", firstCmdArgs...).ShouldPass()
    97  				value := helper.GetPreferenceValue(pref.name)
    98  				Expect(value).To(ContainSubstring(pref.value))
    99  
   100  				helper.Cmd("odo", "preference", "set", "-f", pref.name, pref.updateValue).ShouldPass()
   101  				value = helper.GetPreferenceValue(pref.name)
   102  				Expect(value).To(ContainSubstring(pref.updateValue))
   103  
   104  				helper.Cmd("odo", "preference", "unset", "-f", pref.name).ShouldPass()
   105  				value = helper.GetPreferenceValue(pref.name)
   106  				Expect(value).To(BeEmpty())
   107  			}
   108  			globalConfPath := os.Getenv("HOME")
   109  			os.RemoveAll(filepath.Join(globalConfPath, ".odo"))
   110  		})
   111  	})
