    15  var _ = Describe("odo create/delete/list/set namespace/project tests", func() {
    16  	var commonVar helper.CommonVar
    17  
    18  	BeforeEach(func() {
    19  		commonVar = helper.CommonBeforeEach(helper.SetupClusterTrue)
    20  	})
    21  
    22  	AfterEach(func() {
    23  		helper.CommonAfterEach(commonVar)
    24  	})
    25  	for _, commandName := range []string{"namespace", "project"} {
    26  		// this is a workaround to ensure that the for loop works with `It` blocks
    27  		commandName := commandName
    28  		Describe("create "+commandName, func() {
    29  
    30  			namespace := fmt.Sprintf("%s-%s", helper.RandString(4), commandName)
    31  
    32  			It(fmt.Sprintf("should successfully create the %s", commandName), func() {
    33  				helper.Cmd("odo", "create", commandName, namespace, "--wait").ShouldPass()
    34  				defer func(ns string) {
    35  					commonVar.CliRunner.DeleteNamespaceProject(ns, false)
    36  				}(namespace)
    37  				Expect(commonVar.CliRunner.HasNamespaceProject(namespace)).To(BeTrue())
    38  				Expect(commonVar.CliRunner.GetActiveNamespace()).To(Equal(namespace))
    39  			})
    40  
    41  			It(fmt.Sprintf("should fail to create %s", commandName), func() {
    42  				By("using an existent name", func() {
    43  					helper.Cmd("odo", "create", commandName, commonVar.Project).ShouldFail()
    44  				})
    45  				By("using an invalid name", func() {
    46  					helper.Cmd("odo", "create", commandName, "12345").ShouldFail()
    47  					Expect(commonVar.CliRunner.GetActiveNamespace()).To(Equal(commonVar.Project))
    48  				})
    49  			})
    50  		})
    51  
    52  		Describe("delete "+commandName, func() {
    53  
    54  			When("force-deleting a valid "+commandName, func() {
    55  				var namespace string
    56  
    57  				BeforeEach(func() {
    58  					namespace = helper.CreateRandProject()
    59  					Expect(commonVar.CliRunner.HasNamespaceProject(namespace)).To(BeTrue())
    60  				})
    61  
    62  				checkNsDeletionFunc := func(additionalArgs []string, nsCheckerFunc func()) {
    63  					args := []string{"delete", commandName, namespace, "--force"}
    64  					if additionalArgs != nil {
    65  						args = append(args, additionalArgs...)
    66  					}
    67  					out := helper.Cmd("odo", args...).ShouldPass().Out()
    68  					if nsCheckerFunc != nil {
    69  						nsCheckerFunc()
    70  					}
    71  					Expect(out).To(
    72  						ContainSubstring(fmt.Sprintf("%s %q deleted", strings.Title(commandName), namespace)))
    73  				}
    74  
    75  				It(fmt.Sprintf("should successfully delete the %s asynchronously", commandName), func() {
    76  					checkNsDeletionFunc(nil, func() {
    77  						Eventually(func() bool {
    78  							return commonVar.CliRunner.HasNamespaceProject(namespace)
    79  						}, 60*time.Second).Should(BeFalse())
    80  					})
    81  				})
    82  
    83  				It(fmt.Sprintf("should successfully delete the %s synchronously with --wait", commandName), func() {
    84  					checkNsDeletionFunc([]string{"--wait"}, func() {
    85  						Expect(commonVar.CliRunner.HasNamespaceProject(namespace)).To(BeFalse())
    86  					})
    87  				})
    88  			})
    89  
    90  			It("should not succeed to delete a non-existent "+commandName, func() {
    91  				fakeNamespace := "my-fake-ns-" + helper.RandString(3)
    92  				By("using the force flag and asynchronously", func() {
    93  					helper.Cmd("odo", "delete", commandName, fakeNamespace, "--force").ShouldFail()
    94  				})
    95  
    96  				By("using the force flag and waiting", func() {
    97  					helper.Cmd("odo", "delete", commandName, fakeNamespace, "--force", "--wait").ShouldFail()
    98  				})
    99  			})
   100  
   101  		})

   119  					Expect(commonVar.CliRunner.GetActiveNamespace()).ShouldNot(Equal(anotherNs))
   120  					helper.Cmd("odo", "set", commandName, anotherNs).ShouldPass()
   121  					Expect(commonVar.CliRunner.GetActiveNamespace()).To(Equal(anotherNs))
   122  				})
   123  
   124  				By("setting it again to its previous value", func() {
   125  					helper.Cmd("odo", "set", commandName, anotherNs).ShouldPass()
   126  					Expect(commonVar.CliRunner.GetActiveNamespace()).To(Equal(anotherNs))
   127  				})
   128  			})

   130  			It(fmt.Sprintf("should not succeed to set the %s", commandName), func() {
   131  				invalidNs := "234567"
   132  				helper.Cmd("odo", "set", commandName, invalidNs).ShouldFail()
   133  				Expect(commonVar.CliRunner.GetActiveNamespace()).ShouldNot(Equal(invalidNs))
   134  			})

   136  			When("running inside a component directory", func() {
   137  				activeNs := "my-current-ns"
   138  
   139  				BeforeEach(func() {
   140  					helper.CopyExampleDevFile(
   141  						filepath.Join("source", "devfiles", "nodejs", "devfile.yaml"),

   165  
   166  					By("displaying warning message", func() {

   174  					By("not changing the namespace of the existing component", func() {

   179  		})
