   192  		It("should successfully extract the project in the specified subDir path", func() {
   193  			var found, notToBeFound int
   194  			pathsToValidate := map[string]bool{
   195  				filepath.Join(commonVar.Context, "java", "com"):                                            true,
   196  				filepath.Join(commonVar.Context, "java", "com", "example"):                                 true,
   197  				filepath.Join(commonVar.Context, "java", "com", "example", "demo"):                         true,
   198  				filepath.Join(commonVar.Context, "java", "com", "example", "demo", "DemoApplication.java"): true,
   199  				filepath.Join(commonVar.Context, "resources", "application.properties"):                    true,
   200  			}
   201  			pathsNotToBePresent := map[string]bool{
   202  				filepath.Join(commonVar.Context, "src"):  true,
   203  				filepath.Join(commonVar.Context, "main"): true,
   204  			}
   205  			err := filepath.Walk(commonVar.Context, func(path string, info os.FileInfo, err error) error {
   206  				if err != nil {
   207  					return err
   208  				}
   209  				if ok := pathsToValidate[path]; ok {
   210  					found++
   211  				}
   212  				if ok := pathsNotToBePresent[path]; ok {
   213  					notToBeFound++
   214  				}
   215  				return nil
   216  			})
   217  			Expect(err).To(BeNil())
   218  
   219  			Expect(found).To(Equal(len(pathsToValidate)))
   220  			Expect(notToBeFound).To(Equal(0))
   221  		})
   222  	})
