    20  // getComponentDeployment returns the deployment associated with the component, if deployed
    21  // and indicate if the deployment has been found
    22  func (a *Adapter) getComponentDeployment() (*appsv1.Deployment, bool, error) {
    23  	// Get the Dev deployment:
    24  	// Since `odo deploy` can theoretically deploy a deployment as well with the same instance name
    25  	// we make sure that we are retrieving the deployment with the Dev mode, NOT Deploy.
    26  	selectorLabels := odolabels.GetSelector(a.ComponentName, a.AppName, odolabels.ComponentDevMode, true)
    27  	deployment, err := a.kubeClient.GetOneDeploymentFromSelector(selectorLabels)
    28  
    29  	if err != nil {
    30  		if _, ok := err.(*kclient.DeploymentNotFoundError); !ok {
    31  			return nil, false, fmt.Errorf("unable to determine if component %s exists: %w", a.ComponentName, err)
    32  		}
    33  	}
    34  	componentExists := deployment != nil
    35  	return deployment, componentExists, nil
    36  }

    38  // pushDevfileKubernetesComponents gets the Kubernetes components from the Devfile and push them to the cluster
    39  // adding the specified labels to them
    40  func (a *Adapter) pushDevfileKubernetesComponents(
    41  	labels map[string]string,
    42  ) ([]v1alpha2.Component, error) {
    43  	// fetch the "kubernetes inlined components" to create them on cluster
    44  	// from odo standpoint, these components contain yaml manifest of ServiceBinding
    45  	k8sComponents, err := devfile.GetKubernetesComponentsToPush(a.Devfile)
    46  	if err != nil {
    47  		return nil, fmt.Errorf("error while trying to fetch service(s) from devfile: %w", err)
    48  	}
    49  
    50  	// validate if the GVRs represented by Kubernetes inlined components are supported by the underlying cluster
    51  	err = ValidateResourcesExist(a.kubeClient, a.Devfile, k8sComponents, a.Context)
    52  	if err != nil {
    53  		return nil, err
    54  	}
    55  
    56  	// Set the annotations for the component type
    57  	annotations := make(map[string]string)
    58  	odolabels.SetProjectType(annotations, component.GetComponentTypeFromDevfileMetadata(a.AdapterContext.Devfile.Data.GetMetadata()))
    59  
    60  	// create the Kubernetes objects from the manifest and delete the ones not in the devfile
    61  	err = service.PushKubernetesResources(a.kubeClient, a.Devfile, k8sComponents, labels, annotations, a.Context)
    62  	if err != nil {
    63  		return nil, fmt.Errorf("failed to create Kubernetes resources associated with the component: %w", err)
    64  	}
    65  	return k8sComponents, nil
    66  }

    85  func (a *Adapter) updatePVCsOwnerReferences(ownerReference metav1.OwnerReference) error {
    86  	// list the latest state of the PVCs
    87  	pvcs, err := a.kubeClient.ListPVCs(fmt.Sprintf("%v=%v", "component", a.ComponentName))
    88  	if err != nil {
    89  		return err
    90  	}
    91  
    92  	// update the owner reference of the PVCs with the deployment
    93  	for i := range pvcs {
    94  		if pvcs[i].OwnerReferences != nil || pvcs[i].DeletionTimestamp != nil {
    95  			continue
    96  		}
    97  		err = a.kubeClient.TryWithBlockOwnerDeletion(ownerReference, func(ownerRef metav1.OwnerReference) error {
    98  			return a.kubeClient.UpdateStorageOwnerReference(&pvcs[i], ownerRef)
    99  		})
   100  		if err != nil {
   101  			return err
   102  		}
   103  	}
   104  	return nil
   105  }
