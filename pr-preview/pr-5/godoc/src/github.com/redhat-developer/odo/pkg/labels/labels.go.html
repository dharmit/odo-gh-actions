    92  func getLabels(componentName string, applicationName string, mode string, additional bool, isPartOfComponent bool) labels.Set {
    93  	labels := getApplicationLabels(applicationName, additional)
    94  	labels[kubernetesInstanceLabel] = componentName
    95  	if mode != ComponentAnyMode {
    96  		labels[odoModeLabel] = mode
    97  	}
    98  	if isPartOfComponent {
    99  		labels[componentLabel] = componentName
   100  	}
   101  	return labels
   102  }
