    49  func (o *PFClient) StartPortForwarding(
    50  	devFileObj parser.DevfileObj,
    51  	componentName string,
    52  	randomPorts bool,
    53  	errOut io.Writer,
    54  ) error {
    55  
    56  	ceMapping, err := o.GetPortsToForward(devFileObj)
    57  	if err != nil {
    58  		return err
    59  	}
    60  
    61  	if o.stopChan != nil && reflect.DeepEqual(ceMapping, o.appliedEndpoints) {
    62  		return nil
    63  	}
    64  
    65  	o.appliedEndpoints = ceMapping
    66  
    67  	o.StopPortForwarding()
    68  
    69  	o.stopChan = make(chan struct{}, 1)
    70  
    71  	var portPairs map[string][]string
    72  	if randomPorts {
    73  		portPairs = randomPortPairsFromContainerEndpoints(ceMapping)
    74  	} else {
    75  		portPairs = portPairsFromContainerEndpoints(ceMapping)
    76  	}
    77  	var portPairsSlice []string
    78  	for _, v1 := range portPairs {
    79  		portPairsSlice = append(portPairsSlice, v1...)
    80  	}
    81  	pod, err := o.kubernetesClient.GetPodUsingComponentName(componentName)
    82  	if err != nil {
    83  		return err
    84  	}
    85  
    86  	o.originalErrorHandlers = append([]func(error){}, runtime.ErrorHandlers...)
    87  
    88  	runtime.ErrorHandlers = append(runtime.ErrorHandlers, func(err error) {
    89  		if err.Error() == "lost connection to pod" {
    90  			// Stop the low-level port forwarding
    91  			// the infinite loop will restart it
    92  			if o.stopChan == nil {
    93  				return
    94  			}
    95  			o.stopChan <- struct{}{}
    96  			o.stopChan = make(chan struct{}, 1)
    97  		}
    98  	})
    99  
   100  	o.isRunning = true
   101  
   102  	devstateChan := make(chan error)
   103  	go func() {
   104  		backo := watch.NewExpBackoff()
   105  		for {
   106  			o.finishedChan = make(chan struct{}, 1)
   107  			portsBuf := NewPortWriter(log.GetStdout(), len(portPairsSlice), ceMapping)
   108  
   109  			go func() {
   110  				portsBuf.Wait()
   111  				err = o.stateClient.SetForwardedPorts(portsBuf.GetForwardedPorts())
   112  				if err != nil {
   113  					err = fmt.Errorf("unable to save forwarded ports to state file: %v", err)
   114  				}
   115  				devstateChan <- err
   116  			}()
   117  
   118  			err = o.kubernetesClient.SetupPortForwarding(pod, portPairsSlice, portsBuf, errOut, o.stopChan)
   119  			if err != nil {
   120  				fmt.Fprintf(errOut, "Failed to setup port-forwarding: %v\n", err)
   121  				d := backo.Delay()
   122  				time.Sleep(d)
   123  			} else {
   124  				backo.Reset()
   125  			}
   126  			if !o.isRunning {
   127  				break
   128  			}
   129  		}
   130  		o.finishedChan <- struct{}{}
   131  	}()
   132  
   133  	// Wait the first time the devstate file is written
   134  	timeout := 1 * time.Minute
   135  	select {
   136  	case err = <-devstateChan:
   137  		return err
   138  	case <-time.After(timeout):
   139  		return errors.New("unable to setup port forwarding")
   140  	}
   141  }

   143  func (o *PFClient) StopPortForwarding() {
   144  	if o.stopChan == nil {
   145  		return
   146  	}
   147  	// Ask the low-level port forward to stop
   148  	o.stopChan <- struct{}{}
   149  	o.stopChan = nil
   150  
   151  	// Ask the infinite loop to stop
   152  	o.isRunning = false
   153  
   154  	// Wait for low level port forward to be finished
   155  	// and the infinite loop to exit
   156  	<-o.finishedChan
   157  	o.finishedChan = nil
   158  	runtime.ErrorHandlers = o.originalErrorHandlers
   159  }

   178  // randomPortPairsFromContainerEndpoints assigns a random (empty) port on localhost to each port in the provided containerEndpoints map
   179  // it returns a map of the format "<container-name>":{"<local-port-1>:<remote-port-1>", "<local-port-2>:<remote-port-2>"}
   180  // "container1": {":3000", ":3001"}
   181  func randomPortPairsFromContainerEndpoints(ceMap map[string][]int) map[string][]string {
   182  	portPairs := make(map[string][]string)
   183  
   184  	for name, ports := range ceMap {
   185  		for _, p := range ports {
   186  			pair := fmt.Sprintf(":%d", p)
   187  			portPairs[name] = append(portPairs[name], pair)
   188  		}
   189  	}
   190  	return portPairs
   191  }

   193  // portPairsFromContainerEndpoints assigns a port on localhost to each port in the provided containerEndpoints map
   194  // it returns a map of the format "<container-name>":{"<local-port-1>:<remote-port-1>", "<local-port-2>:<remote-port-2>"}
   195  // "container1": {"400001:3000", "400002:3001"}
   196  func portPairsFromContainerEndpoints(ceMap map[string][]int) map[string][]string {
   197  	portPairs := make(map[string][]string)
   198  	port := 40000
   199  
   200  	for name, ports := range ceMap {
   201  		for _, p := range ports {
   202  			port++
   203  			for {
   204  				isPortFree := util.IsPortFree(port)
   205  				if isPortFree {
   206  					pair := fmt.Sprintf("%d:%d", port, p)
   207  					portPairs[name] = append(portPairs[name], pair)
   208  					break
   209  				}
   210  				port++
   211  			}
   212  		}
   213  	}
   214  	return portPairs
   215  }
