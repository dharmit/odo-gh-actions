   111  // addRecursiveWatch handles adding watches recursively for the path provided
   112  // and its subdirectories.  If a non-directory is specified, this call is a no-op.
   113  // Files matching glob pattern defined in ignores will be ignored.
   114  // Taken from https://github.com/openshift/origin/blob/85eb37b34f0657631592356d020cef5a58470f8e/pkg/util/fsnotification/fsnotification.go
   115  // rootPath is the root path of the file or directory,
   116  // path is the recursive path of the file or the directory,
   117  // ignores contains the glob rules for matching
   118  func addRecursiveWatch(watcher *fsnotify.Watcher, rootPath string, path string, ignores []string) error {
   119  
   120  	file, err := os.Stat(path)
   121  	if err != nil {
   122  		if os.IsNotExist(err) {
   123  			return nil
   124  		}
   125  		return fmt.Errorf("error introspecting path %s: %v", path, err)
   126  	}
   127  
   128  	ignoreMatcher := gitignore.CompileIgnoreLines(ignores...)
   129  
   130  	mode := file.Mode()
   131  	if mode.IsRegular() {
   132  		var rel string
   133  		rel, err = filepath.Rel(rootPath, path)
   134  		if err != nil {
   135  			return err
   136  		}
   137  		matched := ignoreMatcher.MatchesPath(rel)
   138  		if !matched {
   139  			klog.V(4).Infof("adding watch on path %s", path)
   140  
   141  			// checking if the file exits before adding the watcher to it
   142  			if !util.CheckPathExists(path) {
   143  				return nil
   144  			}
   145  
   146  			err = watcher.Add(path)
   147  			if err != nil {
   148  				klog.V(4).Infof("error adding watcher for path %s: %v", path, err)
   149  			}
   150  			return nil
   151  		}
   152  	}
   153  
   154  	folders := []string{}
   155  	err = filepath.Walk(path, func(newPath string, info os.FileInfo, err error) error {
   156  		if err != nil {
   157  			// Ignore the error if it's a 'path does not exist' error, no need to walk a non-existent path
   158  			if !util.CheckPathExists(newPath) {
   159  				klog.V(4).Infof("Walk func received an error for path %s, but the path doesn't exist so this is likely not an error. err: %v", path, err)
   160  				return nil
   161  			}
   162  			return fmt.Errorf("unable to walk path: %s: %w", newPath, err)
   163  		}
   164  
   165  		if info.IsDir() {
   166  			// If the current directory matches any of the ignore patterns, ignore them so that their contents are also not ignored
   167  			rel, err := filepath.Rel(rootPath, newPath)
   168  			if err != nil {
   169  				return err
   170  			}
   171  			matched := ignoreMatcher.MatchesPath(rel)
   172  			if err != nil {
   173  				return fmt.Errorf("unable to addRecursiveWatch on %s: %w", newPath, err)
   174  			}
   175  			if matched {
   176  				klog.V(4).Infof("ignoring watch on path %s", newPath)
   177  				return filepath.SkipDir
   178  			}
   179  			// Append the folder we just walked on
   180  			folders = append(folders, newPath)
   181  		}
   182  		return nil
   183  	})
   184  	if err != nil {
   185  		return err
   186  	}
   187  	for _, folder := range folders {
   188  
   189  		rel, err := filepath.Rel(rootPath, folder)
   190  		if err != nil {
   191  			return err
   192  		}
   193  		matched := ignoreMatcher.MatchesPath(rel)
   194  
   195  		if matched {
   196  			klog.V(4).Infof("ignoring watch for %s", folder)
   197  			continue
   198  		}
   199  
   200  		// checking if the file exits before adding the watcher to it
   201  		if !util.CheckPathExists(path) {
   202  			continue
   203  		}
   204  
   205  		klog.V(4).Infof("adding watch on path %s", folder)
   206  		err = watcher.Add(folder)
   207  		if err != nil {
   208  			// Linux "no space left on device" issues are usually resolved via
   209  			// $ sudo sysctl fs.inotify.max_user_watches=65536
   210  			// BSD / OSX: "too many open files" issues are ussualy resolved via
   211  			// $ sysctl variables "kern.maxfiles" and "kern.maxfilesperproc",
   212  			klog.V(4).Infof("error adding watcher for path %s: %v", folder, err)
   213  		}
   214  	}
   215  	return nil
   216  }

   218  func (o *WatchClient) WatchAndPush(out io.Writer, parameters WatchParameters, ctx context.Context, componentStatus ComponentStatus) error {
   219  	klog.V(4).Infof("starting WatchAndPush, path: %s, component: %s, ignores %s", parameters.Path, parameters.ComponentName, parameters.FileIgnores)
   220  
   221  	var sourcesWatcher *fsnotify.Watcher
   222  	var err error
   223  	if parameters.WatchFiles {
   224  		sourcesWatcher, err = getFullSourcesWatcher(parameters.Path, parameters.FileIgnores)
   225  		if err != nil {
   226  			return err
   227  		}
   228  	} else {
   229  		sourcesWatcher, err = fsnotify.NewWatcher()
   230  		if err != nil {
   231  			return err
   232  		}
   233  	}
   234  	defer sourcesWatcher.Close()
   235  
   236  	selector := labels.GetSelector(parameters.ComponentName, parameters.ApplicationName, labels.ComponentDevMode, true)
   237  	deploymentWatcher, err := o.kubeClient.DeploymentWatcher(ctx, selector)
   238  	if err != nil {
   239  		return fmt.Errorf("error watching deployment: %v", err)
   240  	}
   241  
   242  	devfileWatcher, err := fsnotify.NewWatcher()
   243  	if err != nil {
   244  		return err
   245  	}
   246  	if parameters.WatchFiles {
   247  		var devfileFiles []string
   248  		devfileFiles, err = libdevfile.GetReferencedLocalFiles(parameters.InitialDevfileObj)
   249  		if err != nil {
   250  			return err
   251  		}
   252  		devfileFiles = append(devfileFiles, parameters.DevfilePath)
   253  		for _, f := range devfileFiles {
   254  			err = devfileWatcher.Add(f)
   255  			if err != nil {
   256  				klog.V(4).Infof("error adding watcher for path %s: %v", f, err)
   257  			}
   258  		}
   259  	}
   260  
   261  	podWatcher, err := o.kubeClient.PodWatcher(ctx, selector)
   262  	if err != nil {
   263  		return err
   264  	}
   265  
   266  	warningsWatcher, isForbidden, err := o.kubeClient.PodWarningEventWatcher(ctx)
   267  	if err != nil {
   268  		return err
   269  	}
   270  	if isForbidden {
   271  		log.Fwarning(out, "Unable to watch Events resource, warning Events won't be displayed")
   272  	}
   273  
   274  	return o.eventWatcher(ctx, sourcesWatcher, deploymentWatcher, devfileWatcher, podWatcher, warningsWatcher, parameters, out, evaluateFileChanges, processEvents, componentStatus)
   275  }

   277  func getFullSourcesWatcher(path string, fileIgnores []string) (*fsnotify.Watcher, error) {
   278  	absIgnorePaths := dfutil.GetAbsGlobExps(path, fileIgnores)
   279  
   280  	watcher, err := fsnotify.NewWatcher()
   281  	if err != nil {
   282  		return nil, fmt.Errorf("error setting up filesystem watcher: %v", err)
   283  	}
   284  
   285  	// adding watch on the root folder and the sub folders recursively
   286  	// so directory and the path in addRecursiveWatch() are the same
   287  	err = addRecursiveWatch(watcher, path, path, absIgnorePaths)
   288  	if err != nil {
   289  		return nil, fmt.Errorf("error watching source path %s: %v", path, err)
   290  	}
   291  	return watcher, nil
   292  }

   294  // eventWatcher loops till the context's Done channel indicates it to stop looping, at which point it performs cleanup.
   295  // While looping, it listens for filesystem events and processes these events using the WatchParameters to push to the remote pod.
   296  // It outputs any logs to the out io Writer
   297  func (o *WatchClient) eventWatcher(ctx context.Context, sourcesWatcher *fsnotify.Watcher, deploymentWatcher watch.Interface, devfileWatcher *fsnotify.Watcher, podWatcher watch.Interface, eventsWatcher watch.Interface, parameters WatchParameters, out io.Writer, evaluateChangesHandler evaluateChangesFunc, processEventsHandler processEventsFunc, componentStatus ComponentStatus) error {
   298  
   299  	expBackoff := NewExpBackoff()
   300  
   301  	var events []fsnotify.Event
   302  
   303  	// sourcesTimer helps collect multiple events that happen in a quick succession. We start with 1ms as we don't care much
   304  	// at this point. In the select block, however, every time we receive an event, we reset the sourcesTimer to watch for
   305  	// 100ms since receiving that event. This is done because a single filesystem event by the user triggers multiple
   306  	// events for fsnotify. It's a known-issue, but not really bug. For more info look at below issues:
   307  	//    - https://github.com/fsnotify/fsnotify/issues/122
   308  	//    - https://github.com/fsnotify/fsnotify/issues/344
   309  	sourcesTimer := time.NewTimer(time.Millisecond)
   310  	<-sourcesTimer.C
   311  
   312  	devfileTimer := time.NewTimer(time.Millisecond)
   313  	<-devfileTimer.C
   314  
   315  	deployTimer := time.NewTimer(time.Millisecond)
   316  	<-deployTimer.C
   317  
   318  	retryTimer := time.NewTimer(time.Millisecond)
   319  	<-retryTimer.C
   320  
   321  	podsPhases := NewPodPhases()
   322  
   323  	for {
   324  		select {
   325  		case event := <-sourcesWatcher.Events:
   326  			events = append(events, event)
   327  			// We are waiting for more events in this interval
   328  			sourcesTimer.Reset(100 * time.Millisecond)
   329  		case <-sourcesTimer.C:
   330  			// timer has fired
   331  			if !componentCanSyncFile(componentStatus.State) {
   332  				continue
   333  			}
   334  			// first find the files that have changed (also includes the ones newly created) or deleted
   335  			changedFiles, deletedPaths := evaluateChangesHandler(events, parameters.Path, parameters.FileIgnores, sourcesWatcher)
   336  			// process the changes and sync files with remote pod
   337  			if len(changedFiles) == 0 && len(deletedPaths) == 0 {
   338  				continue
   339  			}
   340  			componentStatus.State = StateSyncOutdated
   341  			fmt.Fprintf(out, "Pushing files...\n\n")
   342  			retry, err := processEventsHandler(changedFiles, deletedPaths, parameters, out, &componentStatus, expBackoff)
   343  			if err != nil {
   344  				return err
   345  			}
   346  			// empty the events to receive new events
   347  			if componentStatus.State == StateReady {
   348  				events = []fsnotify.Event{} // empty the events slice to capture new events
   349  			}
   350  
   351  			if retry != nil {
   352  				retryTimer.Reset(*retry)
   353  			} else {
   354  				retryTimer.Reset(time.Millisecond)
   355  				<-retryTimer.C
   356  			}
   357  
   358  		case watchErr := <-sourcesWatcher.Errors:
   359  			return watchErr
   360  
   361  		case ev := <-deploymentWatcher.ResultChan():
   362  			switch obj := ev.Object.(type) {
   363  			case *appsv1.Deployment:
   364  				klog.V(4).Infof("deployment watcher Event: Type: %s, name: %s, rv: %s, pods: %d\n",
   365  					ev.Type, obj.GetName(), obj.GetResourceVersion(), obj.Status.ReadyReplicas)
   366  				deployTimer.Reset(300 * time.Millisecond)
   367  
   368  			case *metav1.Status:
   369  				klog.V(4).Infof("Status: %+v\n", obj)
   370  			}
   371  
   372  		case <-deployTimer.C:
   373  			retry, err := processEventsHandler(nil, nil, parameters, out, &componentStatus, expBackoff)
   374  			if err != nil {
   375  				return err
   376  			}
   377  			if retry != nil {
   378  				retryTimer.Reset(*retry)
   379  			} else {
   380  				retryTimer.Reset(time.Millisecond)
   381  				<-retryTimer.C
   382  			}
   383  
   384  		case <-devfileWatcher.Events:
   385  			devfileTimer.Reset(100 * time.Millisecond)
   386  
   387  		case <-devfileTimer.C:
   388  			fmt.Fprintf(out, "Updating Component...\n\n")
   389  			retry, err := processEventsHandler(nil, nil, parameters, out, &componentStatus, expBackoff)
   390  			if err != nil {
   391  				return err
   392  			}
   393  			if retry != nil {
   394  				retryTimer.Reset(*retry)
   395  			} else {
   396  				retryTimer.Reset(time.Millisecond)
   397  				<-retryTimer.C
   398  			}
   399  
   400  		case <-retryTimer.C:
   401  			retry, err := processEventsHandler(nil, nil, parameters, out, &componentStatus, expBackoff)
   402  			if err != nil {
   403  				return err
   404  			}
   405  			if retry != nil {
   406  				retryTimer.Reset(*retry)
   407  			} else {
   408  				retryTimer.Reset(time.Millisecond)
   409  				<-retryTimer.C
   410  			}
   411  
   412  		case ev := <-podWatcher.ResultChan():
   413  			switch ev.Type {
   414  			case watch.Deleted:
   415  				pod, ok := ev.Object.(*corev1.Pod)
   416  				if !ok {
   417  					return errors.New("unable to decode watch event")
   418  				}
   419  				podsPhases.Delete(out, pod)
   420  			case watch.Added, watch.Modified:
   421  				pod, ok := ev.Object.(*corev1.Pod)
   422  				if !ok {
   423  					return errors.New("unable to decode watch event")
   424  				}
   425  				podsPhases.Add(out, pod.GetCreationTimestamp(), pod)
   426  			}
   427  
   428  		case ev := <-eventsWatcher.ResultChan():
   429  			switch kevent := ev.Object.(type) {
   430  			case *corev1.Event:
   431  				podName := kevent.InvolvedObject.Name
   432  				selector := labels.GetSelector(parameters.ComponentName, parameters.ApplicationName, labels.ComponentDevMode, true)
   433  				matching, err := o.kubeClient.IsPodNameMatchingSelector(ctx, podName, selector)
   434  				if err != nil {
   435  					return err
   436  				}
   437  				if matching {
   438  					log.Fwarning(out, kevent.Message)
   439  				}
   440  			}
   441  
   442  		case watchErr := <-devfileWatcher.Errors:
   443  			return watchErr
   444  
   445  		case <-ctx.Done():
   446  			return errors.New("Dev mode interrupted by user")
   447  		}
   448  	}
   449  }

   451  // evaluateFileChanges evaluates any file changes for the events. It ignores the files in fileIgnores slice related to path, and removes
   452  // any deleted paths from the watcher
   453  func evaluateFileChanges(events []fsnotify.Event, path string, fileIgnores []string, watcher *fsnotify.Watcher) ([]string, []string) {
   454  	var changedFiles []string
   455  	var deletedPaths []string
   456  
   457  	ignoreMatcher := gitignore.CompileIgnoreLines(fileIgnores...)
   458  
   459  	for _, event := range events {
   460  		klog.V(4).Infof("filesystem watch event: %s", event)
   461  		isIgnoreEvent := shouldIgnoreEvent(event)
   462  
   463  		// add file name to changedFiles only once
   464  		alreadyInChangedFiles := false
   465  		for _, cfile := range changedFiles {
   466  			if cfile == event.Name {
   467  				alreadyInChangedFiles = true
   468  				break
   469  			}
   470  		}
   471  
   472  		// Filter out anything in ignores list from the list of changed files
   473  		// This is important in spite of not watching the
   474  		// ignores paths because, when a directory that is ignored, is deleted,
   475  		// because its parent is watched, the fsnotify automatically raises an event
   476  		// for it.
   477  		var watchError error
   478  		rel, err := filepath.Rel(path, event.Name)
   479  		if err != nil {
   480  			watchError = fmt.Errorf("unable to get relative path of %q on %q", event.Name, path)
   481  		}
   482  		matched := ignoreMatcher.MatchesPath(rel)
   483  		if !alreadyInChangedFiles && !matched && !isIgnoreEvent {
   484  			// Append the new file change event to changedFiles if and only if the event is not a file remove event
   485  			if event.Op&fsnotify.Remove != fsnotify.Remove {
   486  				changedFiles = append(changedFiles, event.Name)
   487  			}
   488  		}
   489  
   490  		// Rename operation triggers RENAME event on old path + CREATE event for renamed path so delete old path in case of rename
   491  		// Also weirdly, fsnotify raises a RENAME event for deletion of files/folders with space in their name so even that should be handled here
   492  		if event.Op&fsnotify.Remove == fsnotify.Remove || event.Op&fsnotify.Rename == fsnotify.Rename {
   493  			// On remove/rename, stop watching the resource
   494  			if e := watcher.Remove(event.Name); e != nil {
   495  				klog.V(4).Infof("error removing watch for %s: %v", event.Name, e)
   496  			}
   497  			// Append the file to list of deleted files
   498  			// When a file/folder is deleted, it raises 2 events:
   499  			//	a. RENAME with event.Name empty
   500  			//	b. REMOVE with event.Name as file name
   501  			if !alreadyInChangedFiles && !matched && event.Name != "" {
   502  				deletedPaths = append(deletedPaths, event.Name)
   503  			}
   504  		} else {
   505  			// On other ops, recursively watch the resource (if applicable)
   506  			if e := addRecursiveWatch(watcher, path, event.Name, fileIgnores); e != nil && watchError == nil {
   507  				klog.V(4).Infof("Error occurred in addRecursiveWatch, setting watchError to %v", e)
   508  				watchError = e
   509  			}
   510  		}
   511  	}
   512  	deletedPaths = removeDuplicates(deletedPaths)
   513  
   514  	return changedFiles, deletedPaths
   515  }

   518  	changedFiles, deletedPaths []string,
   519  	parameters WatchParameters,
   520  	out io.Writer,
   521  	componentStatus *ComponentStatus,

   546  		ErrOut:                   parameters.ErrOut,
   547  	}
   548  	oldStatus := *componentStatus
   549  	err := parameters.DevfileWatchHandler(pushParams, parameters, componentStatus)
   550  	if err != nil {
   551  		if isFatal(err) {
   552  			return nil, err
   553  		}
   554  		klog.V(4).Infof("Error from Push: %v", err)
   555  		if parameters.WatchFiles {
   556  			// Log and output, but intentionally not exiting on error here.
   557  			// We don't want to break watch when push failed, it might be fixed with the next change.
   558  			fmt.Fprintf(out, "%s - %s\n\n", PushErrorString, err.Error())
   559  		} else {
   560  			return nil, err
   561  		}
   562  		wait := backoff.Delay()
   563  		return &wait, nil
   564  	}
   565  	backoff.Reset()
   566  	if oldStatus.State != StateReady && componentStatus.State == StateReady ||
   567  		!reflect.DeepEqual(oldStatus.EndpointsForwarded, componentStatus.EndpointsForwarded) {
   568  
   569  		printInfoMessage(out, parameters.Path)
   570  	}
   571  	return nil, nil

   597  func shouldIgnoreEvent(event fsnotify.Event) (ignoreEvent bool) {
   598  	if !(event.Op&fsnotify.Remove == fsnotify.Remove || event.Op&fsnotify.Rename == fsnotify.Rename) {
   599  		stat, err := os.Lstat(event.Name)
   600  		if err != nil {
   601  			// Some of the editors like vim and gedit, generate temporary buffer files during update to the file and deletes it soon after exiting from the editor
   602  			// So, its better to log the error rather than feeding it to error handler via `watchError = fmt.Errorf("unable to watch changes: %w", err)`,
   603  			// which will terminate the watch
   604  			klog.V(4).Infof("Failed getting details of the changed file %s. Ignoring the change", event.Name)
   605  		}
   606  		// Some of the editors generate temporary buffer files during update to the file and deletes it soon after exiting from the editor
   607  		// So, its better to log the error rather than feeding it to error handler via `watchError = fmt.Errorf("unable to watch changes: %w", err)`,
   608  		// which will terminate the watch
   609  		if stat == nil {
   610  			klog.V(4).Infof("Ignoring event for file %s as details about the file couldn't be fetched", event.Name)
   611  			ignoreEvent = true
   612  		}
   613  
   614  		// In windows, every new file created under a sub-directory of the watched directory, raises 2 events:
   615  		// 1. Write event for the directory under which the file was created
   616  		// 2. Create event for the file that was created
   617  		// Ignore 1 to avoid duplicate events.
   618  		if ignoreEvent || (stat.IsDir() && event.Op&fsnotify.Write == fsnotify.Write) {
   619  			ignoreEvent = true
   620  		}
   621  	}
   622  	return ignoreEvent
   623  }
