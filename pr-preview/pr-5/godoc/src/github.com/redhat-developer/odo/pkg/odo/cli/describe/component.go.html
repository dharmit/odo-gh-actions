    62  func (o *ComponentOptions) Complete(cmdline cmdline.Cmdline, args []string) (err error) {
    63  	// 1. Name is not passed, and odo has access to devfile.yaml; Name is not passed so we assume that odo has access to the devfile.yaml
    64  	if o.nameFlag == "" {
    65  
    66  		if len(o.namespaceFlag) > 0 && len(o.nameFlag) == 0 {
    67  			return errors.New("--namespace can be used only with --name")
    68  		}
    69  
    70  		o.Context, err = genericclioptions.New(genericclioptions.NewCreateParameters(cmdline).NeedDevfile(""))
    71  		if err != nil {
    72  			return err
    73  		}
    74  		// this ensures that the namespace set in env.yaml is used
    75  		o.clientset.KubernetesClient.SetNamespace(o.GetProject())
    76  		return nil
    77  	}
    78  	// 2. Name is passed, and odo does not have access to devfile.yaml; if Name is passed, then we assume that odo does not have access to the devfile.yaml
    79  	if o.namespaceFlag != "" {
    80  		o.clientset.KubernetesClient.SetNamespace(o.namespaceFlag)
    81  	} else {
    82  		o.namespaceFlag = o.clientset.KubernetesClient.GetCurrentNamespace()
    83  	}
    84  	return nil
    85  }

   113  func (o *ComponentOptions) describeNamedComponent(name string) (result api.Component, devfileObj *parser.DevfileObj, err error) {
   114  	runningIn, err := component.GetRunningModes(o.clientset.KubernetesClient, name)
   115  	if err != nil {
   116  		return api.Component{}, nil, err
   117  	}
   118  	devfile, err := component.GetDevfileInfoFromCluster(o.clientset.KubernetesClient, name)
   119  	if err != nil {
   120  		return api.Component{}, nil, err
   121  	}
   122  	return api.Component{
   123  		DevfileData: &api.DevfileData{
   124  			Devfile: devfile.Data,
   125  		},
   126  		RunningIn: runningIn,
   127  		ManagedBy: "odo",
   128  	}, &devfile, nil
   129  }

   131  // describeDevfileComponent describes the component defined by the devfile in the current directory
   132  func (o *ComponentOptions) describeDevfileComponent() (result api.Component, devfile *parser.DevfileObj, err error) {
   133  	devfileObj := o.EnvSpecificInfo.GetDevfileObj()
   134  	path, err := filepath.Abs(".")
   135  	if err != nil {
   136  		return api.Component{}, nil, err
   137  	}
   138  	forwardedPorts, err := o.clientset.StateClient.GetForwardedPorts()
   139  	if err != nil {
   140  		return api.Component{}, nil, err
   141  	}
   142  	runningIn, err := component.GetRunningModes(o.clientset.KubernetesClient, devfileObj.GetMetadataName())
   143  	if err != nil {
   144  		if !errors.As(err, &component.NoComponentFoundError{}) {
   145  			return api.Component{}, nil, err
   146  		} else {
   147  			// it is ok if the component is not deployed
   148  			runningIn = nil
   149  		}
   150  	}
   151  	return api.Component{
   152  		DevfilePath:       filepath.Join(path, o.Context.GetDevfilePath()),
   153  		DevfileData:       api.GetDevfileData(devfileObj),
   154  		DevForwardedPorts: forwardedPorts,
   155  		RunningIn:         runningIn,
   156  		ManagedBy:         "odo",
   157  	}, &devfileObj, nil
   158  }

   160  func printHumanReadableOutput(cmp api.Component, devfileObj *parser.DevfileObj) error {
   161  	if cmp.DevfileData != nil {
   162  		log.Describef("Name: ", cmp.DevfileData.Devfile.GetMetadata().Name)
   163  		log.Describef("Display Name: ", cmp.DevfileData.Devfile.GetMetadata().DisplayName)
   164  		log.Describef("Project Type: ", cmp.DevfileData.Devfile.GetMetadata().ProjectType)
   165  		log.Describef("Language: ", cmp.DevfileData.Devfile.GetMetadata().Language)
   166  		log.Describef("Version: ", cmp.DevfileData.Devfile.GetMetadata().Version)
   167  		log.Describef("Description: ", cmp.DevfileData.Devfile.GetMetadata().Description)
   168  		log.Describef("Tags: ", strings.Join(cmp.DevfileData.Devfile.GetMetadata().Tags, ", "))
   169  		fmt.Println()
   170  	}
   171  
   172  	log.Describef("Running in: ", cmp.RunningIn.String())
   173  	fmt.Println()
   174  
   175  	if len(cmp.DevForwardedPorts) > 0 {
   176  		log.Info("Forwarded ports:")
   177  		for _, port := range cmp.DevForwardedPorts {
   178  			log.Printf("%s:%d -> %s:%d", port.LocalAddress, port.LocalPort, port.ContainerName, port.ContainerPort)
   179  		}
   180  		fmt.Println()
   181  	}
   182  
   183  	log.Info("Supported odo features:")
   184  	if cmp.DevfileData != nil && cmp.DevfileData.SupportedOdoFeatures != nil {
   185  		log.Printf("Dev: %v", cmp.DevfileData.SupportedOdoFeatures.Dev)
   186  		log.Printf("Deploy: %v", cmp.DevfileData.SupportedOdoFeatures.Deploy)
   187  		log.Printf("Debug: %v", cmp.DevfileData.SupportedOdoFeatures.Debug)
   188  	} else {
   189  		log.Printf("Dev: Unknown")
   190  		log.Printf("Deploy: Unknown")
   191  		log.Printf("Debug: Unknown")
   192  	}
   193  	fmt.Println()
   194  
   195  	err := listComponentsNames("Container components:", devfileObj, v1alpha2.ContainerComponentType)
   196  	if err != nil {
   197  		return err
   198  	}
   199  
   200  	err = listComponentsNames("Kubernetes components:", devfileObj, v1alpha2.KubernetesComponentType)
   201  	if err != nil {
   202  		return err
   203  	}
   204  	return nil
   205  }

   207  func listComponentsNames(title string, devfileObj *parser.DevfileObj, typ v1alpha2.ComponentType) error {
   208  	if devfileObj == nil {
   209  		log.Describef(title, " Unknown")
   210  		return nil
   211  	}
   212  	containers, err := devfileObj.Data.GetComponents(common.DevfileOptions{
   213  		ComponentOptions: common.ComponentOptions{ComponentType: typ},
   214  	})
   215  	if err != nil {
   216  		return err
   217  	}
   218  	if len(containers) == 0 {
   219  		return nil
   220  	}
   221  	log.Info(title)
   222  	for _, container := range containers {
   223  		log.Printf("%s", container.Name)
   224  	}
   225  	fmt.Println()
   226  	return nil
   227  }
