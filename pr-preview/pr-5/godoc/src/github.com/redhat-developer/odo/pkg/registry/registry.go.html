    58  // if registry name is specified return the specific registry, otherwise return all registries
    59  func (o RegistryClient) GetDevfileRegistries(registryName string) ([]api.Registry, error) {
    60  	var devfileRegistries []api.Registry
    61  
    62  	hasName := len(registryName) != 0
    63  	if o.preferenceClient.RegistryList() != nil {
    64  		registryList := *o.preferenceClient.RegistryList()
    65  		// Loop backwards here to ensure the registry display order is correct (display latest newly added registry firstly)
    66  		for i := len(registryList) - 1; i >= 0; i-- {
    67  			registry := registryList[i]
    68  			if hasName {
    69  				if registryName == registry.Name {
    70  					reg := api.Registry{
    71  						Name:   registry.Name,
    72  						URL:    registry.URL,
    73  						Secure: registry.Secure,
    74  					}
    75  					devfileRegistries = append(devfileRegistries, reg)
    76  					return devfileRegistries, nil
    77  				}
    78  			} else {
    79  				reg := api.Registry{
    80  					Name:   registry.Name,
    81  					URL:    registry.URL,
    82  					Secure: registry.Secure,
    83  				}
    84  				devfileRegistries = append(devfileRegistries, reg)
    85  			}
    86  		}
    87  	} else {
    88  		return nil, nil
    89  	}
    90  
    91  	return devfileRegistries, nil
    92  }

    94  // ListDevfileStacks lists all the available devfile stacks in devfile registry
    95  func (o RegistryClient) ListDevfileStacks(registryName, devfileFlag, filterFlag string, detailsFlag bool) (DevfileStackList, error) {
    96  	catalogDevfileList := &DevfileStackList{}
    97  	var err error
    98  
    99  	// TODO: consider caching registry information for better performance since it should be fairly stable over time
   100  	// Get devfile registries
   101  	catalogDevfileList.DevfileRegistries, err = o.GetDevfileRegistries(registryName)
   102  	if err != nil {
   103  		return *catalogDevfileList, err
   104  	}
   105  	if catalogDevfileList.DevfileRegistries == nil {
   106  		return *catalogDevfileList, nil
   107  	}
   108  
   109  	// first retrieve the indices for each registry, concurrently
   110  	devfileIndicesMutex := &sync.Mutex{}
   111  	retrieveRegistryIndices := util.NewConcurrentTasks(len(catalogDevfileList.DevfileRegistries))
   112  
   113  	// The 2D slice index is the priority of the registry (highest priority has highest index)
   114  	// and the element is the devfile slice that belongs to the registry
   115  	registrySlice := make([][]api.DevfileStack, len(catalogDevfileList.DevfileRegistries))
   116  	for regPriority, reg := range catalogDevfileList.DevfileRegistries {
   117  		// Load the devfile registry index.json
   118  		registry := reg                 // Needed to prevent the lambda from capturing the value
   119  		registryPriority := regPriority // Needed to prevent the lambda from capturing the value
   120  		retrieveRegistryIndices.Add(util.ConcurrentTask{ToRun: func(errChannel chan error) {
   121  			registryDevfiles, err := getRegistryStacks(o.preferenceClient, registry)
   122  			if err != nil {
   123  				log.Warningf("Registry %s is not set up properly with error: %v, please check the registry URL, and credential and remove add the registry again (refer to `odo preference add registry --help`)\n", registry.Name, err)
   124  				return
   125  			}
   126  
   127  			devfileIndicesMutex.Lock()
   128  			registrySlice[registryPriority] = registryDevfiles
   129  			devfileIndicesMutex.Unlock()
   130  		}})
   131  	}
   132  	if err := retrieveRegistryIndices.Run(); err != nil {
   133  		return *catalogDevfileList, err
   134  	}
   135  
   136  	// Go through all the devfiles and filter based on:
   137  	// What's in the name or description
   138  	// The exact name of the devfile
   139  	//
   140  	// We also add additional details such as supported odo features (which we
   141  	// manually http get) if the details flag has been passed in.
   142  	for priorityNumber, registryDevfiles := range registrySlice {
   143  
   144  		devfiles := []api.DevfileStack{}
   145  
   146  		for _, devfile := range registryDevfiles {
   147  
   148  			// Add the "priority" of the registry to the devfile
   149  			devfile.Registry.Priority = priorityNumber
   150  
   151  			if filterFlag != "" {
   152  				if !strings.Contains(devfile.Name, filterFlag) && !strings.Contains(devfile.Description, filterFlag) {
   153  					continue
   154  				}
   155  			}
   156  
   157  			if devfileFlag != "" {
   158  				if devfileFlag != devfile.Name {
   159  					continue
   160  				}
   161  			}
   162  
   163  			if detailsFlag {
   164  				devfileData, err := o.retrieveDevfileDataFromRegistry(devfile.Registry.Name, devfile.Name)
   165  				if err != nil {
   166  					return *catalogDevfileList, err
   167  				}
   168  				devfile.DevfileData = &devfileData
   169  			}
   170  
   171  			devfiles = append(devfiles, devfile)
   172  		}
   173  
   174  		catalogDevfileList.Items = append(catalogDevfileList.Items, devfiles...)
   175  	}
   176  
   177  	// Sort catalogDevfileList.Items by:
   178  	// 1. Priority of the registry (highest priority has highest index)
   179  	// 2. Name of the devfile
   180  	sort.Slice(catalogDevfileList.Items[:], func(i, j int) bool {
   181  		if catalogDevfileList.Items[i].Name == catalogDevfileList.Items[j].Name {
   182  			return catalogDevfileList.Items[i].Registry.Priority < catalogDevfileList.Items[j].Registry.Priority
   183  		}
   184  		return catalogDevfileList.Items[i].Name < catalogDevfileList.Items[j].Name
   185  	})
   186  
   187  	return *catalogDevfileList, nil
   188  }

   207  func createRegistryDevfiles(registry api.Registry, devfileIndex []indexSchema.Schema) ([]api.DevfileStack, error) {
   208  	registryDevfiles := make([]api.DevfileStack, 0, len(devfileIndex))
   209  	for _, devfileIndexEntry := range devfileIndex {
   210  		stackDevfile := api.DevfileStack{
   211  			Name:            devfileIndexEntry.Name,
   212  			DisplayName:     devfileIndexEntry.DisplayName,
   213  			Description:     devfileIndexEntry.Description,
   214  			Registry:        registry,
   215  			Language:        devfileIndexEntry.Language,
   216  			Tags:            devfileIndexEntry.Tags,
   217  			ProjectType:     devfileIndexEntry.ProjectType,
   218  			StarterProjects: devfileIndexEntry.StarterProjects,
   219  			Version:         devfileIndexEntry.Version,
   220  		}
   221  		registryDevfiles = append(registryDevfiles, stackDevfile)
   222  	}
   223  
   224  	return registryDevfiles, nil

   227  func (o RegistryClient) retrieveDevfileDataFromRegistry(registryName string, devfileName string) (api.DevfileData, error) {
   228  
   229  	// Create random temporary file
   230  	tmpFile, err := ioutil.TempDir("", "odo")
   231  	if err != nil {
   232  		return api.DevfileData{}, err
   233  	}
   234  	defer os.Remove(tmpFile)
   235  
   236  	registries := o.preferenceClient.RegistryList()
   237  	var reg preference.Registry
   238  
   239  	// Get the file and save it to the temporary file
   240  	// Why do we do that?
   241  	// 1. We need to get the file from the registry
   242  	// 2. The devfile api library does not support saving in memory
   243  	// 3. We need to get the file from the registry and save it to the temporary file
   244  	// 4. We need to read the file from the temporary file, unmarshal it and then return the devfile data
   245  	for _, reg = range *registries {
   246  		if reg.Name == registryName {
   247  			err = o.PullStackFromRegistry(reg.URL, devfileName, tmpFile, segment.GetRegistryOptions())
   248  			if err != nil {
   249  				return api.DevfileData{}, err
   250  			}
   251  		}
   252  	}
   253  
   254  	// Get the devfile yaml file from the directory
   255  	devfileYamlFile := location.DevfileFilenamesProvider(tmpFile)
   256  
   257  	// Parse and validate the file and return the devfile data
   258  	devfileObj, err := devfile.ParseAndValidateFromFile(path.Join(tmpFile, devfileYamlFile))
   259  	if err != nil {
   260  		return api.DevfileData{}, err
   261  	}
   262  
   263  	// Convert DevfileObj to DevfileData
   264  	// use api.GetDevfileData to get supported features
   265  	return *api.GetDevfileData(devfileObj), nil
   266  }
