   135  // printSingleBindingHumanReadableOutput prints information about a binding and returns true if status is unknown
   136  func printSingleBindingHumanReadableOutput(binding api.ServiceBinding) bool {
   137  	log.Describef("Service Binding Name: ", binding.Name)
   138  	log.Info("Services:")
   139  	for _, service := range binding.Spec.Services {
   140  		gvk := schema.FromAPIVersionAndKind(service.APIVersion, service.Kind)
   141  		if service.Namespace != "" {
   142  			log.Printf("%s (%s.%s) (namespace: %s)", service.Name, gvk.Kind, gvk.Group, service.Namespace)
   143  		} else {
   144  			log.Printf("%s (%s.%s)", service.Name, gvk.Kind, gvk.Group)
   145  		}
   146  	}
   147  	log.Describef("Bind as files: ", strconv.FormatBool(binding.Spec.BindAsFiles))
   148  	log.Describef("Detect binding resources: ", strconv.FormatBool(binding.Spec.DetectBindingResources))
   149  
   150  	if binding.Spec.NamingStrategy != "" {
   151  		log.Describef("Naming strategy: ", binding.Spec.NamingStrategy)
   152  	}
   153  
   154  	if binding.Status == nil {
   155  		log.Describef("Available binding information: ", "unknown")
   156  		return true
   157  	}
   158  	log.Info("Available binding information:")
   159  	for _, info := range binding.Status.BindingFiles {
   160  		log.Printf(info)
   161  	}
   162  	for _, info := range binding.Status.BindingEnvVars {
   163  		log.Printf(info)
   164  	}
   165  	return false
   166  }

   168  func printBindingsHumanReadableOutput(bindings []api.ServiceBinding) {
   169  	if len(bindings) == 0 {
   170  		log.Info("No ServiceBinding used by the current component")
   171  		return
   172  	}
   173  
   174  	log.Info("ServiceBinding used by the current component:")
   175  	someStatusUnknown := false
   176  	for _, binding := range bindings {
   177  		fmt.Println()
   178  		statusUnknown := printSingleBindingHumanReadableOutput(binding)
   179  		if statusUnknown {
   180  			someStatusUnknown = true
   181  		}
   182  	}
   183  	if someStatusUnknown {
   184  		fmt.Println()
   185  		log.Info(`Binding information for one or more ServiceBinding is not available because they don't exist on the cluster yet.
   186  Start "odo dev" first to see binding information.`)
   187  	}
   188  }
