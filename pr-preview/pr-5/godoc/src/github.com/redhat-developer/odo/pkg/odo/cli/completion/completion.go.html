    59  // NewCmdCompletion implements the utils completion odo command
    60  func NewCmdCompletion(name, fullName string) *cobra.Command {
    61  	completionCmd := &cobra.Command{
    62  		Use:                   name,
    63  		Short:                 "Add odo completion support to your development environment",
    64  		Long:                  completionLongDesc,
    65  		Example:               fmt.Sprintf(completionExample, fullName),
    66  		DisableFlagsInUseLine: true,
    67  		ValidArgs:             []string{"bash", "zsh", "fish", "powershell"},
    68  		Args:                  cobra.ExactValidArgs(1),
    69  		Run: func(cmd *cobra.Command, args []string) {
    70  			// Below we ignore the error returns in order to pass golint validation
    71  			// We will handle the error in the main function / output when the user inputs `odo completion`.
    72  			switch args[0] {
    73  			case "bash":
    74  				_ = cmd.Root().GenBashCompletion(os.Stdout)
    75  			case "zsh":
    76  				// Due to https://github.com/spf13/cobra/issues/1529 we cannot load zsh
    77  				// via using source, so we need to add compdef to the beginning of the output so we can easily do:
    78  				// source <(odo completion zsh)
    79  				zsh := "#compdef odo\ncompdef _odo odo\n"
    80  				out := os.Stdout
    81  				_, _ = out.Write([]byte(zsh))
    82  				_ = cmd.Root().GenZshCompletion(out)
    83  			case "fish":
    84  				_ = cmd.Root().GenFishCompletion(os.Stdout, true)
    85  			case "powershell":
    86  				_ = cmd.Root().GenPowerShellCompletionWithDesc(os.Stdout)
    87  			}
    88  		},
    89  	}
    90  
    91  	completionCmd.SetUsageTemplate(util.CmdUsageTemplate)
    92  	completionCmd.Annotations = map[string]string{"command": "utility"}
    93  	return completionCmd
    94  }
