    34  	testComponentName := "test"
    35  	testAppName := "app"
    36  	deployment := v1.Deployment{

    48  	tests := []struct {

    75  		},

    77  			name:          "Case 4: Valid devfile, already running component",

    83  	}
    84  	for _, tt := range tests {
    85  		t.Run(tt.name, func(t *testing.T) {
    86  			var comp devfilev1.Component

    88  				odolabels.SetProjectType(deployment.Annotations, string(tt.componentType))
    89  				comp = testingutil.GetFakeContainerComponent("component")
    90  			}
    91  			devObj := devfileParser.DevfileObj{
    92  				Data: func() data.DevfileData {
    93  					devfileData, err := data.NewDevfileData(string(data.APISchemaVersion200))
    94  					if err != nil {
    95  						t.Error(err)
    96  					}
    97  					metadata := devfileData.GetMetadata()
    98  					metadata.ProjectType = string(tt.componentType)
    99  					err = devfileData.AddComponents([]devfilev1.Component{comp})
   100  					if err != nil {
   101  						t.Error(err)
   102  					}
   103  					err = devfileData.AddCommands([]devfilev1.Command{getExecCommand("run", devfilev1.RunCommandGroupKind)})
   104  					if err != nil {
   105  						t.Error(err)
   106  					}
   107  					return devfileData
   108  				}(),
   109  			}
   110  
   111  			adapterCtx := AdapterContext{
   112  				ComponentName: testComponentName,
   113  				AppName:       testAppName,
   114  				Devfile:       devObj,
   115  			}
   116  
   117  			fkclient, fkclientset := kclient.FakeNew()
   118

   120  				return true, &deployment, nil
   121  			})
   122  
   123  			if tt.running {
   124  				fkclientset.Kubernetes.PrependReactor("get", "deployments", func(action ktesting.Action) (bool, runtime.Object, error) {
   125  					return true, &deployment, nil
   126  				})
   127  			}
   128  			tt.envInfo.EnvInfo = *envinfo.GetFakeEnvInfo(envinfo.ComponentSettings{
   129  				Name:    testComponentName,
   130  				AppName: testAppName,
   131  			})
   132  			ctrl := gomock.NewController(t)
   133  			fakePrefClient := preference.NewMockClient(ctrl)
   134  			fakePrefClient.EXPECT().GetEphemeralSourceVolume()

   136  			_, _, err := componentAdapter.createOrUpdateComponent(tt.running, tt.envInfo, libdevfile.DevfileCommands{}, 0, nil)
   137  
   138  			// Checks for unexpected error cases
