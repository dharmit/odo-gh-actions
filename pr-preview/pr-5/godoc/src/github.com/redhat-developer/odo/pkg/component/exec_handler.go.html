    49  func (o *execHandler) Execute(command v1alpha2.Command) error {
    50  	msg := o.msg
    51  	if msg == "" {
    52  		msg = fmt.Sprintf("Executing %s command %q on container %q", command.Id, command.Exec.CommandLine, command.Exec.Component)
    53  	} else {
    54  		msg += " (command: " + command.Id + ")"
    55  	}
    56  	spinner := log.Spinner(msg)
    57  	defer spinner.End(false)
    58  
    59  	logger := machineoutput.NewMachineEventLoggingClient()
    60  	stdoutWriter, stdoutChannel, stderrWriter, stderrChannel := logger.CreateContainerOutputWriter()
    61  
    62  	cmdline := getCmdline(command)
    63  	_, _, err := remotecmd.ExecuteCommand(cmdline, o.kubeClient, o.podName, command.Exec.Component, o.show, stdoutWriter, stderrWriter)
    64  
    65  	closeWriterAndWaitForAck(stdoutWriter, stdoutChannel, stderrWriter, stderrChannel)
    66  
    67  	spinner.End(err == nil)
    68  	if err != nil {
    69  		rd, errLog := Log(o.kubeClient, o.componentName, o.appName, false, command)
    70  		if errLog != nil {
    71  			return fmt.Errorf("unable to log error %v: %w", err, errLog)
    72  		}
    73  
    74  		// Use GetStderr in order to make sure that colour output is correct
    75  		// on non-TTY terminals
    76  		errLog = util.DisplayLog(false, rd, log.GetStderr(), o.componentName, -1)
    77  		if errLog != nil {
    78  			return fmt.Errorf("unable to log error %v: %w", err, errLog)
    79  		}
    80  	}
    81  	return err
    82  }

    84  func getCmdline(command v1alpha2.Command) []string {
    85  	// deal with environment variables
    86  	var cmdLine string
    87  	setEnvVariable := util.GetCommandStringFromEnvs(command.Exec.Env)
    88  
    89  	if setEnvVariable == "" {
    90  		cmdLine = command.Exec.CommandLine
    91  	} else {
    92  		cmdLine = setEnvVariable + " && " + command.Exec.CommandLine
    93  	}
    94  
    95  	// Change to the workdir and execute the command
    96  	// Redirecting to /proc/1/fd/* allows to redirect the process output to the output streams of PID 1 process inside the container.
    97  	// This way, returning the container logs with 'odo logs' or 'kubectl logs' would work seamlessly.
    98  	// See https://stackoverflow.com/questions/58716574/where-exactly-do-the-logs-of-kubernetes-pods-come-from-at-the-container-level
    99  	redirectString := "1>>/proc/1/fd/1 2>>/proc/1/fd/2"
   100  	var cmd []string
   101  	if command.Exec.WorkingDir != "" {
   102  		// since we are using /bin/sh -c, the command needs to be within a single double quote instance, for example "cd /tmp && pwd"
   103  		cmd = []string{ShellExecutable, "-c", "cd " + command.Exec.WorkingDir + " && (" + cmdLine + ") " + redirectString}
   104  	} else {
   105  		cmd = []string{ShellExecutable, "-c", "(" + cmdLine + ") " + redirectString}
   106  	}
   107  	return cmd
   108  }
