    23  // ExecCMDInContainer execute command in the container of a pod, pass an empty string for containerName to execute in the first container of the pod
    24  func (c *Client) ExecCMDInContainer(containerName, podName string, cmd []string, stdout io.Writer, stderr io.Writer, stdin io.Reader, tty bool) error {
    25  	podExecOptions := corev1.PodExecOptions{
    26  		Command: cmd,
    27  		Stdin:   stdin != nil,
    28  		Stdout:  stdout != nil,
    29  		Stderr:  stderr != nil,
    30  		TTY:     tty,
    31  	}
    32  
    33  	// If a container name was passed in, set it in the exec options, otherwise leave it blank
    34  	if containerName != "" {
    35  		podExecOptions.Container = containerName
    36  	}
    37  
    38  	req := c.KubeClient.CoreV1().RESTClient().
    39  		Post().
    40  		Namespace(c.Namespace).
    41  		Resource("pods").
    42  		Name(podName).
    43  		SubResource("exec").
    44  		VersionedParams(&podExecOptions, scheme.ParameterCodec)
    45  
    46  	config, err := c.KubeConfig.ClientConfig()
    47  	if err != nil {
    48  		return fmt.Errorf("unable to get Kubernetes client config: %w", err)
    49  	}
    50  
    51  	// Connect to url (constructed from req) using SPDY (HTTP/2) protocol which allows bidirectional streams.
    52  	exec, err := remotecommand.NewSPDYExecutor(config, "POST", req.URL())
    53  	if err != nil {
    54  		return fmt.Errorf("unable execute command via SPDY: %w", err)
    55  	}
    56  	// initialize the transport of the standard shell streams
    57  	err = exec.Stream(remotecommand.StreamOptions{
    58  		Stdin:  stdin,
    59  		Stdout: stdout,
    60  		Stderr: stderr,
    61  		Tty:    tty,
    62  	})
    63  	if err != nil {
    64  		return fmt.Errorf("error while streaming command: %w", err)
    65  	}
    66  
    67  	return nil
    68  }

    70  // ExtractProjectToComponent extracts the project archive(tar) to the target path from the reader stdin
    71  func (c *Client) ExtractProjectToComponent(containerName, podName string, targetPath string, stdin io.Reader) error {
    72  	// cmdArr will run inside container
    73  	cmdArr := []string{"tar", "xf", "-", "-C", targetPath, "--no-same-owner"}
    74  	var stdout bytes.Buffer
    75  	var stderr bytes.Buffer
    76  	klog.V(3).Infof("Executing command %s", strings.Join(cmdArr, " "))
    77  	err := c.ExecCMDInContainer(containerName, podName, cmdArr, &stdout, &stderr, stdin, false)
    78  	if err != nil {
    79  		log.Errorf("Command '%s' in container failed.\n", strings.Join(cmdArr, " "))
    80  		log.Errorf("stdout: %s\n", stdout.String())
    81  		log.Errorf("stderr: %s\n", stderr.String())
    82  		log.Errorf("err: %s\n", err.Error())
    83  		return err
    84  	}
    85  	return nil
    86  }

    94  // GetRunningPodFromSelector gets a pod from the selector
    95  func (c *Client) GetRunningPodFromSelector(selector string) (*corev1.Pod, error) {
    96  	pods, err := c.KubeClient.CoreV1().Pods(c.Namespace).List(context.TODO(), metav1.ListOptions{
    97  		LabelSelector: selector,
    98  		FieldSelector: "status.phase=Running",
    99  	})
   100  	if err != nil {
   101  		// Don't wrap error since we want to know if it's a forbidden error
   102  		// if we wrap, we lose the err status reason and callers of this api rely on it
   103  		return nil, err
   104  	}
   105  	numPods := len(pods.Items)
   106  	if numPods == 0 {
   107  		return nil, &PodNotFoundError{Selector: selector}
   108  	} else if numPods > 1 {
   109  		return nil, fmt.Errorf("multiple Pods exist for the selector: %v. Only one must be present", selector)
   110  	}
   111  
   112  	// check if the pod is in the terminating state
   113  	if pods.Items[0].DeletionTimestamp != nil {
   114  		return nil, &PodNotFoundError{Selector: selector}
   115  	}
   116  
   117  	return &pods.Items[0], nil
   118  }

   120  // GetPodLogs prints the log from pod to stdout
   121  func (c *Client) GetPodLogs(podName, containerName string, followLog bool) (io.ReadCloser, error) {
   122  
   123  	// Set standard log options
   124  	podLogOptions := corev1.PodLogOptions{Follow: false, Container: containerName}
   125  
   126  	// If the log is being followed, set it to follow / don't wait
   127  	if followLog {
   128  		podLogOptions = corev1.PodLogOptions{
   129  			Follow:    true,
   130  			Previous:  false,
   131  			Container: containerName,
   132  		}
   133  	}
   134  
   135  	// RESTClient call to kubernetes
   136  	rd, err := c.KubeClient.CoreV1().RESTClient().Get().
   137  		Namespace(c.Namespace).
   138  		Name(podName).
   139  		Resource("pods").
   140  		SubResource("log").
   141  		VersionedParams(&podLogOptions, scheme.ParameterCodec).
   142  		Stream(context.TODO())
   143  
   144  	return rd, err
   145  }
