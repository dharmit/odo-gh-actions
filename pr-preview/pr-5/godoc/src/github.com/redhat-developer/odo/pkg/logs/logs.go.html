    62  func (o *LogsClient) getLogsForMode(
    63  	events Events,
    64  	mode string,
    65  	componentName string,
    66  	namespace string,
    67  	follow bool,
    68  ) {
    69  	var selector string
    70  	podChan := make(chan corev1.Pod) // grab the logs of the pod put on this channel
    71  	errChan := make(chan error)
    72  	doneChan := make(chan struct{}) // because populating doneChan directly would cause odo logs to exit prematurely.
    73  
    74  	go func() {
    75  		// this go routine gets the logs of the pods put on the podChan
    76  		for {
    77  			select {
    78  			case pod := <-podChan:
    79  				for _, container := range pod.Spec.Containers {
    80  					containerLogs, err := o.kubernetesClient.GetPodLogs(pod.Name, container.Name, follow)
    81  					if err != nil {
    82  						events.Err <- fmt.Errorf("failed to get logs for container %s; error: %v", container.Name, err)
    83  					}
    84  					events.Logs <- ContainerLogs{container.Name, containerLogs}
    85  				}
    86  			case err := <-errChan:
    87  				events.Err <- err
    88  			case <-doneChan:
    89  				events.Done <- struct{}{}
    90  			}
    91  		}
    92  	}()
    93  
    94  	if mode == odolabels.ComponentDevMode || mode == odolabels.ComponentAnyMode {
    95  		selector = odolabels.GetSelector(componentName, "app", odolabels.ComponentDevMode, false)
    96  		err := o.getPodsForSelector(selector, namespace, podChan)
    97  		if err != nil {
    98  			errChan <- err
    99  		}
   100  	}
   101  	if mode == odolabels.ComponentDeployMode || mode == odolabels.ComponentAnyMode {
   102  		selector = odolabels.GetSelector(componentName, "app", odolabels.ComponentDeployMode, false)
   103  		err := o.getPodsForSelector(selector, namespace, podChan)
   104  		if err != nil {
   105  			errChan <- err
   106  		}
   107  	}
   108  
   109  	doneChan <- struct{}{}
   110  }

   112  // getPodsForSelector gets pods for the resources matching selector in the namespace; Pods found by this method will be
   113  // put on podChan so that caller function can fetch its logs
   114  func (o *LogsClient) getPodsForSelector(
   115  	selector string,
   116  	namespace string,
   117  	podChan chan corev1.Pod,
   118  ) error {
   119  	// set of unique Pods with Pod name as key; these are the Pods whose logs we want to get from the cluster
   120  	pods := map[string]struct{}{}
   121  
   122  	podList, err := o.kubernetesClient.GetPodsMatchingSelector(selector)
   123  	if err != nil {
   124  		return err
   125  	}
   126  	for _, pod := range podList.Items {
   127  		podChan <- pod
   128  		pods[pod.GetName()] = struct{}{}
   129  	}
   130  
   131  	resources, err := o.kubernetesClient.GetAllResourcesFromSelector(selector, namespace)
   132  	if err != nil {
   133  		return err
   134  	}
   135  
   136  	// get all pods in the namespace
   137  	podList, err = o.kubernetesClient.GetAllPodsInNamespace()
   138  	if err != nil {
   139  		return err
   140  	}
   141  
   142  	// match pod ownerReference (if any) with resources matching the selector
   143  	for _, pod := range podList.Items {
   144  		if _, ok := pods[pod.GetName()]; ok {
   145  			// Pod's logs have already been displayed to user
   146  			continue
   147  		}
   148  		match := false
   149  		for _, owner := range pod.GetOwnerReferences() {
   150  			match, err = o.matchOwnerReferenceWithResources(owner, resources)
   151  			if err != nil {
   152  				return err
   153  			} else if match {
   154  				pods[pod.GetName()] = struct{}{}
   155  				podChan <- pod
   156  				break // because we don't need to check other owner references of the pod anymore
   157  			}
   158  		}
   159  	}
   160  	return nil
   161  }

   163  // matchOwnerReferenceWithResources recursively checks if the owner reference passed to it matches any of the resources
   164  // This is useful when trying to find if a pod is owned by any of the ReplicaSet or Deployment in the cluster.
   165  func (o *LogsClient) matchOwnerReferenceWithResources(owner metav1.OwnerReference, resources []unstructured.Unstructured) (bool, error) {
   166  	// first, check if ownerReference belongs to any of the resources
   167  	for _, resource := range resources {
   168  		if resource.GetUID() != "" && owner.UID != "" && resource.GetUID() == owner.UID {
   169  			return true, nil
   170  		}
   171  	}
   172  	// second, get the resource indicated by ownerReference and check its ownerReferences field
   173  	restMapping, err := o.kubernetesClient.GetRestMappingFromGVK(schema.FromAPIVersionAndKind(owner.APIVersion, owner.Kind))
   174  	if err != nil {
   175  		return false, err
   176  	}
   177  	resource, err := o.kubernetesClient.GetDynamicResource(restMapping.Resource, owner.Name)
   178  	if err != nil {
   179  		return false, err
   180  	}
   181  	ownerReferences := resource.GetOwnerReferences()
   182  	// recursively check if ownerReference matches any of the resources' UID
   183  	for _, ownerReference := range ownerReferences {
   184  		return o.matchOwnerReferenceWithResources(ownerReference, resources)
   185  	}
   186  	return false, nil
   187  }
