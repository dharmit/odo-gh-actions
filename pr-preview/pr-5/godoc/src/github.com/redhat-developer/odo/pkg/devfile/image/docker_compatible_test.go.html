   149  	fakeFs := filesystem.NewFakeFs()
   150  
   151  	for _, tt := range []struct {
   152  		name       string
   153  		uriFunc    func() (*httptest.Server, string)
   154  		wantErr    bool
   155  		wantIsTemp bool
   156  		want       string
   157  	}{
   158  		{
   159  			name:    "local file",
   160  			uriFunc: func() (*httptest.Server, string) { return nil, "Dockerfile" },
   161  			want:    "Dockerfile",
   162  		},
   163  		{
   164  			name:    "remote file (non-HTTP)",
   165  			uriFunc: func() (*httptest.Server, string) { return nil, "ftp://example.com/Dockerfile" },
   166  			want:    "ftp://example.com/Dockerfile",
   167  		},
   168  		{
   169  			name: "remote file with error (HTTP)",
   170  			uriFunc: func() (*httptest.Server, string) {
   171  				s := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
   172  					w.WriteHeader(http.StatusNotFound)
   173  				}))
   174  				return s, s.URL + "/404"
   175  			},
   176  			wantErr:    true,
   177  			wantIsTemp: true,
   178  		},
   179  		{
   180  			name: "remote file (HTTP)",
   181  			uriFunc: func() (*httptest.Server, string) {
   182  				s := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
   183  					fmt.Fprintln(w, "FROM alpine:3.6")
   184  					fmt.Fprintln(w, "RUN echo Hello World")
   185  					fmt.Fprintln(w, "ENTRYPOINT [\"/bin/tail\"]")
   186  					fmt.Fprintln(w, "CMD [\"-f\", \"/dev/null\"]")
   187  				}))
   188  				return s, s.URL
   189  			},
   190  			wantIsTemp: true,
   191  		},
   192  	} {
   193  		t.Run(tt.name, func(t *testing.T) {
   194  			server, uri := tt.uriFunc()
   195  			if server != nil {
   196  				defer server.Close()
   197  			}
   198  			got, gotIsTemp, err := resolveAndDownloadDockerfile(fakeFs, uri)
   199  			if (err != nil) != tt.wantErr {
   200  				t.Errorf("%s:\n  Expected error %v,\n       got %v", tt.name, tt.wantErr, err)
   201  			}
   202  			if gotIsTemp != tt.wantIsTemp {
   203  				t.Errorf("%s:\n  For 'isTemp', expected %v,\n       got %v", tt.name, tt.wantIsTemp, gotIsTemp)
   204  			}
   205  			if gotIsTemp {
   206  				defer func(fs filesystem.Filesystem, name string) {
   207  					_ = fs.Remove(name)
   208  				}(fakeFs, got)
   209  				// temp file is created, so we can't compare the path, but we can check the path is not blank
   210  				if strings.TrimSpace(got) == "" {
   211  					t.Errorf("%s:\n  Expected non-blank path,\n       got blank path: %s", tt.name, got)
   212  				}
   213  			} else if got != tt.want {
   214  				t.Errorf("%s:\n  Expected %v,\n       got %v", tt.name, tt.want, got)
   215  			}
   216  		})
   217  	}
