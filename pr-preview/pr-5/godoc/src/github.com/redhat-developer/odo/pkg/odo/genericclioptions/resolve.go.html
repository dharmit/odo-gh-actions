    14  // resolveProjectAndNamespace resolve project in Context and namespace in Kubernetes and OpenShift clients
    15  func (o *internalCxt) resolveProjectAndNamespace(cmdline cmdline.Cmdline, configProvider localConfigProvider.LocalConfigProvider) error {
    16  	var namespace string
    17  	projectFlag := cmdline.FlagValueIfSet(util.ProjectFlagName)
    18  	if len(projectFlag) > 0 {
    19  		// if namespace flag was set, check that the specified namespace exists and use it
    20  		_, err := o.KClient.GetNamespaceNormal(projectFlag)
    21  
    22  		// do not error out when the user is running `odo project`
    23  		if err != nil {
    24  			if cmdline.GetParentName() != "project" {
    25  				return err
    26  			}
    27  		}
    28  		namespace = projectFlag
    29  	} else {
    30  		namespace = configProvider.GetNamespace()
    31  		if namespace == "" {
    32  			namespace = o.KClient.GetCurrentNamespace()
    33  			if len(namespace) <= 0 {
    34  				errFormat := "Could not get current namespace. Please create or set a namespace\n"
    35  				err := checkProjectCreateOrDeleteOnlyOnInvalidNamespace(cmdline, errFormat)
    36  				if err != nil {
    37  					return err
    38  				}
    39  			}
    40  		}
    41  
    42  		// check that the specified namespace exists
    43  		_, err := o.KClient.GetNamespaceNormal(namespace)
    44  		if err != nil {
    45  			var errFormat string
    46  			if kerrors.IsForbidden(err) {
    47  				errFormat = "You are currently not logged in into the cluster. Use `odo login` first to perform any operation on cluster"
    48  			} else {
    49  				errFormat = fmt.Sprintf("You don't have permission to create or set namespace %q or the namespace doesn't exist. Please create or set a different namespace\n\t", namespace)
    50  			}
    51  
    52  			// errFormat := fmt.Sprint(e1, "%s project create|set <project_name>")
    53  			err = checkProjectCreateOrDeleteOnlyOnInvalidNamespaceNoFmt(cmdline, errFormat)
    54  			if err != nil {
    55  				return err
    56  			}
    57  		}
    58  	}
    59  	o.KClient.SetNamespace(namespace)
    60  	o.project = namespace
    61  	return nil
    62  }
