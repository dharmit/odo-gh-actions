    68  // Complete completes SetOptions after they've been created
    69  func (so *SetOptions) Complete(cmdline cmdline.Cmdline, args []string) (err error) {
    70  	so.namespaceName = args[0]
    71  	so.Context, err = genericclioptions.New(genericclioptions.NewCreateParameters(cmdline))
    72  	if err != nil {
    73  		return err
    74  	}
    75  	so.contextDir, err = so.clientset.FS.Getwd()
    76  	if err != nil {
    77  		return err
    78  	}
    79  	if scontext.GetTelemetryStatus(cmdline.Context()) {
    80  		scontext.SetClusterType(cmdline.Context(), so.KClient)
    81  	}
    82  	return nil
    83  }

    90  // Run runs the 'set namespace' command
    91  func (so *SetOptions) Run(ctx context.Context) error {
    92  	devfilePresent, err := location.DirectoryContainsDevfile(so.clientset.FS, so.contextDir)
    93  	if err != nil {
    94  		// Purposely ignoring the error, as it is not mandatory for this command
    95  		klog.V(2).Infof("Unexpected error while checking if running inside a component directory: %v", err)
    96  	}
    97  	if devfilePresent {
    98  		log.Warningf("This is being executed inside a component directory. This will not update the %s of the existing component",
    99  			so.commandName)
   100  	}
   101  
   102  	err = so.clientset.ProjectClient.SetCurrent(so.namespaceName)
   103  	if err != nil {
   104  		return err
   105  	}
   106  
   107  	log.Successf("Current active %[1]s set to %q", so.commandName, so.namespaceName)
   108  
   109  	return nil
   110  }
