    28  func getAllResources(client dynamic.Interface, apis []apiResource, ns string, selector string) ([]unstructured.Unstructured, error) {
    29  	var out []unstructured.Unstructured
    30  	outChan := make(chan []unstructured.Unstructured)
    31  
    32  	var apisOfInterest []apiResource
    33  	for _, api := range apis {
    34  		if !api.r.Namespaced {
    35  			klog.V(4).Infof("[query api] api (%s) is non-namespaced, skipping", api.r.Name)
    36  			continue
    37  		}
    38  		apisOfInterest = append(apisOfInterest, api)
    39  	}
    40  
    41  	start := time.Now()
    42  	group := new(errgroup.Group) // an error group errors when any of the go routines encounters an error
    43  	klog.V(2).Infof("starting to concurrently query %d APIs", len(apis))
    44  
    45  	for _, api := range apisOfInterest {
    46  		api := api // shadowing because go vet complains "loop variable api captured by func literal"
    47  		group.Go(func() error {
    48  			klog.V(4).Infof("[query api] start: %s", api.GroupVersionResource())
    49  			v, err := queryAPI(client, api, ns, selector)
    50  			if err != nil {
    51  				klog.V(4).Infof("[query api] error querying: %s, error=%v", api.GroupVersionResource(), err)
    52  				return err
    53  			}
    54  			outChan <- v
    55  			klog.V(4).Infof("[query api]  done: %s, found %d apis", api.GroupVersionResource(), len(v))
    56  			return nil
    57  		})
    58  	}
    59  	klog.V(2).Infof("fired up all goroutines to query APIs")
    60  
    61  	errChan := make(chan error)
    62  	go func() {
    63  		err := group.Wait()
    64  		klog.V(2).Infof("all goroutines have returned in %v", time.Since(start))
    65  		close(outChan)
    66  		errChan <- err
    67  	}()
    68  
    69  	for v := range outChan {
    70  		out = append(out, v...)
    71  	}
    72  
    73  	klog.V(2).Infof("query result: objects=%d", len(out))
    74  	return out, <-errChan
    75  }

    77  func queryAPI(client dynamic.Interface, api apiResource, ns string, selector string) ([]unstructured.Unstructured, error) {
    78  	var out []unstructured.Unstructured
    79  
    80  	var next string
    81  	for {
    82  		var intf dynamic.ResourceInterface
    83  		nintf := client.Resource(api.GroupVersionResource())
    84  		intf = nintf.Namespace(ns)
    85  		resp, err := intf.List(context.TODO(), metav1.ListOptions{
    86  			Limit:         250,
    87  			Continue:      next,
    88  			LabelSelector: selector,
    89  		})
    90  		if err != nil {
    91  			klog.V(2).Infof("listing resources failed (%s): %v", api.GroupVersionResource(), err)
    92  			return nil, nil
    93  		}
    94  		out = append(out, resp.Items...)
    95  
    96  		next = resp.GetContinue()
    97  		if next == "" {
    98  			break
    99  		}
   100  	}
   101  	return out, nil
   102  }

   124  func findAPIs(client discovery.DiscoveryInterface) (*resourceMap, error) {
   125  	start := time.Now()
   126  	resList, err := client.ServerPreferredResources()
   127  	if err != nil {
   128  		return nil, fmt.Errorf("failed to fetch api groups from kubernetes: %w", err)
   129  	}
   130  	klog.V(2).Infof("queried api discovery in %v", time.Since(start))
   131  	klog.V(3).Infof("found %d items (groups) in server-preferred APIResourceList", len(resList))
   132  
   133  	rm := &resourceMap{
   134  		m: make(resourceNameLookup),
   135  	}
   136  	for _, group := range resList {
   137  		klog.V(5).Infof("iterating over group %s/%s (%d apis)", group.GroupVersion, group.APIVersion, len(group.APIResources))
   138  		gv, err := schema.ParseGroupVersion(group.GroupVersion)
   139  		if err != nil {
   140  			return nil, fmt.Errorf("%q cannot be parsed into groupversion: %w", group.GroupVersion, err)
   141  		}
   142  
   143  		for _, apiRes := range group.APIResources {
   144  			klog.V(5).Infof("  api=%s namespaced=%v", apiRes.Name, apiRes.Namespaced)
   145  			if !contains(apiRes.Verbs, "list") {
   146  				klog.V(4).Infof("    api (%s) doesn't have required verb, skipping: %v", apiRes.Name, apiRes.Verbs)
   147  				continue
   148  			}
   149  			v := apiResource{
   150  				gv: gv,
   151  				r:  apiRes,
   152  			}
   153  			names := apiNames(apiRes, gv)
   154  			klog.V(6).Infof("names: %s", strings.Join(names, ", "))
   155  			for _, name := range names {
   156  				rm.m[name] = append(rm.m[name], v)
   157  			}
   158  			rm.list = append(rm.list, v)
   159  		}
   160  	}
   161  	klog.V(5).Infof("  found %d apis", len(rm.m))
   162  	return rm, nil
   163  }

   175  func apiNames(a metav1.APIResource, gv schema.GroupVersion) []string {
   176  	var out []string
   177  	singularName := a.SingularName
   178  	if singularName == "" {
   179  		// TODO(ahmetb): sometimes SingularName is empty (e.g. Deployment), use lowercase Kind as fallback - investigate why
   180  		singularName = strings.ToLower(a.Kind)
   181  	}
   182  	pluralName := a.Name
   183  	shortNames := a.ShortNames
   184  	names := append([]string{singularName, pluralName}, shortNames...)
   185  	for _, n := range names {
   186  		fmtBare := n                                                                // e.g. deployment
   187  		fmtWithGroup := strings.Join([]string{n, gv.Group}, ".")                    // e.g. deployment.apps
   188  		fmtWithGroupVersion := strings.Join([]string{n, gv.Version, gv.Group}, ".") // e.g. deployment.v1.apps
   189  
   190  		out = append(out,
   191  			fmtBare, fmtWithGroup, fmtWithGroupVersion)
   192  	}
   193  	return out
   194  }
