    90  // CheckIfConfigurationNeeded checks against a set of commands that need configuration.
    91  func (o *Cobra) CheckIfConfigurationNeeded() (bool, error) {
    92  	// Here we will check for parent commands, if the match a certain criteria, we will skip
    93  	// using the configuration.
    94  	//
    95  	// For example, `odo init` should NOT check to see if there is actually a configuration yet.
    96  	if o.cmd.HasParent() {
    97  
    98  		// Find the first child of the command, as some groups are allowed even with non existent configuration
    99  		firstChildCommand := getFirstChildOfCommand(o.cmd)
   100  
   101  		// This should *never* happen, but added just to be safe
   102  		if firstChildCommand == nil {
   103  			return false, errors.New("unable to get first child of command")
   104  		}
   105  
   106  		// Gather necessary preliminary information
   107  		componentNameFlagValue := o.FlagValueIfSet(util.ComponentNameFlagName)
   108  		// if command is `odo delete component` and name flag is not used, require configuration
   109  		if firstChildCommand.Name() == "delete" && o.cmd.Name() == "component" && len(componentNameFlagValue) == 0 {
   110  			return true, nil
   111  		}
   112  		// if command is `odo build-images`, require configuration
   113  		if o.cmd.Name() == "build-images" {
   114  			return true, nil
   115  		}
   116  	}
   117  	return false, nil
   118  }

   120  // getFirstChildOfCommand gets the first child command of the root command of command
   121  func getFirstChildOfCommand(command *cobra.Command) *cobra.Command {
   122  	// If command does not have a parent no point checking
   123  	if command.HasParent() {
   124  		// Get the root command and set current command and its parent
   125  		rootCommand := command.Root()
   126  		parentCommand := command.Parent()
   127  		mainCommand := command
   128  		for {
   129  			// if parent is root, then we have our first child in c
   130  			if parentCommand == rootCommand {
   131  				return mainCommand
   132  			}
   133  			// Traverse backwards making current command as the parent and parent as the grandparent
   134  			mainCommand = parentCommand
   135  			parentCommand = mainCommand.Parent()
   136  		}
   137  	}
   138  	return nil
   139  }
