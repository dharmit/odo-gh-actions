    32  // Create creates a pvc from the given Storage
    33  func (k kubernetesClient) Create(storage Storage) error {
    34  
    35  	if k.componentName == "" || k.appName == "" {
    36  		return fmt.Errorf("the component name and the app name should be provided")
    37  	}
    38  
    39  	pvcName, err := generatePVCName(storage.Name, k.componentName, k.appName)
    40  	if err != nil {
    41  		return err
    42  	}
    43  
    44  	labels := odolabels.GetLabels(k.componentName, k.appName, odolabels.ComponentDevMode, false)
    45  	odolabels.AddStorageInfo(labels, storage.Name, strings.Contains(storage.Name, OdoSourceVolume))
    46  
    47  	objectMeta := generator.GetObjectMeta(pvcName, k.client.GetCurrentNamespace(), labels, nil)
    48  
    49  	quantity, err := resource.ParseQuantity(storage.Spec.Size)
    50  	if err != nil {
    51  		return fmt.Errorf("unable to parse size: %v: %w", storage.Spec.Size, err)
    52  	}
    53  
    54  	pvcParams := generator.PVCParams{
    55  		ObjectMeta: objectMeta,
    56  		Quantity:   quantity,
    57  	}
    58  	pvc := generator.GetPVC(pvcParams)
    59  
    60  	// Create PVC
    61  	klog.V(2).Infof("Creating a PVC with name %v and labels %v", pvcName, labels)
    62  	_, err = k.client.CreatePVC(*pvc)
    63  	if err != nil {
    64  		return fmt.Errorf("unable to create PVC: %w", err)
    65  	}
    66  	return nil
    67  }

    85  // List lists pvc based Storage from the cluster
    86  func (k kubernetesClient) List() (StorageList, error) {
    87  	if k.deployment == nil {
    88  		var err error
    89  		selector := odolabels.GetSelector(k.componentName, k.appName, odolabels.ComponentAnyMode, true)
    90  		k.deployment, err = k.client.GetOneDeploymentFromSelector(selector)
    91  		if err != nil {
    92  			if _, ok := err.(*kclient.DeploymentNotFoundError); ok {
    93  				return StorageList{}, nil
    94  			}
    95  			return StorageList{}, err
    96  		}
    97  	}
    98  
    99  	initContainerVolumeMounts := make(map[string]bool)
   100  	for _, container := range k.deployment.Spec.Template.Spec.InitContainers {
   101  		for _, volumeMount := range container.VolumeMounts {
   102  			initContainerVolumeMounts[volumeMount.Name] = true
   103  		}
   104  	}
   105  
   106  	containerVolumeMounts := make(map[string]bool)
   107  	for _, container := range k.deployment.Spec.Template.Spec.Containers {
   108  		for _, volumeMount := range container.VolumeMounts {
   109  			containerVolumeMounts[volumeMount.Name] = true
   110  		}
   111  	}
   112  
   113  	var storage []Storage
   114  	var volumeMounts []Storage
   115  	for _, container := range k.deployment.Spec.Template.Spec.Containers {
   116  		for _, volumeMount := range container.VolumeMounts {
   117  
   118  			// avoid the volume mounts only from the init containers
   119  			// and the source volume mount
   120  			_, initOK := initContainerVolumeMounts[volumeMount.Name]
   121  			_, ok := containerVolumeMounts[volumeMount.Name]
   122  			if (!ok && initOK) || volumeMount.Name == OdoSourceVolume || volumeMount.Name == SharedDataVolumeName {
   123  				continue
   124  			}
   125  
   126  			volumeMounts = append(volumeMounts, Storage{
   127  				ObjectMeta: metav1.ObjectMeta{Name: volumeMount.Name},
   128  				Spec: StorageSpec{
   129  					Path:          volumeMount.MountPath,
   130  					ContainerName: container.Name,
   131  				},
   132  			})
   133  
   134  		}
   135  	}
   136  
   137  	if len(volumeMounts) <= 0 {
   138  		return StorageList{}, nil
   139  	}
   140  
   141  	selector := labels.SelectorBuilder().WithComponent(k.componentName).WithoutSourcePVC(OdoSourceVolume).Selector()
   142  	pvcs, err := k.client.ListPVCs(selector)
   143  	if err != nil {
   144  		return StorageList{}, fmt.Errorf("unable to get PVC using selector %q: %w", selector, err)
   145  	}
   146  
   147  	// to track volume mounts used by a PVC
   148  	validVolumeMounts := make(map[string]bool)
   149  
   150  	for _, pvc := range pvcs {
   151  		found := false
   152  		for _, volumeMount := range volumeMounts {
   153  			if volumeMount.Name == pvc.Name+"-vol" {
   154  				// this volume mount is used by a PVC
   155  				validVolumeMounts[volumeMount.Name] = true
   156  
   157  				found = true
   158  				size := pvc.Spec.Resources.Requests[corev1.ResourceStorage]
   159  				storage = append(storage, NewStorageWithContainer(odolabels.GetDevfileStorageName(pvc.Labels), size.String(), volumeMount.Spec.Path, volumeMount.Spec.ContainerName, nil))
   160  			}
   161  		}
   162  		if !found {
   163  			return StorageList{}, fmt.Errorf("mount path for pvc %s not found", pvc.Name)
   164  		}
   165  	}
   166  
   167  	// to track non PVC volumes
   168  	for _, volume := range k.deployment.Spec.Template.Spec.Volumes {
   169  		if volume.PersistentVolumeClaim == nil {
   170  			validVolumeMounts[volume.Name] = true
   171  		}
   172  	}
   173  
   174  	for _, volumeMount := range volumeMounts {
   175  		if _, ok := validVolumeMounts[volumeMount.Name]; !ok {
   176  			return StorageList{}, fmt.Errorf("pvc not found for mount path %s", volumeMount.Name)
   177  		}
   178  	}
   179  
   180  	return StorageList{Items: storage}, nil
   181  }
