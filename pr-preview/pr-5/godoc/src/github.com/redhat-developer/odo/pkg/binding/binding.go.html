    53  // GetFlags gets the flag specific to add binding operation so that it can correctly decide on the backend to be used
    54  // It ignores all the flags except the ones specific to add binding operation, for e.g. verbosity flag
    55  func (o *BindingClient) GetFlags(flags map[string]string) map[string]string {
    56  	bindingFlags := map[string]string{}
    57  	for flag, value := range flags {
    58  		if flag == backendpkg.FLAG_NAME ||
    59  			flag == backendpkg.FLAG_WORKLOAD ||
    60  			flag == backendpkg.FLAG_SERVICE_NAMESPACE ||
    61  			flag == backendpkg.FLAG_SERVICE ||
    62  			flag == backendpkg.FLAG_BIND_AS_FILES ||
    63  			flag == backendpkg.FLAG_NAMING_STRATEGY {
    64  			bindingFlags[flag] = value
    65  		}
    66  	}
    67  	return bindingFlags
    68  }

    70  func (o *BindingClient) GetServiceInstances(namespace string) (map[string]unstructured.Unstructured, error) {
    71  	err := o.checkServiceBindingOperatorInstalled()
    72  	if err != nil {
    73  		return nil, err
    74  	}
    75  
    76  	// Get the BindableKinds/bindable-kinds object
    77  	bindableKind, err := o.kubernetesClient.GetBindableKinds()
    78  	if err != nil {
    79  		return nil, err
    80  	}
    81  
    82  	// get a list of restMappings of all the GVKs present in bindableKind's Status
    83  	bindableKindRestMappings, err := o.kubernetesClient.GetBindableKindStatusRestMapping(bindableKind.Status)
    84  	if err != nil {
    85  		return nil, err
    86  	}
    87  
    88  	var bindableObjectMap = map[string]unstructured.Unstructured{}
    89  	for _, restMapping := range bindableKindRestMappings {
    90  		// TODO: Debug into why List returns all the versions instead of the GVR version
    91  		// List all the instances of the restMapping object
    92  		resources, err := o.kubernetesClient.ListDynamicResources(namespace, restMapping.Resource)
    93  		if err != nil {
    94  			return nil, err
    95  		}
    96  
    97  		for _, item := range resources.Items {
    98  			// format: `<name> (<kind>.<group>)`
    99  			serviceName := fmt.Sprintf("%s (%s.%s)", item.GetName(), item.GetKind(), item.GroupVersionKind().Group)
   100  			bindableObjectMap[serviceName] = item
   101  		}
   102  
   103  	}
   104  
   105  	return bindableObjectMap, nil
   106  }

   109  // from group binding.operators.coreos.com/v1alpha1 or servicebinding.io/v1alpha3
   110  func (o *BindingClient) GetBindingsFromDevfile(devfileObj parser.DevfileObj, context string) ([]api.ServiceBinding, error) {
   111  	result := []api.ServiceBinding{}
   112  	kubeComponents, err := devfileObj.Data.GetComponents(parsercommon.DevfileOptions{
   113  		ComponentOptions: parsercommon.ComponentOptions{
   114  			ComponentType: devfilev1alpha2.KubernetesComponentType,
   115  		},
   116  	})
   117  	if err != nil {
   118  		return nil, err
   119  	}
   120  
   121  	for _, component := range kubeComponents {
   122  		strCRD, err := libdevfile.GetK8sManifestWithVariablesSubstituted(devfileObj, component.Name, context, devfilefs.DefaultFs{})
   123  		if err != nil {
   124  			return nil, err
   125  		}
   126  
   127  		u := unstructured.Unstructured{}
   128  		if err := yaml.Unmarshal([]byte(strCRD), &u.Object); err != nil {
   129  			return nil, err
   130  		}
   131  
   132  		switch u.GetObjectKind().GroupVersionKind() {
   133  		case bindingApi.GroupVersionKind:
   134  
   135  			var sbo bindingApi.ServiceBinding
   136  			err := kclient.ConvertUnstructuredToResource(u, &sbo)
   137  			if err != nil {
   138  				return nil, err
   139  			}
   140  
   141  			sb, err := o.kubernetesClient.APIServiceBindingFromBinding(sbo)
   142  			if err != nil {
   143  				return nil, err
   144  			}
   145  			sb.Status, err = o.getStatusFromBinding(sb.Name)
   146  			if err != nil {
   147  				return nil, err
   148  			}
   149  
   150  			result = append(result, sb)
   151  
   152  		case specApi.GroupVersion.WithKind("ServiceBinding"):
   153  
   154  			var sbc specApi.ServiceBinding
   155  			err := kclient.ConvertUnstructuredToResource(u, &sbc)
   156  			if err != nil {
   157  				return nil, err
   158  			}
   159  
   160  			sb := o.kubernetesClient.APIServiceBindingFromSpec(sbc)
   161  			sb.Status, err = o.getStatusFromSpec(sb.Name)
   162  			if err != nil {
   163  				return nil, err
   164  			}
   165  
   166  			result = append(result, sb)
   167  
   168  		}
   169  	}
   170  	return result, nil
   171  }

   173  // GetBindingFromCluster returns the ServiceBinding resource with the given name
   174  // from the cluster, from group binding.operators.coreos.com/v1alpha1 or servicebinding.io/v1alpha3
   175  func (o *BindingClient) GetBindingFromCluster(name string) (api.ServiceBinding, error) {
   176  
   177  	bindingSB, err := o.kubernetesClient.GetBindingServiceBinding(name)
   178  	if err == nil {
   179  		var sb api.ServiceBinding
   180  		sb, err = o.kubernetesClient.APIServiceBindingFromBinding(bindingSB)
   181  		if err != nil {
   182  			return api.ServiceBinding{}, err
   183  		}
   184  		sb.Status, err = o.getStatusFromBinding(bindingSB.Name)
   185  		if err != nil {
   186  			return api.ServiceBinding{}, err
   187  		}
   188  		return sb, nil
   189  	}
   190  	if err != nil && !kerrors.IsNotFound(err) {
   191  		return api.ServiceBinding{}, err
   192  	}
   193  
   194  	specSB, err := o.kubernetesClient.GetSpecServiceBinding(name)
   195  	if err == nil {
   196  		sb := o.kubernetesClient.APIServiceBindingFromSpec(specSB)
   197  		sb.Status, err = o.getStatusFromSpec(specSB.Name)
   198  		if err != nil {
   199  			return api.ServiceBinding{}, err
   200  		}
   201  		return sb, nil
   202  	}
   203  
   204  	// In case of notFound error, this time we return the error
   205  	if kerrors.IsNotFound(err) {
   206  		return api.ServiceBinding{}, fmt.Errorf("ServiceBinding %q not found", name)
   207  	}
   208  	return api.ServiceBinding{}, err
   209  }

   213  func (o *BindingClient) getStatusFromBinding(name string) (*api.ServiceBindingStatus, error) {
   214  	bindingSB, err := o.kubernetesClient.GetBindingServiceBinding(name)
   215  	if err != nil {
   216  		if kerrors.IsNotFound(err) {
   217  			return nil, nil
   218  		}
   219  		return nil, err
   220  	}
   221  
   222  	if injected := meta.IsStatusConditionTrue(bindingSB.Status.Conditions, bindingApis.InjectionReady); !injected {
   223  		return nil, nil
   224  	}
   225  
   226  	secretName := bindingSB.Status.Secret
   227  	secret, err := o.kubernetesClient.GetSecret(secretName, o.kubernetesClient.GetCurrentNamespace())
   228  	if err != nil {
   229  		return nil, err
   230  	}
   231  
   232  	bindings := make([]string, 0, len(secret.Data))
   233  	if bindingSB.Spec.BindAsFiles {
   234  		for k := range secret.Data {
   235  			bindingName := filepath.ToSlash(filepath.Join("${SERVICE_BINDING_ROOT}", name, k))
   236  			bindings = append(bindings, bindingName)
   237  		}
   238  		return &api.ServiceBindingStatus{
   239  			BindingFiles: bindings,
   240  		}, nil
   241  	}
   242  
   243  	for k := range secret.Data {
   244  		bindings = append(bindings, k)
   245  	}
   246  	return &api.ServiceBindingStatus{
   247  		BindingEnvVars: bindings,
   248  	}, nil
   249  }

   253  func (o *BindingClient) getStatusFromSpec(name string) (*api.ServiceBindingStatus, error) {
   254  	specSB, err := o.kubernetesClient.GetSpecServiceBinding(name)
   255  	if err != nil {
   256  		if kerrors.IsNotFound(err) {
   257  			return nil, nil
   258  		}
   259  		return nil, err
   260  	}
   261  
   262  	if injected := meta.IsStatusConditionTrue(specSB.Status.Conditions, bindingApis.InjectionReady); !injected {
   263  		return nil, nil
   264  	}
   265  
   266  	if specSB.Status.Binding == nil {
   267  		return nil, nil
   268  	}
   269  	secretName := specSB.Status.Binding.Name
   270  	secret, err := o.kubernetesClient.GetSecret(secretName, o.kubernetesClient.GetCurrentNamespace())
   271  	if err != nil {
   272  		return nil, err
   273  	}
   274  	bindingFiles := make([]string, 0, len(secret.Data))
   275  	bindingEnvVars := make([]string, 0, len(specSB.Spec.Env))
   276  	for k := range secret.Data {
   277  		bindingName := filepath.ToSlash(filepath.Join("${SERVICE_BINDING_ROOT}", name, k))
   278  		bindingFiles = append(bindingFiles, bindingName)
   279  	}
   280  	for _, env := range specSB.Spec.Env {
   281  		bindingEnvVars = append(bindingEnvVars, env.Name)
   282  	}
   283  	return &api.ServiceBindingStatus{
   284  		BindingFiles:   bindingFiles,
   285  		BindingEnvVars: bindingEnvVars,
   286  	}, nil
   287  }

   303  func (o *BindingClient) CheckServiceBindingsInjectionDone(componentName string, appName string) (bool, error) {
   304  
   305  	deployment, err := o.kubernetesClient.GetOneDeployment(componentName, appName, true)
   306  	if err != nil {
   307  		// If not deployment yet => all bindings are done
   308  		if _, ok := err.(*kclient.DeploymentNotFoundError); ok {
   309  			return true, nil
   310  		}
   311  		return false, err
   312  	}
   313  	deploymentName := deployment.GetName()
   314  
   315  	specList, bindingList, err := o.kubernetesClient.ListServiceBindingsFromAllGroups()
   316  	if err != nil {
   317  		// If ServiceBinding kind is not registered => all bindings are done
   318  		if runtime.IsNotRegisteredError(err) {
   319  			return true, nil
   320  		}
   321  		return false, err
   322  	}
   323  
   324  	for _, binding := range bindingList {
   325  		app := binding.Spec.Application
   326  		if app.Group != appsv1.SchemeGroupVersion.Group ||
   327  			app.Version != appsv1.SchemeGroupVersion.Version ||
   328  			(app.Kind != "Deployment" && app.Resource != "deployments") {
   329  			continue
   330  		}
   331  		if app.Name != deploymentName {
   332  			continue
   333  		}
   334  		if injected := meta.IsStatusConditionTrue(binding.Status.Conditions, bindingApis.InjectionReady); !injected {
   335  			return false, nil
   336  		}
   337  	}
   338  
   339  	for _, binding := range specList {
   340  		app := binding.Spec.Workload
   341  		if app.APIVersion != appsv1.SchemeGroupVersion.String() ||
   342  			app.Kind != "Deployment" {
   343  			continue
   344  		}
   345  		if app.Name != deploymentName {
   346  			continue
   347  		}
   348  		if injected := meta.IsStatusConditionTrue(binding.Status.Conditions, bindingApis.InjectionReady); !injected {
   349  			return false, nil
   350  		}
   351  	}
   352  
   353  	return true, nil
   354  }
