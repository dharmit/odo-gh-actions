    93  	Describe("displaying welcoming messages", func() {
    94  
    95  		// testFunc is a function that returns a `Tester` function (intended to be used via `helper.RunInteractive`),
    96  		// which first expects all messages in `welcomingMsgs` to be read from the console,
    97  		// then runs an `additionalTester` and finally expects the asking of a component name
    98  		// (based on the `language` specified)
    99  		testFunc := func(language string, welcomingMsgs []string, additionalTester helper.Tester) helper.Tester {
   100  			return func(ctx helper.InteractiveContext) {
   101  				for _, msg := range welcomingMsgs {
   102  					helper.ExpectString(ctx, msg)
   103  				}
   104  
   105  				if additionalTester != nil {
   106  					additionalTester(ctx)
   107  				}
   108  
   109  				helper.ExpectString(ctx, "Enter component name")
   110  				helper.SendLine(ctx, fmt.Sprintf("my-%s-app", language))
   111  
   112  				helper.ExpectString(ctx,
   113  					fmt.Sprintf("Your new component 'my-%s-app' is ready in the current directory", language))
   114  			}
   115  		}
   116  
   117  		assertBehavior := func(language string, output string, err error, msgs []string, additionalAsserter func()) {
   118  			Expect(err).To(BeNil())
   119  
   120  			lines, err := helper.ExtractLines(output)
   121  			if err != nil {
   122  				log.Fatal(err)
   123  			}
   124  			Expect(len(lines)).To(BeNumerically(">", len(msgs)))
   125  			Expect(lines[0:len(msgs)]).To(Equal(msgs))
   126  			Expect(lines).To(
   127  				ContainElement(fmt.Sprintf("Your new component 'my-%s-app' is ready in the current directory", language)))
   128  
   129  			Expect(helper.ListFilesInDir(commonVar.Context)).To(ContainElements("devfile.yaml"))
   130  
   131  			if additionalAsserter != nil {
   132  				additionalAsserter()
   133  			}
   134  		}
   135  
   136  		testRunner := func(language string, welcomingMsgs []string, tester helper.Tester) (string, error) {
   137  			command := []string{"odo", "init"}
   138  			return helper.RunInteractive(command,
   139  				// Setting verbosity level to 0, because we would be asserting the welcoming message is the first
   140  				// message displayed to the end user. So we do not want any potential debug lines to be printed first.
   141  				// Using envvars here (and not via the -v flag), because of https://github.com/redhat-developer/odo/issues/5513
   142  				[]string{"ODO_LOG_LEVEL=0"},
   143  				testFunc(language, welcomingMsgs, tester))
   144  		}
