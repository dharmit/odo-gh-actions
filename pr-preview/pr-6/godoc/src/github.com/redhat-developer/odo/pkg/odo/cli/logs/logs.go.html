   105  func (o *LogsOptions) Run(_ context.Context) error {
   106  	var logMode logsMode
   107  	var err error
   108  
   109  	if o.devMode {
   110  		logMode = DevMode
   111  	} else if o.deployMode {
   112  		logMode = DeployMode
   113  	}
   114  
   115  	var mode string
   116  	switch logMode {
   117  	case DevMode:
   118  		mode = odolabels.ComponentDevMode
   119  	case DeployMode:
   120  		mode = odolabels.ComponentDeployMode
   121  	default:
   122  		mode = odolabels.ComponentAnyMode
   123  	}
   124  
   125  	events, err := o.clientset.LogsClient.GetLogsForMode(
   126  		mode,
   127  		o.componentName,
   128  		o.Context.GetProject(),
   129  		o.follow,
   130  	)
   131  	if err != nil {
   132  		return err
   133  	}
   134  
   135  	uniqueContainerNames := map[string]struct{}{}
   136  	var goroutines struct{ count int64 } // keep a track of running goroutines so that we don't exit prematurely
   137  	errChan := make(chan error)          // errors are put on this channel
   138  	var mu sync.Mutex
   139  
   140  	for {
   141  		select {
   142  		case containerLogs := <-events.Logs:
   143  			uniqueName := getUniqueContainerName(containerLogs.Name, uniqueContainerNames)
   144  			uniqueContainerNames[uniqueName] = struct{}{}
   145  			colour := log.ColorPicker()
   146  			logs := containerLogs.Logs
   147  
   148  			if o.follow {
   149  				atomic.AddInt64(&goroutines.count, 1)
   150  				go func(out io.Writer) {
   151  					defer func() {
   152  						atomic.AddInt64(&goroutines.count, -1)
   153  					}()
   154  					err = printLogs(uniqueName, logs, out, colour, &mu)
   155  					if err != nil {
   156  						errChan <- err
   157  					}
   158  					events.Done <- struct{}{}
   159  				}(o.out)
   160  			} else {
   161  				err = printLogs(uniqueName, logs, o.out, colour, &mu)
   162  				if err != nil {
   163  					return err
   164  				}
   165  			}
   166  		case err = <-errChan:
   167  			return err
   168  		case err = <-events.Err:
   169  			return err
   170  		case <-events.Done:
   171  			if goroutines.count == 0 {
   172  				if len(uniqueContainerNames) == 0 {
   173  					// This will be the case when:
   174  					// 1. user specifies --dev flag, but the component's running in Deploy mode
   175  					// 2. user specified --deploy flag, but the component's running in Dev mode
   176  					// 3. user passes no flag, but component is running in neither Dev nor Deploy mode
   177  					fmt.Fprintf(o.out, "no containers running in the specified mode for the component %q\n", o.componentName)
   178  				}
   179  				return nil
   180  			}
   181  		}
   182  	}
   183  }

   185  func getUniqueContainerName(name string, uniqueNames map[string]struct{}) string {
   186  	if _, ok := uniqueNames[name]; ok {
   187  		// name already present in uniqueNames; find another name
   188  		// first check if last character in name is a number; if so increment it, else append name with [1]
   189  		var numStr string
   190  		var last int
   191  		var err error
   192  
   193  		split := strings.Split(name, "[")
   194  		if len(split) == 2 {
   195  			numStr = strings.Trim(split[1], "]")
   196  			last, err = strconv.Atoi(numStr)
   197  			if err != nil {
   198  				return ""
   199  			}
   200  			last++
   201  		} else {
   202  			last = 1
   203  		}
   204  		name = fmt.Sprintf("%s[%d]", split[0], last)
   205  		return getUniqueContainerName(name, uniqueNames)
   206  	}
   207  	return name
   208  }

   210  // printLogs prints the logs of the containers with container name prefixed to the log message
   211  func printLogs(containerName string, rd io.ReadCloser, out io.Writer, colour color.Attribute, mu *sync.Mutex) error {
   212  	scanner := bufio.NewScanner(rd)
   213  	scanner.Split(bufio.ScanLines)
   214  
   215  	for scanner.Scan() {
   216  		line := scanner.Text()
   217  		err := func() error {
   218  			mu.Lock()
   219  			defer mu.Unlock()
   220  			color.Set(colour)
   221  			defer color.Unset()
   222  
   223  			_, err := fmt.Fprintln(out, containerName+": "+line)
   224  			return err
   225  		}()
   226  		if err != nil {
   227  			return err
   228  		}
   229  	}
   230  
   231  	return nil
   232  }

   243  		Run: func(cmd *cobra.Command, args []string) {

   248  	logsCmd.Flags().BoolVar(&o.deployMode, string(DeployMode), false, "Show logs for containers running only in Deploy mode")

   250  
   251  	clientset.Add(logsCmd, clientset.LOGS, clientset.FILESYSTEM)
