   842  // waitTimeout waits for the waitgroup for the specified max timeout.
   843  // Returns true if waiting timed out.
   844  func waitTimeout(wg *sync.WaitGroup, timeout time.Duration) bool {
   845  	c := make(chan struct{})
   846  	go func() {
   847  		defer close(c)
   848  		wg.Wait()
   849  	}()
   850  	select {
   851  	case <-c:
   852  		return false // completed normally
   853  	case <-time.After(timeout):
   854  		return true // timed out
   855  	}
   856  }
