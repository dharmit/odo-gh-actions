    66  func (o *ComponentOptions) Complete(cmdline cmdline.Cmdline, args []string) (err error) {
    67  	// 1. Name is not passed, and odo has access to devfile.yaml; Name is not passed so we assume that odo has access to the devfile.yaml
    68  	if o.name == "" {
    69  		o.Context, err = genericclioptions.New(genericclioptions.NewCreateParameters(cmdline).NeedDevfile(""))
    70  		if err != nil {
    71  			return err
    72  		}
    73  		// this ensures that the namespace set in env.yaml is used
    74  		o.clientset.KubernetesClient.SetNamespace(o.GetProject())
    75  		return nil
    76  	}
    77  	// 2. Name is passed, and odo does not have access to devfile.yaml; if Name is passed, then we assume that odo does not have access to the devfile.yaml
    78  	if o.namespace != "" {
    79  		o.clientset.KubernetesClient.SetNamespace(o.namespace)
    80  	} else {
    81  		o.namespace = o.clientset.KubernetesClient.GetCurrentNamespace()
    82  	}
    83  	return nil
    84  }

    97  // deleteNamedComponent deletes a component given its name
    98  func (o *ComponentOptions) deleteNamedComponent() error {
    99  	log.Info("Searching resources to delete, please wait...")
   100  	list, err := o.clientset.DeleteClient.ListClusterResourcesToDelete(o.name, o.namespace)
   101  	if err != nil {
   102  		return err
   103  	}
   104  	if len(list) == 0 {
   105  		log.Infof("No resource found for component %q in namespace %q\n", o.name, o.namespace)
   106  		return nil
   107  	}
   108  	printDevfileComponents(o.name, o.namespace, list)
   109  	if o.forceFlag || ui.Proceed("Are you sure you want to delete these resources?") {
   110  		failed := o.clientset.DeleteClient.DeleteResources(list, o.waitFlag)
   111  		for _, fail := range failed {
   112  			log.Warningf("Failed to delete the %q resource: %s\n", fail.GetKind(), fail.GetName())
   113  		}
   114  		log.Infof("The component %q is successfully deleted from namespace %q", o.name, o.namespace)
   115  		return nil
   116  	}
   117  
   118  	log.Error("Aborting deletion of component")
   119  	return nil
   120  }

   122  // deleteDevfileComponent deletes all the components defined by the devfile in the current directory
   123  func (o *ComponentOptions) deleteDevfileComponent() error {
   124  	devfileObj := o.EnvSpecificInfo.GetDevfileObj()
   125  	componentName := devfileObj.GetMetadataName()
   126  	namespace := o.GetProject()
   127  	appName := "app"
   128  
   129  	log.Info("Searching resources to delete, please wait...")
   130  	isInnerLoopDeployed, devfileResources, err := o.clientset.DeleteClient.ListResourcesToDeleteFromDevfile(devfileObj, appName, labels.ComponentAnyMode)
   131  	if err != nil {
   132  		return err
   133  	}
   134  	if len(devfileResources) == 0 {
   135  		log.Infof("No resource found for component %q in namespace %q\n", componentName, namespace)
   136  		return nil
   137  	}
   138  	// Print all the resources that odo will attempt to delete
   139  	printDevfileComponents(componentName, namespace, devfileResources)
   140  
   141  	if o.forceFlag || ui.Proceed(fmt.Sprintf("Are you sure you want to delete %q and all its resources?", componentName)) {
   142  		// Get a list of component's resources present on the cluster
   143  		clusterResources, _ := o.clientset.DeleteClient.ListClusterResourcesToDelete(componentName, namespace)
   144  		// Get a list of component's resources absent from the devfile, but present on the cluster
   145  		remainingResources := listResourcesMissingFromDevfilePresentOnCluster(componentName, devfileResources, clusterResources)
   146  
   147  		// if innerloop deployment resource is present, then execute preStop events
   148  		if isInnerLoopDeployed {
   149  			err = o.clientset.DeleteClient.ExecutePreStopEvents(devfileObj, appName)
   150  			if err != nil {
   151  				log.Errorf("Failed to execute preStop events")
   152  			}
   153  		}
   154  
   155  		// delete all the resources
   156  		failed := o.clientset.DeleteClient.DeleteResources(devfileResources, o.waitFlag)
   157  		for _, fail := range failed {
   158  			log.Warningf("Failed to delete the %q resource: %s\n", fail.GetKind(), fail.GetName())
   159  		}
   160  		log.Infof("The component %q is successfully deleted from namespace %q", componentName, namespace)
   161  
   162  		if len(remainingResources) != 0 {
   163  			log.Printf("There are still resources left in the cluster that might be belonging to the deleted component.")
   164  			for _, resource := range remainingResources {
   165  				fmt.Printf("\t- %s: %s\n", resource.GetKind(), resource.GetName())
   166  			}
   167  			log.Infof("If you want to delete those, execute `odo delete component --name %s --namespace %s`", componentName, namespace)
   168  		}
   169  		return nil
   170  	}
   171  
   172  	log.Error("Aborting deletion of component")
   173  
   174  	return nil
   175  }

   177  // listResourcesMissingFromDevfilePresentOnCluster returns a list of resources belonging to a component name that are present on cluster, but missing from devfile
   178  func listResourcesMissingFromDevfilePresentOnCluster(componentName string, devfileResources, clusterResources []unstructured.Unstructured) []unstructured.Unstructured {
   179  	var remainingResources []unstructured.Unstructured
   180  	// get resources present in k8sResources(present on the cluster) but not in devfileResources(not present in the devfile)
   181  	for _, k8sresource := range clusterResources {
   182  		var present bool
   183  		for _, dresource := range devfileResources {
   184  			//  skip if the cluster and devfile resource are same OR if the cluster resource is the component's Endpoints resource
   185  			if reflect.DeepEqual(dresource, k8sresource) || (k8sresource.GetKind() == "Endpoints" && strings.Contains(k8sresource.GetName(), componentName)) {
   186  				present = true
   187  				break
   188  			}
   189  		}
   190  		if !present {
   191  			remainingResources = append(remainingResources, k8sresource)
   192  		}
   193  	}
   194  	return remainingResources
   195  }
