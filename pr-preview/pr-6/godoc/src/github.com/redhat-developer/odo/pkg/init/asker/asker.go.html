   171  func buildPersonalizedConfigurationOptions(configuration ContainerConfiguration) (options []string, tracker []OperationOnContainer) {
   172  	options = []string{
   173  		"NOTHING - configuration is correct",
   174  	}
   175  	// track the option selected by the user without relying on the UI message
   176  	tracker = []OperationOnContainer{{Ops: "Nothing"}}
   177  
   178  	// Add available ports
   179  	for _, port := range configuration.Ports {
   180  		options = append(options, fmt.Sprintf("Delete port %q", port))
   181  		tracker = append(tracker, OperationOnContainer{
   182  			Ops:  "Delete",
   183  			Kind: "Port",
   184  			Key:  port,
   185  		})
   186  	}
   187  	options = append(options, "Add new port")
   188  	tracker = append(tracker, OperationOnContainer{
   189  		Ops:  "Add",
   190  		Kind: "Port",
   191  	})
   192  
   193  	// Add available env vars
   194  	for key := range configuration.Envs {
   195  		options = append(options, fmt.Sprintf("Delete environment variable %q", key))
   196  		tracker = append(tracker, OperationOnContainer{
   197  			Ops:  "Delete",
   198  			Kind: "EnvVar",
   199  			Key:  key,
   200  		})
   201  	}
   202  	options = append(options, "Add new environment variable")
   203  	tracker = append(tracker, OperationOnContainer{
   204  		Ops:  "Add",
   205  		Kind: "EnvVar",
   206  	})
   207  
   208  	return
   209  }
