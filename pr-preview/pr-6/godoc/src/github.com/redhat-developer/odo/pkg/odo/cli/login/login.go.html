    63  func (o *LoginOptions) Complete(cmdline cmdline.Cmdline, args []string) (err error) {
    64  	if len(args) == 1 {
    65  		// if the user specifies server without --server flag. Example:
    66  		// odo login -u developer -p developer https://api.crc.testing:6443
    67  		// odo login --token=<some-token> https://api.crc.testing:6443
    68  		o.server = args[0]
    69  	}
    70  	return
    71  }

    73  // Validate validates the LoginOptions based on completed values
    74  func (o *LoginOptions) Validate() (err error) {
    75  	if o.server != "" && o.serverFlag != "" && o.server != o.serverFlag {
    76  		// if user has passed server value as parameter as well as used --server flag:
    77  		// * odo errors *if* the values are different
    78  		// * odo silently continues if the values are same
    79  		return fmt.Errorf("either use --server flag or pass server link as a paremeter, don't use both")
    80  	} else if o.serverFlag == "" {
    81  		o.serverFlag = o.server //	set o.serverFlag to same as o.server if there was no error
    82  	}
    83  
    84  	return
    85  }
