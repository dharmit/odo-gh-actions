    16  // kubeExecProcessHandler implements RemoteProcessHandler by executing Devfile commands right away in the container
    17  // (like a 'kubectl exec'-like approach). Command execution is done in the background, in a separate goroutine.
    18  // It works by storing the parent process PID in a file in the container,
    19  // then fires the exec command in the background.
    20  // The goroutine started can then be stopped by killing the process stored in the state file (_startCmdProcessPidFile)
    21  // in the container.
    22  type kubeExecProcessHandler struct{}

    50  // StartProcessForCommand runs the (potentially never finishing) Devfile command in the background.
    51  // The goroutine spawned here can be stopped either by stopping the parent process (e.g., 'odo dev'),
    52  // or by stopping the underlying remote process by calling the StopProcessForCommand method.
    53  func (k *kubeExecProcessHandler) StartProcessForCommand(
    54  	def CommandDefinition,
    55  	kclient kclient.ClientInterface,
    56  	podName string,
    57  	containerName string,
    58  	outputHandler CommandOutputHandler,
    59  ) error {
    60  	klog.V(4).Infof("StartProcessForCommand for %q", def.Id)
    61  
    62  	// deal with environment variables
    63  	cmdLine := def.CmdLine
    64  	envCommands := make([]string, 0, len(def.EnvVars))
    65  	for _, envVar := range def.EnvVars {
    66  		envCommands = append(envCommands, fmt.Sprintf("%s='%s'", envVar.Key, envVar.Value))
    67  	}
    68  	var setEnvCmd string
    69  	if len(envCommands) != 0 {
    70  		setEnvCmd = fmt.Sprintf("export %s &&", strings.Join(envCommands, " "))
    71  	}
    72  
    73  	var cdCmd string
    74  	if def.WorkingDir != "" {
    75  		// Change to the workdir and execute the command
    76  		cdCmd = fmt.Sprintf("cd %s &&", def.WorkingDir)
    77  	}
    78  
    79  	// since we are using /bin/sh -c, the command needs to be within a single double quote instance,
    80  	// for example "cd /tmp && pwd"
    81  	// Full command is: /bin/sh -c "[cd $workingDir && ] echo $$ > $pidFile && (envVar1='value1' envVar2='value2' $cmdLine) 1>>/proc/1/fd/1 2>>/proc/1/fd/2"
    82  	//
    83  	// "echo $$ > $pidFile" allows to store the /bin/sh parent process PID. It will allow to determine its children later on and kill them when a stop is requested.
    84  	// ($cmdLine) runs the command passed in a subshell (to handle cases where the command does more complex things like running processes in the background),
    85  	// which will be the child process of the /bin/sh one.
    86  	//
    87  	// Redirecting to /proc/1/fd/* allows to redirect the process output to the output streams of PID 1 process inside the container.
    88  	// This way, returning the container logs with 'odo logs' or 'kubectl logs' would work seamlessly.
    89  	// See https://stackoverflow.com/questions/58716574/where-exactly-do-the-logs-of-kubernetes-pods-come-from-at-the-container-level
    90  	pidFile := getPidFileForCommand(def)
    91  	cmd := []string{
    92  		ShellExecutable, "-c",
    93  		fmt.Sprintf("echo $$ > %[1]s && %s %s (%s) 1>>/proc/1/fd/1 2>>/proc/1/fd/2; echo $? >> %[1]s", pidFile, cdCmd, setEnvCmd, cmdLine),
    94  	}
    95  
    96  	go func() {
    97  		if outputHandler != nil {
    98  			outputHandler(Starting, nil, nil, nil)
    99  		}
   100  
   101  		stdout, stderr, err := ExecuteCommand(cmd, kclient, podName, containerName, false, nil, nil)
   102  		if err != nil {
   103  			klog.V(2).Infof("error while running background command: %v", err)
   104  		}
   105  
   106  		if outputHandler != nil {
   107  			processInfo, infoErr := k.GetProcessInfoForCommand(def, kclient, podName, containerName)
   108  			if infoErr != nil {
   109  				outputHandler(Errored, stdout, stderr, err)
   110  				return
   111  			}
   112  			outputHandler(processInfo.Status, stdout, stderr, err)
   113  		}
   114  	}()
   115  
   116  	return nil
   117  }

   119  // StopProcessForCommand stops the process representing the specified Devfile command.
   120  // Because of the way this process is launched and its PID stored (see StartProcessForCommand),
   121  // we need to determine the process children (there should be only one child which is the sub-shell running the command passed to StartProcessForCommand).
   122  // Then killing those children will exit the parent 'sh' process.
   123  func (k *kubeExecProcessHandler) StopProcessForCommand(
   124  	def CommandDefinition,
   125  	kclient kclient.ClientInterface,
   126  	podName string,
   127  	containerName string,
   128  ) error {
   129  	klog.V(4).Infof("StopProcessForCommand for %q", def.Id)
   130  	defer func() {
   131  		pidFile := getPidFileForCommand(def)
   132  		_, _, err := ExecuteCommand([]string{ShellExecutable, "-c", fmt.Sprintf("rm -f %s", pidFile)},
   133  			kclient, podName, containerName, false, nil, nil)
   134  		if err != nil {
   135  			klog.V(2).Infof("Could not remove file %q: %v", pidFile, err)
   136  		}
   137  	}()
   138  
   139  	kill := func(p int) error {
   140  		_, _, err := ExecuteCommand([]string{ShellExecutable, "-c", fmt.Sprintf("kill %d || true", p)},
   141  			kclient, podName, containerName, false, nil, nil)
   142  		if err != nil {
   143  			return err
   144  		}
   145  
   146  		//Because the process we just stopped might take longer to exit (it might have caught the signal and is performing additional cleanup),
   147  		//retry detecting its actual state till it is stopped or timeout expires
   148  		var processInfo interface{}
   149  		processInfo, err = task.NewRetryable(fmt.Sprintf("status for remote process %d", p), func() (bool, interface{}, error) {
   150  			pInfo, e := k.getProcessInfoFromPid(p, 0, kclient, podName, containerName)
   151  			return e == nil && (pInfo.Status == Stopped || pInfo.Status == Errored), pInfo, e
   152  		}).RetryWithSchedule([]time.Duration{2 * time.Second, 4 * time.Second, 8 * time.Second}, true)
   153  		if err != nil {
   154  			return err
   155  		}
   156  
   157  		pInfo, ok := processInfo.(RemoteProcessInfo)
   158  		if !ok {
   159  			klog.V(2).Infof("invalid type for remote process (%d) info, expected RemoteProcessInfo", p)
   160  			return fmt.Errorf("internal error while checking remote process status: %d", p)
   161  		}
   162  		if pInfo.Status != Stopped && pInfo.Status != Errored {
   163  			return fmt.Errorf("invalid status for remote process %d: %+v", p, processInfo)
   164  		}
   165  		return nil
   166  	}
   167  
   168  	ppid, _, err := getRemoteProcessPID(kclient, def, podName, containerName)
   169  	if err != nil {
   170  		return err
   171  	}
   172  	if ppid == 0 {
   173  		return nil
   174  	}
   175  
   176  	children, err := getProcessChildren(ppid, kclient, podName, containerName)
   177  	if err != nil {
   178  		return err
   179  	}
   180  
   181  	if len(children) == 0 {
   182  		//TODO(rm3l): A length of 0 might indicate that there is no children file, which might happen if the host kernel
   183  		//was not built with the CONFIG_PROC_CHILDREN config.
   184  		//This happened for example with the Minikube VM when using its (non-default) VirtualBox driver.
   185  		//In this case, we should find a fallback mechanism to identify those children processes and kill them.
   186  		return kill(ppid)
   187  	}
   188  
   189  	for _, child := range children {
   190  		if err = kill(child); err != nil {
   191  			return err
   192  		}
   193  	}
   194  
   195  	return nil
   196  }

   198  func getRemoteProcessPID(kclient kclient.ClientInterface, def CommandDefinition, podName string, containerName string) (int, int, error) {
   199  	pidFile := getPidFileForCommand(def)
   200  	stdout, stderr, err := ExecuteCommand(
   201  		[]string{ShellExecutable, "-c", fmt.Sprintf("cat %s || true", pidFile)},
   202  		kclient, podName, containerName, false, nil, nil)
   203  
   204  	if err != nil {
   205  		return 0, 0, err
   206  	}
   207  
   208  	if len(stdout) == 0 {
   209  		//The file does not exist. We assume the process has not run yet.
   210  		return 0, 0, nil
   211  	}
   212  	if len(stdout) > 2 {
   213  		return 0, 0, fmt.Errorf(
   214  			"unable to determine command status, due to unexpected number of lines for %s, output: %v %v",
   215  			pidFile, stdout, stderr)
   216  	}
   217  
   218  	line := stdout[0]
   219  	var pid int
   220  	pid, err = strconv.Atoi(strings.TrimSpace(line))
   221  	if err != nil {
   222  		klog.V(2).Infof("error while trying to retrieve status of command: %+v", err)
   223  		return 0, 0, fmt.Errorf("unable to determine command status, due to unexpected PID content for %s: %s",
   224  			pidFile, line)
   225  	}
   226  	if len(stdout) == 1 {
   227  		return pid, 0, nil
   228  	}
   229  
   230  	line2 := stdout[1]
   231  	var exitStatus int
   232  	exitStatus, err = strconv.Atoi(strings.TrimSpace(line2))
   233  	if err != nil {
   234  		klog.V(2).Infof("error while trying to retrieve status of command: %+v", err)
   235  		return pid, 0, fmt.Errorf("unable to determine command status, due to unexpected exit status content for %s: %s",
   236  			pidFile, line2)
   237  	}
   238  
   239  	return pid, exitStatus, nil
   240  }

   242  func (k *kubeExecProcessHandler) getProcessInfoFromPid(
   243  	pid int,
   244  	lastKnownExitStatus int,
   245  	kclient kclient.ClientInterface,
   246  	podName string,
   247  	containerName string,
   248  ) (RemoteProcessInfo, error) {
   249  	process := RemoteProcessInfo{Pid: pid}
   250  
   251  	if pid < 0 {
   252  		process.Status = Unknown
   253  		return process, fmt.Errorf("invalid PID value for remote process: %d", pid)
   254  	}
   255  	if pid == 0 {
   256  		process.Status = Stopped
   257  		return process, nil
   258  	}
   259  
   260  	//Now check that the PID value is a valid process
   261  	stdout, _, err := ExecuteCommand(
   262  		[]string{ShellExecutable, "-c", fmt.Sprintf("kill -0 %d; echo $?", pid)},
   263  		kclient, podName, containerName, false, nil, nil)
   264  
   265  	if err != nil {
   266  		process.Status = Unknown
   267  		return process, err
   268  	}
   269  
   270  	var killStatus int
   271  	killStatus, err = strconv.Atoi(strings.TrimSpace(stdout[0]))
   272  	if err != nil {
   273  		process.Status = Unknown
   274  		return process, err
   275  	}
   276  
   277  	if killStatus == 0 {
   278  		process.Status = Running
   279  	} else {
   280  		if lastKnownExitStatus == 0 {
   281  			process.Status = Stopped
   282  		} else {
   283  			process.Status = Errored
   284  		}
   285  	}
   286  
   287  	return process, nil
   288  }

   290  // getProcessChildren returns the children of the specified process in the given container.
   291  // It works by reading the /proc/<pid>/task/<pid>/children file, which is a space-separated list of children
   292  func getProcessChildren(pid int, kclient kclient.ClientInterface, podName string, containerName string) ([]int, error) {
   293  	if pid <= 0 {
   294  		return nil, fmt.Errorf("invalid pid: %d", pid)
   295  	}
   296  
   297  	stdout, _, err := ExecuteCommand(
   298  		[]string{ShellExecutable, "-c", fmt.Sprintf("cat /proc/%[1]d/task/%[1]d/children || true", pid)},
   299  		kclient, podName, containerName, false, nil, nil)
   300  	if err != nil {
   301  		return nil, err
   302  	}
   303  
   304  	var children []int
   305  	for _, line := range stdout {
   306  		l := strings.Split(strings.TrimSpace(line), " ")
   307  		for _, p := range l {
   308  			c, err := strconv.Atoi(p)
   309  			if err != nil {
   310  				return children, err
   311  			}
   312  			children = append(children, c)
   313  		}
   314  	}
   315  
   316  	return children, nil
   317  }
