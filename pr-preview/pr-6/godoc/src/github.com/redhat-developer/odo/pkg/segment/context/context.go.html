    56  // SetClusterType sets clusterType property for telemetry data when a component is pushed or a project is created/set
    57  func SetClusterType(ctx context.Context, client kclient.ClientInterface) {
    58  	var value string
    59  	if client == nil {
    60  		value = NOTFOUND
    61  	} else {
    62  		// We are not checking ServerVersion to decide the cluster type because it does not always return the version,
    63  		// it sometimes fails to retrieve the data if user is using minishift or plain oc cluster
    64  		isOC, err := client.IsProjectSupported()
    65  		if err != nil {
    66  			klog.V(3).Info(fmt.Errorf("unable to detect project support: %w", err))
    67  			value = NOTFOUND
    68  		} else {
    69  			if isOC {
    70  				isOC4, err := client.IsCSVSupported()
    71  				// TODO: Add a unit test for this case
    72  				if err != nil {
    73  					value = "openshift"
    74  				} else {
    75  					if isOC4 {
    76  						value = "openshift4"
    77  					} else {
    78  						value = "openshift3"
    79  					}
    80  				}
    81  			} else {
    82  				value = "kubernetes"
    83  			}
    84  		}
    85  	}
    86  	setContextProperty(ctx, ClusterType, value)
    87  }

   123  // set safely sets value for a key in storage
   124  func (p *properties) set(name string, value interface{}) {
   125  	p.lock.Lock()
   126  	defer p.lock.Unlock()
   127  	p.storage[name] = value
   128  }

   130  // values safely retrieves a deep copy of the storage
   131  func (p *properties) values() map[string]interface{} {
   132  	p.lock.Lock()
   133  	defer p.lock.Unlock()
   134  	ret := make(map[string]interface{})
   135  	for k, v := range p.storage {
   136  		ret[k] = v
   137  	}
   138  	return ret
   139  }

   141  // propertiesFromContext retrieves the properties instance from the context
   142  func propertiesFromContext(ctx context.Context) *properties {
   143  	value := ctx.Value(key)
   144  	if cast, ok := value.(*properties); ok {
   145  		return cast
   146  	}
   147  	return nil
   148  }
