    87  func (o *AddBindingOptions) Run(_ context.Context) error {
    88  	withDevfile := o.EnvSpecificInfo.GetDevfileObj().Data != nil
    89  
    90  	ns, err := o.clientset.BindingClient.SelectNamespace(o.flags)
    91  	if err != nil {
    92  		return err
    93  	}
    94  
    95  	serviceMap, err := o.clientset.BindingClient.GetServiceInstances(ns)
    96  	if err != nil {
    97  		return err
    98  	}
    99  
   100  	if len(serviceMap) == 0 {
   101  		msg := "current namespace"
   102  		if ns != "" {
   103  			msg = fmt.Sprintf("namespace %q", ns)
   104  		}
   105  		return fmt.Errorf("No bindable service instances found in %s", msg)
   106  	}
   107  
   108  	service, err := o.clientset.BindingClient.SelectServiceInstance(o.flags, serviceMap)
   109  	if err != nil {
   110  		return err
   111  	}
   112  	splitService := strings.Split(service, " ")
   113  	serviceName := splitService[0]
   114  
   115  	var componentName string
   116  	var workloadName string
   117  	var workloadGVK schema.GroupVersionKind
   118  
   119  	if !withDevfile {
   120  		workloadName, workloadGVK, err = o.clientset.BindingClient.SelectWorkloadInstance(o.flags)
   121  		if err != nil {
   122  			return err
   123  		}
   124  		componentName = workloadName
   125  	} else {
   126  		componentName = o.EnvSpecificInfo.GetDevfileObj().GetMetadataName()
   127  	}
   128  
   129  	bindingName, err := o.clientset.BindingClient.AskBindingName(serviceName, componentName, o.flags)
   130  	if err != nil {
   131  		return err
   132  	}
   133  
   134  	bindAsFiles, err := o.clientset.BindingClient.AskBindAsFiles(o.flags)
   135  	if err != nil {
   136  		return err
   137  	}
   138  
   139  	namingStrategy, err := o.clientset.BindingClient.AskNamingStrategy(o.flags)
   140  	if err != nil {
   141  		return err
   142  	}
   143  
   144  	if withDevfile {
   145  		var devfileobj parser.DevfileObj
   146  		devfileobj, err = o.clientset.BindingClient.AddBindingToDevfile(
   147  			bindingName, bindAsFiles, ns, namingStrategy, serviceMap[service], o.EnvSpecificInfo.GetDevfileObj())
   148  		if err != nil {
   149  			return err
   150  		}
   151  		err = devfileobj.WriteYamlDevfile()
   152  		if err != nil {
   153  			return err
   154  		}
   155  		log.Success("Successfully added the binding to the devfile.")
   156  
   157  		exitMessage := "Run `odo dev` to create it on the cluster."
   158  		if len(o.flags) == 0 {
   159  			kindGroup := strings.ReplaceAll(strings.ReplaceAll(splitService[1], "(", ""), ")", "")
   160  			exitMessage += fmt.Sprintf("\nYou can automate this command by executing:\n  odo add binding --service %s.%s --name %s", serviceName, kindGroup, bindingName)
   161  			if ns != "" {
   162  				exitMessage += fmt.Sprintf(" --service-namespace %s", ns)
   163  			}
   164  			if !bindAsFiles {
   165  				exitMessage += " --bind-as-files=false"
   166  			}
   167  			if namingStrategy != "" {
   168  				exitMessage += " --naming-strategy='" + namingStrategy + "'"
   169  			}
   170  		}
   171  		log.Infof(exitMessage)
   172  		return nil
   173  	}
   174  
   175  	options, output, filename, err := o.clientset.BindingClient.AddBinding(
   176  		o.flags, bindingName, bindAsFiles, ns, namingStrategy, serviceMap[service], workloadName, workloadGVK)
   177  	if err != nil {
   178  		return err
   179  	}
   180  
   181  	if output != "" {
   182  		fmt.Println(output)
   183  	}
   184  
   185  	// Display the info after outputting to stdout
   186  	for _, option := range options {
   187  		switch option {
   188  		case asker.OutputToFile:
   189  			log.Infof("The ServiceBinding has been written to the file %q", filename)
   190  
   191  		case asker.CreateOnCluster:
   192  			log.Infof("The ServiceBinding has been created in the cluster")
   193  		}
   194  	}
   195  
   196  	return nil
   197  }

   213  	bindingCmd.Flags().String(backend.FLAG_NAME, "", "Name of the Binding to create")

   216  	bindingCmd.Flags().String(backend.FLAG_SERVICE_NAMESPACE, "", "Namespace of the service to bind to. Default is the component namespace.")
