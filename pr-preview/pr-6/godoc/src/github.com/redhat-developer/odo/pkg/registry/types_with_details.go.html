    11  // GetOrderedLabels returns a list of labels for a list of project types
    12  func (types TypesWithDetails) GetOrderedLabels() []string {
    13  	sortedTypes := sortTypes(types)
    14  	stringTypes := []string{}
    15  	for _, typ := range sortedTypes {
    16  		detailsList := types[typ]
    17  		if len(detailsList) == 1 {
    18  			stringTypes = append(stringTypes, typ)
    19  		} else {
    20  			for _, details := range detailsList {
    21  				stringTypes = append(stringTypes, fmt.Sprintf("%s (%s, registry: %s)", typ, details.Name, details.Registry.Name))
    22  			}
    23  		}
    24  	}
    25  	return stringTypes
