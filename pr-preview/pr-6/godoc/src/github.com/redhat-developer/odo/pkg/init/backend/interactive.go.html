    51  	var err error

    53  loop:

    68  			if err != nil {
    69  				return nil, err
    70  			}
    71  			if back {
    72  				state = STATE_ASK_LANG
    73  				continue loop
    74  			}
    75  			result.DevfileRegistry = details.Registry.Name
    76  			result.Devfile = details.Name
    77  			state = STATE_END
    78  		case STATE_END:
    79  			break loop
    80  		}
    81  	}
    82  
    83  	return result, nil

   110  func (o *InteractiveBackend) PersonalizeDevfileConfig(devfileobj parser.DevfileObj) (parser.DevfileObj, error) {
   111  	config, err := getPortsAndEnvVar(devfileobj)
   112  	var zeroDevfile parser.DevfileObj
   113  	if err != nil {
   114  		return zeroDevfile, err
   115  	}
   116  
   117  	var selectContainerAnswer string
   118  	containerOptions := config.GetContainers()
   119  	containerOptions = append(containerOptions, "NONE - configuration is correct")
   120  
   121  	for selectContainerAnswer != "NONE - configuration is correct" {
   122  		PrintConfiguration(config)
   123  		selectContainerAnswer, err = o.askerClient.AskContainerName(containerOptions)
   124  		if err != nil {
   125  			return zeroDevfile, err
   126  		}
   127  
   128  		selectedContainer := config[selectContainerAnswer]
   129  		if selectContainerAnswer == "NONE - configuration is correct" {
   130  			break
   131  		}
   132  
   133  		var configOps asker.OperationOnContainer
   134  		for configOps.Ops != "Nothing" {
   135  			configOps, err = o.askerClient.AskPersonalizeConfiguration(selectedContainer)
   136  			if err != nil {
   137  				return zeroDevfile, err
   138  			}
   139  			switch configOps.Ops {
   140  			case "Add":
   141  				switch configOps.Kind {
   142  				case "Port":
   143  					var newPort string
   144  					newPort, err = o.askerClient.AskAddPort()
   145  					if err != nil {
   146  						return zeroDevfile, err
   147  					}
   148  
   149  					err = devfileobj.Data.SetPorts(map[string][]string{selectContainerAnswer: {newPort}})
   150  					if err != nil {
   151  						return zeroDevfile, err
   152  					}
   153  					selectedContainer.Ports = append(selectedContainer.Ports, newPort)
   154  
   155  				case "EnvVar":
   156  					var newEnvNameAnswer, newEnvValueAnswer string
   157  					newEnvNameAnswer, newEnvValueAnswer, err = o.askerClient.AskAddEnvVar()
   158  					if err != nil {
   159  						return zeroDevfile, err
   160  					}
   161  					err = devfileobj.Data.AddEnvVars(map[string][]v1alpha2.EnvVar{selectContainerAnswer: {{
   162  						Name:  newEnvNameAnswer,
   163  						Value: newEnvValueAnswer,
   164  					}}})
   165  					if err != nil {
   166  						return zeroDevfile, err
   167  					}
   168  					selectedContainer.Envs[newEnvNameAnswer] = newEnvValueAnswer
   169  				}
   170  			case "Delete":
   171  				switch configOps.Kind {
   172  				case "Port":
   173  					portToDelete := configOps.Key
   174  					indexToDelete := -1
   175  					for i, port := range selectedContainer.Ports {
   176  						if port == portToDelete {
   177  							indexToDelete = i
   178  						}
   179  					}
   180  					if indexToDelete == -1 {
   181  						log.Warningf(fmt.Sprintf("unable to delete port %q, not found", portToDelete))
   182  					}
   183  					err = devfileobj.Data.RemovePorts(map[string][]string{selectContainerAnswer: {portToDelete}})
   184  					if err != nil {
   185  						return zeroDevfile, err
   186  					}
   187  					selectedContainer.Ports = append(selectedContainer.Ports[:indexToDelete], selectedContainer.Ports[indexToDelete+1:]...)
   188  
   189  				case "EnvVar":
   190  					envToDelete := configOps.Key
   191  					if _, ok := selectedContainer.Envs[envToDelete]; !ok {
   192  						log.Warningf(fmt.Sprintf("unable to delete env %q, not found", envToDelete))
   193  					}
   194  					err = devfileobj.Data.RemoveEnvVars(map[string][]string{selectContainerAnswer: {envToDelete}})
   195  					if err != nil {
   196  						return zeroDevfile, err
   197  					}
   198  					delete(selectedContainer.Envs, envToDelete)
   199  				}
   200  			case "Nothing":
   201  			default:
   202  				return zeroDevfile, fmt.Errorf("unknown configuration selected %q", fmt.Sprintf("%v %v %v", configOps.Ops, configOps.Kind, configOps.Key))
   203  			}
   204  			// Update the current configuration
   205  			config[selectContainerAnswer] = selectedContainer
   206  		}
   207  	}
   208  	return devfileobj, nil
   209  }

   237  	}

   243  func getPortsAndEnvVar(obj parser.DevfileObj) (asker.DevfileConfiguration, error) {
   244  	var config = asker.DevfileConfiguration{}
   245  	components, err := obj.Data.GetComponents(parsercommon.DevfileOptions{ComponentOptions: parsercommon.ComponentOptions{ComponentType: v1alpha2.ContainerComponentType}})
   246  	if err != nil {
   247  		return config, err
   248  	}
   249  	for _, component := range components {
   250  		var ports = []string{}
   251  		var envMap = map[string]string{}
   252  
   253  		for _, ep := range component.Container.Endpoints {
   254  			ports = append(ports, strconv.Itoa(ep.TargetPort))
   255  		}
   256  		for _, env := range component.Container.Env {
   257  			envMap[env.Name] = env.Value
   258  		}
   259  		config[component.Name] = asker.ContainerConfiguration{
   260  			Ports: ports,
   261  			Envs:  envMap,
   262  		}
   263  	}
   264  	return config, nil
   265  }
