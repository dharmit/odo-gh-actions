   132  // HumanReadableOutput outputs the list of bindings in a human readable format
   133  func HumanReadableOutput(namespace string, list api.ResourcesList) {
   134  	bindings := list.Bindings
   135  	if len(bindings) == 0 {
   136  		log.Errorf("There are no service bindings in the %q namespace.", namespace)
   137  		return
   138  
   139  	}
   140  
   141  	t := ui.NewTable()
   142  
   143  	// Create the header and then sort accordingly
   144  	t.AppendHeader(table.Row{"NAME", "APPLICATION", "SERVICES", "RUNNING IN"})
   145  	t.SortBy([]table.SortBy{
   146  		{Name: "NAME", Mode: table.Asc},
   147  	})
   148  
   149  	for _, binding := range bindings {
   150  
   151  		// Mark the name as yellow in the index to it's easier to see.
   152  		name := text.Colors{text.FgHiYellow}.Sprint(binding.Name)
   153  
   154  		for _, inDevfile := range list.BindingsInDevfile {
   155  			if binding.Name == inDevfile {
   156  				name = fmt.Sprintf("* %s", name)
   157  				break
   158  			}
   159  		}
   160  
   161  		appSpec := binding.Spec.Application
   162  		application := fmt.Sprintf("%s (%s)", appSpec.Name, appSpec.Kind)
   163  
   164  		servicesSpecs := binding.Spec.Services
   165  		services := ""
   166  		for i, serviceSpec := range servicesSpecs {
   167  			if i > 0 {
   168  				services += "\n"
   169  			}
   170  			group := schema.FromAPIVersionAndKind(serviceSpec.APIVersion, "").Group
   171  			if group != "" {
   172  				group = "." + group
   173  			}
   174  			svcDesc := fmt.Sprintf("%s (%s%s)",
   175  				serviceSpec.Name,
   176  				serviceSpec.Kind,
   177  				group,
   178  			)
   179  			if serviceSpec.Namespace != "" {
   180  				svcDesc += fmt.Sprintf(" (namespace: %s)", serviceSpec.Namespace)
   181  			}
   182  			services += svcDesc
   183  		}
   184  
   185  		runningIn := "None"
   186  		if binding.Status != nil && len(binding.Status.RunningIn) > 0 {
   187  			runningIn = binding.Status.RunningIn.String()
   188  		}
   189  		t.AppendRow(table.Row{name, application, services, runningIn})
   190  	}
   191  	t.Render()
   192  }
