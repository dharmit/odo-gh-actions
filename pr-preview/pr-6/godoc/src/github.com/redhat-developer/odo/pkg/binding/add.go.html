   137  func (o *BindingClient) AddBinding(
   138  	flags map[string]string,
   139  	bindingName string,
   140  	bindAsFiles bool,
   141  	serviceNs string,
   142  	namingStrategy string,
   143  	unstructuredService unstructured.Unstructured,
   144  	workloadName string,
   145  	workloadGVK schema.GroupVersionKind,
   146  ) ([]asker.CreationOption, string, string, error) {
   147  	service, err := o.kubernetesClient.NewServiceBindingServiceObject(serviceNs, unstructuredService, bindingName)
   148  	if err != nil {
   149  		return nil, "", "", err
   150  	}
   151  
   152  	serviceBinding := kclient.NewServiceBindingObject(
   153  		bindingName, bindAsFiles, workloadName, namingStrategy, workloadGVK, []sboApi.Mapping{}, []sboApi.Service{service}, sboApi.ServiceBindingStatus{})
   154  
   155  	var backend backendpkg.AddBindingBackend
   156  	if len(flags) == 0 {
   157  		backend = o.interactiveBackend
   158  	} else {
   159  		backend = o.flagsBackend
   160  	}
   161  
   162  	var options []asker.CreationOption
   163  	for len(options) == 0 {
   164  		options, err = backend.SelectCreationOptions(flags)
   165  		if err != nil {
   166  			return nil, "", "", err
   167  		}
   168  	}
   169  
   170  	// Note: we cannot directly marshal the serviceBinding object to yaml because it doesn't do that in the correct k8s manifest format
   171  	serviceBindingUnstructured, err := kclient.ConvertK8sResourceToUnstructured(serviceBinding)
   172  	if err != nil {
   173  		return nil, "", "", err
   174  	}
   175  	yamlDesc, err := yaml.Marshal(serviceBindingUnstructured.UnstructuredContent())
   176  	if err != nil {
   177  		return nil, "", "", err
   178  	}
   179  
   180  	var filename string
   181  	for _, option := range options {
   182  		if option == asker.OutputToFile {
   183  			filename, err = backend.AskOutputFilePath(flags, filepath.Join("kubernetes", serviceBinding.GetName()+".yaml"))
   184  			if err != nil {
   185  				return nil, "", "", err
   186  			}
   187  			break
   188  		}
   189  	}
   190  
   191  	var output string
   192  	for _, option := range options {
   193  		switch option {
   194  		case asker.OutputToFile:
   195  			err = os.MkdirAll(filepath.Dir(filename), 0750)
   196  			if err != nil {
   197  				return nil, "", "", err
   198  			}
   199  			err = os.WriteFile(filename, yamlDesc, 0600)
   200  			if err != nil {
   201  				return nil, "", "", err
   202  			}
   203  
   204  		case asker.OutputToStdout:
   205  			output = string(yamlDesc)
   206  
   207  		case asker.CreateOnCluster:
   208  			_, err = o.kubernetesClient.PatchDynamicResource(serviceBindingUnstructured)
   209  			if err != nil {
   210  				return nil, "", "", err
   211  			}
   212  		}
   213  	}
   214  
   215  	return options, output, filename, nil
   216  }
