     1  // Package clientset is used to inject clients inside commands
     2  //
     3  // To use this package:
     4  // From a command definition, use the `Add` function to declare the clients needed by the command
     5  // Then, from the `SetClientset` method of the `Runnable` interface, you can access the clients
     6  //
     7  // To add a new client to this package:
     8  // - add a new constant for the client
     9  // - if the client has sub-dependencies, define a new entry in the map of sub-dependencies
    10  // - add the packages's client to the Clientset structure
    11  // - complete the Fetch function to instantiate the package's client

   109  func Add(command *cobra.Command, dependencies ...string) {
   110  	if command.Annotations == nil {
   111  		command.Annotations = map[string]string{}
   112  	}
   113  	for _, dependency := range dependencies {
   114  		_, ok := command.Annotations[dependency]
   115  		// prevent infinite loop with circular dependencies
   116  		if !ok {
   117  			command.Annotations[dependency] = "true"
   118  			Add(command, subdeps[dependency]...)
   119  		}
   120  	}
   121  }

   123  func isDefined(command *cobra.Command, dependency string) bool {
   124  	_, ok := command.Annotations[dependency]
   125  	return ok
   126  }

   128  func Fetch(command *cobra.Command) (*Clientset, error) {
   129  	dep := Clientset{}
   130  	var err error
   131  
   132  	/* Without sub-dependencies */
   133  	if isDefined(command, FILESYSTEM) {
   134  		dep.FS = filesystem.DefaultFs{}
   135  	}
   136  	if isDefined(command, KUBERNETES) || isDefined(command, KUBERNETES_NULLABLE) {
   137  		dep.KubernetesClient, err = kclient.New()
   138  		if err != nil && isDefined(command, KUBERNETES) {
   139  			return nil, err
   140  		}
   141  	}
   142  	if isDefined(command, PREFERENCE) {
   143  		dep.PreferenceClient, err = preference.NewClient()
   144  		if err != nil {
   145  			return nil, err
   146  		}
   147  	}
   148  	if isDefined(command, REGISTRY) {
   149  		dep.RegistryClient = registry.NewRegistryClient(dep.FS, dep.PreferenceClient)
   150  	}
   151  
   152  	/* With sub-dependencies */
   153  	if isDefined(command, ALIZER) {
   154  		dep.AlizerClient = alizer.NewAlizerClient(dep.RegistryClient)
   155  	}
   156  	if isDefined(command, DELETE_COMPONENT) {
   157  		dep.DeleteClient = _delete.NewDeleteComponentClient(dep.KubernetesClient)
   158  	}
   159  	if isDefined(command, DEPLOY) {
   160  		dep.DeployClient = deploy.NewDeployClient(dep.KubernetesClient)
   161  	}
   162  	if isDefined(command, INIT) {
   163  		dep.InitClient = _init.NewInitClient(dep.FS, dep.PreferenceClient, dep.RegistryClient, dep.AlizerClient)
   164  	}
   165  	if isDefined(command, LOGS) {
   166  		dep.LogsClient = logs.NewLogsClient(dep.KubernetesClient)
   167  	}
   168  	if isDefined(command, PROJECT) {
   169  		dep.ProjectClient = project.NewClient(dep.KubernetesClient)
   170  	}
   171  	if isDefined(command, STATE) {
   172  		dep.StateClient = state.NewStateClient(dep.FS)
   173  	}
   174  	if isDefined(command, WATCH) {
   175  		dep.WatchClient = watch.NewWatchClient(dep.KubernetesClient, dep.DeleteClient, dep.StateClient)
   176  	}
   177  	if isDefined(command, BINDING) {
   178  		dep.BindingClient = binding.NewBindingClient(dep.ProjectClient, dep.KubernetesClient)
   179  	}
   180  	if isDefined(command, PORT_FORWARD) {
   181  		dep.PortForwardClient = portForward.NewPFClient(dep.KubernetesClient, dep.StateClient)
   182  	}
   183  	if isDefined(command, DEV) {
   184  		dep.DevClient = dev.NewDevClient(dep.KubernetesClient, dep.PreferenceClient, dep.PortForwardClient, dep.WatchClient, dep.BindingClient)
   185  	}
   186  
   187  	/* Instantiate new clients here. Take care to instantiate after all sub-dependencies */
   188  	return &dep, nil
   189  }
