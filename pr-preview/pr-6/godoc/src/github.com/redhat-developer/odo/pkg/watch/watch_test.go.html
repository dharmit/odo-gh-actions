    16  func evaluateChangesHandler(events []fsnotify.Event, path string, fileIgnores []string, watcher *fsnotify.Watcher) ([]string, []string) {
    17  	var changedFiles []string
    18  	var deletedPaths []string
    19  
    20  	for _, event := range events {
    21  		for _, file := range fileIgnores {
    22  			// if file is in fileIgnores, don't do anything
    23  			if event.Name == file {
    24  				continue
    25  			}
    26  		}
    27  		// this if condition is copied from original implementation in watch.go. Code within the if block is simplified for tests
    28  		if event.Op&fsnotify.Remove == fsnotify.Remove || event.Op&fsnotify.Rename == fsnotify.Rename {
    29  			// On remove/rename, stop watching the resource
    30  			_ = watcher.Remove(event.Name)
    31  			// Append the file to list of deleted files
    32  			deletedPaths = append(deletedPaths, event.Name)
    33  			continue
    34  		}
    35  		// this if condition is copied from original implementation in watch.go. Code within the if block is simplified for tests
    36  		if event.Op&fsnotify.Remove != fsnotify.Remove {
    37  			changedFiles = append(changedFiles, event.Name)
    38  		}
    39  	}
    40  	return changedFiles, deletedPaths
    41  }
