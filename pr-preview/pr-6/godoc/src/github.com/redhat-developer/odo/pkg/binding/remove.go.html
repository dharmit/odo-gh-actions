    28  func (o *BindingClient) RemoveBinding(servicebindingName string, obj parser.DevfileObj) (parser.DevfileObj, error) {
    29  	var componentName string
    30  	var options []string
    31  	// Get all the K8s type devfile components
    32  	components, err := obj.Data.GetComponents(common.DevfileOptions{
    33  		ComponentOptions: common.ComponentOptions{ComponentType: v1alpha2.KubernetesComponentType},
    34  	})
    35  	if err != nil {
    36  		return obj, err
    37  	}
    38  	for _, component := range components {
    39  		var unstructuredObj unstructured.Unstructured
    40  		// Parse the K8s manifest
    41  		unstructuredObj, err = libdevfile.GetK8sComponentAsUnstructured(obj, component.Name, filepath.Dir(obj.Ctx.GetAbsPath()), devfilefs.DefaultFs{})
    42  		if err != nil {
    43  			continue
    44  		}
    45  		if unstructuredObj.GetKind() == kclient.ServiceBindingKind {
    46  			options = append(options, unstructuredObj.GetName())
    47  			if unstructuredObj.GetName() == servicebindingName {
    48  				componentName = component.Name
    49  				break
    50  			}
    51  		}
    52  	}
    53  	if componentName == "" {
    54  		return obj, fmt.Errorf("Service Binding %q not found in the devfile. Available Service Bindings: %s", servicebindingName, strings.Join(options, ", "))
    55  	}
    56  
    57  	err = obj.Data.DeleteComponent(componentName)
    58  	return obj, err
    59  }
