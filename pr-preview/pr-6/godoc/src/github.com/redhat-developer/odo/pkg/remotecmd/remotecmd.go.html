    16  // ExecuteCommand executes the given command in the pod's container,
    17  // writing the output to the specified respective pipe writers
    18  func ExecuteCommand(
    19  	command []string,
    20  	client kclient.ClientInterface,
    21  	podName string,
    22  	containerName string,
    23  	show bool,
    24  	stdoutWriter *io.PipeWriter,
    25  	stderrWriter *io.PipeWriter,
    26  ) (stdout []string, stderr []string, err error) {
    27  	soutReader, soutWriter := io.Pipe()
    28  	serrReader, serrWriter := io.Pipe()
    29  
    30  	klog.V(2).Infof("Executing command %v for pod: %v in container: %v", command, podName, containerName)
    31  
    32  	// Read stdout and stderr, store their output in cmdOutput, and also pass output to consoleOutput Writers (if non-nil)
    33  	stdoutCompleteChannel := startReaderGoroutine(soutReader, show, &stdout, stdoutWriter)
    34  	stderrCompleteChannel := startReaderGoroutine(serrReader, show, &stderr, stderrWriter)
    35  
    36  	err = client.ExecCMDInContainer(containerName, podName, command, soutWriter, serrWriter, nil, false)
    37  
    38  	// Block until we have received all the container output from each stream
    39  	_ = soutWriter.Close()
    40  	<-stdoutCompleteChannel
    41  	_ = serrWriter.Close()
    42  	<-stderrCompleteChannel
    43  
    44  	if err != nil {
    45  		// It is safe to read from stdout and stderr here, as the goroutines are guaranteed to have terminated at this point.
    46  		klog.V(2).Infof("ExecuteCommand returned an an err: %v. for command '%v'\nstdout: %v\nstderr: %v",
    47  			err, command, stdout, stderr)
    48  
    49  		msg := fmt.Sprintf("unable to exec command %v", command)
    50  		if len(stdout) != 0 {
    51  			msg += fmt.Sprintf("\n=== stdout===\n%s", strings.Join(stdout, "\n"))
    52  		}
    53  		if len(stderr) != 0 {
    54  			msg += fmt.Sprintf("\n=== stderr===\n%s", strings.Join(stderr, "\n"))
    55  		}
    56  		return stdout, stderr, fmt.Errorf("%s: %w", msg, err)
    57  	}
    58  
    59  	return stdout, stderr, err
    60  }

    64  // The returned channel will contain a single nil entry once the reader has closed.
    65  func startReaderGoroutine(reader io.Reader, show bool, cmdOutput *[]string, consoleOutput *io.PipeWriter) chan interface{} {
    66  	result := make(chan interface{})
    67  
    68  	go func() {
    69  		scanner := bufio.NewScanner(reader)
    70  		for scanner.Scan() {
    71  			line := scanner.Text()
    72  
    73  			if show || log.IsDebug() {
    74  				_, err := fmt.Fprintln(os.Stdout, line)
    75  				if err != nil {
    76  					log.Errorf("Unable to print to stdout: %s", err.Error())
    77  				}
    78  			}
    79  
    80  			if cmdOutput != nil {
    81  				*cmdOutput = append(*cmdOutput, line)
    82  			}
    83  
    84  			if consoleOutput != nil {
    85  				_, err := consoleOutput.Write([]byte(line + "\n"))
    86  				if err != nil {
    87  					log.Errorf("Error occurred on writing string to consoleOutput writer: %s", err.Error())
    88  				}
    89  			}
    90  		}
    91  		result <- nil
    92  	}()
    93  
    94  	return result
