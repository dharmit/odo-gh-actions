    48  func (a *adapterHandler) Execute(devfileCmd devfilev1.Command) error {
    49  	remoteProcessHandler := remotecmd.NewKubeExecProcessHandler()
    50  
    51  	statusHandlerFunc := func(s *log.Status) remotecmd.CommandOutputHandler {
    52  		return func(status remotecmd.RemoteProcessStatus, stdout []string, stderr []string, err error) {
    53  			switch status {
    54  			case remotecmd.Starting:
    55  				// Creating with no spin because the command could be long-running, and we cannot determine when it will end.
    56  				s.Start(fmt.Sprintf("Executing the application (command: %s)", devfileCmd.Id), true)
    57  			case remotecmd.Stopped, remotecmd.Errored:
    58  				s.EndWithStatus(fmt.Sprintf("Finished executing the application (command: %s)", devfileCmd.Id), status == remotecmd.Stopped)
    59  				if err != nil {
    60  					klog.V(2).Infof("error while running background command: %v", err)
    61  				}
    62  			}
    63  		}
    64  	}
    65  
    66  	// Spinner created but not started yet.
    67  	// It will be displayed when the statusHandlerFunc function is called with the "Starting" state.
    68  	spinner := log.NewStatus(log.GetStdout())
    69  
    70  	// if we need to restart, issue the remote process handler command to stop all running commands first.
    71  	// We do not need to restart Hot reload capable commands.
    72  	if a.componentExists {
    73  		if devfileCmd.Exec == nil || !util.SafeGetBool(devfileCmd.Exec.HotReloadCapable) {
    74  			klog.V(2).Infof("restart required for command %s", devfileCmd.Id)
    75  
    76  			cmdDef, err := devfileCommandToRemoteCmdDefinition(devfileCmd)
    77  			if err != nil {
    78  				return err
    79  			}
    80  
    81  			err = remoteProcessHandler.StopProcessForCommand(cmdDef, a.kubeClient, a.podName, devfileCmd.Exec.Component)
    82  			if err != nil {
    83  				return err
    84  			}
    85  
    86  			if err = remoteProcessHandler.StartProcessForCommand(cmdDef, a.kubeClient, a.podName, devfileCmd.Exec.Component, statusHandlerFunc(spinner)); err != nil {
    87  				return err
    88  			}
    89  		} else {
    90  			klog.V(2).Infof("command is hot-reload capable, not restarting %s", devfileCmd.Id)
    91  		}
    92  	} else {
    93  		cmdDef, err := devfileCommandToRemoteCmdDefinition(devfileCmd)
    94  		if err != nil {
    95  			return err
    96  		}
    97  
    98  		if err := remoteProcessHandler.StartProcessForCommand(cmdDef, a.kubeClient, a.podName, devfileCmd.Exec.Component, statusHandlerFunc(spinner)); err != nil {
    99  			return err
   100  		}
   101  	}
   102  
   103  	retrySchedule := []time.Duration{
   104  		5 * time.Second,
   105  		6 * time.Second,
   106  		9 * time.Second,
   107  	}
   108  	var totalWaitTime float64
   109  	for _, s := range retrySchedule {
   110  		totalWaitTime += s.Seconds()
   111  	}
   112  
   113  	_, err := task.NewRetryable(fmt.Sprintf("process for command %q", devfileCmd.Id), func() (bool, interface{}, error) {
   114  		klog.V(4).Infof("checking if process for command %q is running", devfileCmd.Id)
   115  		remoteProcess, err := remoteProcessHandler.GetProcessInfoForCommand(
   116  			remotecmd.CommandDefinition{Id: devfileCmd.Id}, a.kubeClient, a.podName, devfileCmd.Exec.Component)
   117  		if err != nil {
   118  			return false, nil, err
   119  		}
   120  		isRunningOrDone := remoteProcess.Status == remotecmd.Running ||
   121  			remoteProcess.Status == remotecmd.Stopped ||
   122  			remoteProcess.Status == remotecmd.Errored
   123  		return isRunningOrDone, nil, err
   124  	}).RetryWithSchedule(retrySchedule, false)
   125  	if err != nil {
   126  		return err
   127  	}
   128  
   129  	return a.checkRemoteCommandStatus(devfileCmd, a.podName,
   130  		fmt.Sprintf("Devfile command %q exited with an error status in %.0f second(s)", devfileCmd.Id, totalWaitTime))
   131  }

   133  // ApplyKubernetes contains the logic to create the k8s resources defined by the `apply` command
   134  // mode(Dev, Deploy): the mode in which the resources are deployed
   135  // appName: application name
   136  // devfile: the devfile object
   137  // kubernetes: the kubernetes devfile component to be deployed
   138  // kubeClient: Kubernetes client to be used to deploy the resource
   139  // path: path to the context directory
   140  func ApplyKubernetes(mode, appName string, devfile parser.DevfileObj, kubernetes devfilev1.Component, kubeClient kclient.ClientInterface, path string) error {
   141  	// Validate if the GVRs represented by Kubernetes inlined components are supported by the underlying cluster
   142  	_, err := ValidateResourceExist(kubeClient, devfile, kubernetes, path)
   143  	if err != nil {
   144  		return err
   145  	}
   146  
   147  	// Get the most common labels that's applicable to all resources being deployed.
   148  	// Set the mode. Regardless of what Kubernetes resource we are deploying.
   149  	labels := odolabels.GetLabels(devfile.Data.GetMetadata().Name, appName, mode, false)
   150  
   151  	klog.V(4).Infof("Injecting labels: %+v into k8s artifact", labels)
   152  
   153  	// Create the annotations
   154  	// Retrieve the component type from the devfile and also inject it into the list of annotations
   155  	annotations := make(map[string]string)
   156  	odolabels.SetProjectType(annotations, component.GetComponentTypeFromDevfileMetadata(devfile.Data.GetMetadata()))
   157  
   158  	// Get the Kubernetes component
   159  	u, err := libdevfile.GetK8sComponentAsUnstructured(devfile, kubernetes.Name, path, devfilefs.DefaultFs{})
   160  	if err != nil {
   161  		return err
   162  	}
   163  
   164  	// Deploy the actual Kubernetes component and error out if there's an issue.
   165  	log.Sectionf("Deploying Kubernetes Component: %s", u.GetName())
   166  	_, err = service.PushKubernetesResource(kubeClient, u, labels, annotations)
   167  	if err != nil {
   168  		return fmt.Errorf("failed to create service(s) associated with the component: %w", err)
   169  	}
   170  
   171  	return nil
   172  }

   174  // devfileCommandToRemoteCmdDefinition builds and returns a new remotecmd.CommandDefinition object from the specified devfileCmd.
   175  // An error is returned for non-exec Devfile commands.
   176  func devfileCommandToRemoteCmdDefinition(devfileCmd devfilev1.Command) (remotecmd.CommandDefinition, error) {
   177  	if devfileCmd.Exec == nil {
   178  		return remotecmd.CommandDefinition{}, errors.New(" only Exec commands are supported")
   179  	}
   180  
   181  	envVars := make([]remotecmd.CommandEnvVar, 0, len(devfileCmd.Exec.Env))
   182  	for _, e := range devfileCmd.Exec.Env {
   183  		envVars = append(envVars, remotecmd.CommandEnvVar{Key: e.Name, Value: e.Value})
   184  	}
   185  
   186  	return remotecmd.CommandDefinition{
   187  		Id:         devfileCmd.Id,
   188  		WorkingDir: devfileCmd.Exec.WorkingDir,
   189  		EnvVars:    envVars,
   190  		CmdLine:    devfileCmd.Exec.CommandLine,
   191  	}, nil
   192  }

   194  // isRemoteProcessForCommandRunning returns true if the command is running
   195  func (a *adapterHandler) isRemoteProcessForCommandRunning(command devfilev1.Command, podName string) (bool, error) {
   196  	remoteProcess, err := remotecmd.NewKubeExecProcessHandler().GetProcessInfoForCommand(
   197  		remotecmd.CommandDefinition{Id: command.Id}, a.kubeClient, podName, command.Exec.Component)
   198  	if err != nil {
   199  		return false, err
   200  	}
   201  
   202  	return remoteProcess.Status == remotecmd.Running, nil
   203  }

   205  // checkRemoteCommandStatus checks if the command is running .
   206  // if the command is not in a running state, we fetch the last 20 lines of the component's log and display it
   207  func (a *adapterHandler) checkRemoteCommandStatus(command devfilev1.Command, podName string, notRunningMessage string) error {
   208  	remoteProcessHandler := remotecmd.NewKubeExecProcessHandler()
   209  	remoteProcess, err := remoteProcessHandler.GetProcessInfoForCommand(remotecmd.CommandDefinition{Id: command.Id}, a.kubeClient, podName, command.Exec.Component)
   210  	if err != nil {
   211  		return err
   212  	}
   213  
   214  	if remoteProcess.Status != remotecmd.Running && remoteProcess.Status != remotecmd.Stopped {
   215  		log.Warningf(notRunningMessage)
   216  		log.Warningf("Last %d lines of log:", numberOfLinesToOutputLog)
   217  
   218  		rd, err := component.Log(a.kubeClient, a.ComponentName, a.AppName, false, command)
   219  		if err != nil {
   220  			return err
   221  		}
   222  
   223  		// Use GetStderr in order to make sure that colour output is correct
   224  		// on non-TTY terminals
   225  		err = util.DisplayLog(false, rd, log.GetStderr(), a.ComponentName, numberOfLinesToOutputLog)
   226  		if err != nil {
   227  			return err
   228  		}
   229  	}
   230  	return nil
   231  }
