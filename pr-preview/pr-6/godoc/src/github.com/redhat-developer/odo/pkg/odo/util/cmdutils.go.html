    13  func LogError(err error, context string) {
    14  	if err != nil {
    15  		// If it's JSON, we'll output  the error
    16  		if log.IsJSON() {
    17  
    18  			// Machine readble error output
    19  			machineOutput := api.GenericError{
    20  				Message: err.Error(),
    21  			}
    22  			// Output the error
    23  			machineoutput.OutputError(machineOutput)
    24  
    25  		} else {
    26  			if context == "" {
    27  				log.Error(err)
    28  			} else {
    29  				printstring := fmt.Sprintf("%s%s", strings.Title(context), "\nError: %v")
    30  				log.Errorf(printstring, err)
    31  			}
    32  		}
    33  	}
    34  }
