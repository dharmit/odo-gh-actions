    68  // IsTerminal returns true if the writer w is a terminal
    69  // This function is modified if we are running within Windows..
    70  // as Golang's built-in "IsTerminal" command only works on UNIX-based systems:
    71  // https://github.com/golang/crypto/blob/master/ssh/terminal/util.go#L5
    72  func IsTerminal(w io.Writer) bool {
    73  	if runtime.GOOS == "windows" {
    74  		return true
    75  	} else if v, ok := (w).(*os.File); ok {
    76  		return term.IsTerminal(int(v.Fd()))
    77  	}
    78  	return false
    79  }

    87  // Updates the status and makes sure that if the previous status was longer, it
    88  // "clears" the rest of the message.
    89  func (s *Status) updateStatus() {
    90  	mu.Lock()
    91  	if s.warningStatus != "" {
    92  		yellow := color.New(color.FgYellow).SprintFunc()
    93  
    94  		// Determine the warning size, so that we can calculate its length and use that length as padding parameter
    95  		warningSubstring := fmt.Sprintf(" [%s %s]", yellow(getWarningString()), yellow(s.warningStatus))
    96  
    97  		// Combine suffix and spacing, then resize them
    98  		newSuffix := fmt.Sprintf(suffixSpacing+"%s", s.status)
    99  		newSuffix = truncateSuffixIfNeeded(newSuffix, s.writer, len(warningSubstring))
   100  
   101  		// Combine the warning and non-warning text (since we don't want to truncate the warning text)
   102  		s.spinner.SetSuffix(fmt.Sprintf("%s%s", newSuffix, warningSubstring))
   103  	} else {
   104  		newSuffix := fmt.Sprintf(suffixSpacing+"%s", s.status)
   105  		s.spinner.SetSuffix(truncateSuffixIfNeeded(newSuffix, s.writer, 0))
   106  	}
   107  	mu.Unlock()
   108  }

   110  // Start starts a new phase of the status, if attached to a terminal
   111  // there will be a loading spinner with this status
   112  func (s *Status) Start(status string, debug bool) {
   113  	s.End(true)
   114  
   115  	// set new status
   116  	isTerm := IsTerminal(s.writer)
   117  	s.status = status
   118  
   119  	// If we are in debug mode, don't spin!
   120  	// In under no circumstances do we output if we're using -o json.. to
   121  	// to avoid parsing errors.
   122  	if !IsJSON() {
   123  		if !isTerm || debug {
   124  			fmt.Fprintf(s.writer, prefixSpacing+getSpacingString()+suffixSpacing+"%s  ...\n", s.status)
   125  		} else {
   126  			s.spinner.SetPrefix(prefixSpacing)
   127  			newSuffix := fmt.Sprintf(suffixSpacing+"%s", s.status)
   128  			s.spinner.SetSuffix(truncateSuffixIfNeeded(newSuffix, s.writer, 0))
   129  			s.spinner.Start()
   130  		}
   131  	}
   132  }

   134  // truncateSuffixIfNeeded returns a representation of the 'suffix' parameter that fits within the terminal
   135  // (including the extra space occupied by the padding parameter).
   136  func truncateSuffixIfNeeded(suffix string, w io.Writer, padding int) string {
   137  
   138  	terminalWidth := getTerminalWidth(w)
   139  	if terminalWidth == nil {
   140  		return suffix
   141  	}
   142  
   143  	// Additional padding to account for animation widget on lefthand side, and to avoid getting too close to the righthand terminal edge
   144  	const additionalPadding = 10
   145  
   146  	maxWidth := *terminalWidth - padding - additionalPadding
   147  
   148  	// For VERY small terminals, or very large padding, just return the suffix unmodified
   149  	if maxWidth <= 20 {
   150  		return suffix
   151  	}
   152  
   153  	// If we are compliant, return the unmodified suffix...
   154  	if len(suffix) <= maxWidth {
   155  		return suffix
   156  	}
   157  
   158  	// Otherwise truncate down to the desired length and append '...'
   159  	abbrevSuffix := "..."
   160  	maxWidth -= len(abbrevSuffix) // maxWidth is necessarily >20 at this point
   161  
   162  	// len(suffix) is necessarily >= maxWidth at this point
   163  	suffix = suffix[:maxWidth] + abbrevSuffix
   164  
   165  	return suffix
   166  }

   168  func getTerminalWidth(w io.Writer) *int {

   170  	if runtime.GOOS != "windows" {

   172  		if v, ok := (w).(*os.File); ok {

   174  			if err == nil {

   179  	}

   181  	return nil

   184  // End completes the current status, ending any previous spinning and
   185  // marking the status as success or failure
   186  func (s *Status) End(success bool) {
   187  	if s.status == "" {
   188  		return
   189  	}
   190  
   191  	isTerm := IsTerminal(s.writer)
   192  	if isTerm {
   193  		s.spinner.Stop()
   194  		if !IsJSON() {
   195  			fmt.Fprint(s.writer, "\r")
   196  		}
   197  	}
   198  
   199  	if !IsJSON() {
   200  
   201  		time := ""
   202  		if s.spinner.TimeSpent() != "" {
   203  			time = fmt.Sprintf("[%s]", s.spinner.TimeSpent())
   204  		}
   205  
   206  		if success {
   207  			// Clear the warning (unneeded now)
   208  			s.WarningStatus("")
   209  			green := color.New(color.FgGreen).SprintFunc()
   210  			fmt.Fprintf(s.writer, prefixSpacing+"%s"+suffixSpacing+"%s %s\n", green(getSuccessString()), s.status, time)
   211  		} else {
   212  			red := color.New(color.FgRed).SprintFunc()
   213  			if s.warningStatus != "" {
   214  				fmt.Fprintf(s.writer, prefixSpacing+"%s"+suffixSpacing+"%s %s [%s]\n", red(getErrString()), s.status, time, s.warningStatus)
   215  			} else {
   216  				fmt.Fprintf(s.writer, prefixSpacing+"%s"+suffixSpacing+"%s %s\n", red(getErrString()), s.status, time)
   217  			}
   218  		}
   219  	}
   220  
   221  	s.status = ""
   222  }
