    31  // GetVolumeInfos returns the PVC name attached to the `odo-projects` directory and a map of other PVCs
    32  func GetVolumeInfos(pvcs []corev1.PersistentVolumeClaim) (odoSourcePVCName string, infos map[string]VolumeInfo, _ error) {
    33  	infos = make(map[string]VolumeInfo)
    34  	for _, pvc := range pvcs {
    35  		// check if the pvc is in the terminating state
    36  		if pvc.DeletionTimestamp != nil {
    37  			continue
    38  		}
    39  
    40  		generatedVolumeName, e := generateVolumeNameFromPVC(pvc.Name)
    41  		if e != nil {
    42  			return "", nil, fmt.Errorf("unable to generate volume name from pvc name: %w", e)
    43  		}
    44  
    45  		storageName := odolabels.GetStorageName(pvc.Labels)
    46  		if storageName == storagepkg.OdoSourceVolume {
    47  			odoSourcePVCName = pvc.Name
    48  			continue
    49  		}
    50  
    51  		infos[storageName] = VolumeInfo{
    52  			PVCName:    pvc.Name,
    53  			VolumeName: generatedVolumeName,
    54  		}
    55  	}
    56  	return odoSourcePVCName, infos, nil
    57  }

    59  // GetPersistentVolumesAndVolumeMounts gets the PVC volumes and updates the containers with the volume mounts.
    60  // volumeNameToVolInfo is a map of the devfile volume name to the volume info containing the pvc name and the volume name.
    61  // To be moved to devfile/library.
    62  func GetPersistentVolumesAndVolumeMounts(devfileObj devfileParser.DevfileObj, containers []corev1.Container, initContainers []corev1.Container, volumeNameToVolInfo map[string]VolumeInfo, options parsercommon.DevfileOptions) ([]corev1.Volume, error) {
    63  
    64  	containerComponents, err := devfileObj.Data.GetDevfileContainerComponents(options)
    65  	if err != nil {
    66  		return nil, err
    67  	}
    68  
    69  	var pvcVols []corev1.Volume
    70  
    71  	// We need to sort volumes to create Deployment in a deterministic way
    72  	keys := make([]string, 0, len(volumeNameToVolInfo))
    73  	for k := range volumeNameToVolInfo {
    74  		keys = append(keys, k)
    75  	}
    76  	sort.Strings(keys)
    77  
    78  	for _, volName := range keys {
    79  		volInfo := volumeNameToVolInfo[volName]
    80  		pvcVols = append(pvcVols, getPVC(volInfo.VolumeName, volInfo.PVCName))
    81  
    82  		// containerNameToMountPaths is a map of the Devfile container name to their Devfile Volume Mount Paths for a given Volume Name
    83  		containerNameToMountPaths := make(map[string][]string)
    84  		for _, containerComp := range containerComponents {
    85  			for _, volumeMount := range containerComp.Container.VolumeMounts {
    86  				if volName == volumeMount.Name {
    87  					containerNameToMountPaths[containerComp.Name] = append(containerNameToMountPaths[containerComp.Name], generator.GetVolumeMountPath(volumeMount))
    88  				}
    89  			}
    90  		}
    91  
    92  		addVolumeMountToContainers(containers, initContainers, volInfo.VolumeName, containerNameToMountPaths)
    93  	}
    94  	return pvcVols, nil
    95  }

   149  func getEmptyDir(volumeName string, size string) (corev1.Volume, error) {
   150  
   151  	emptyDir := &corev1.EmptyDirVolumeSource{}
   152  	qty, err := resource.ParseQuantity(size)
   153  	if err != nil {
   154  		return corev1.Volume{}, err
   155  	}
   156  	emptyDir.SizeLimit = &qty
   157  	return corev1.Volume{
   158  		Name: volumeName,
   159  		VolumeSource: corev1.VolumeSource{
   160  			EmptyDir: emptyDir,
   161  		},
   162  	}, nil

   165  // addVolumeMountToContainers adds the Volume Mounts in containerNameToMountPaths to the containers for a given pvc and volumeName
   166  // containerNameToMountPaths is a map of a container name to an array of its Mount Paths.
   167  // To be moved to devfile/library.
   168  func addVolumeMountToContainers(containers []corev1.Container, initContainers []corev1.Container, volumeName string, containerNameToMountPaths map[string][]string) {
   169  
   170  	for containerName, mountPaths := range containerNameToMountPaths {
   171  		for i := range containers {
   172  			if containers[i].Name == containerName {
   173  				for _, mountPath := range mountPaths {
   174  					containers[i].VolumeMounts = append(containers[i].VolumeMounts, corev1.VolumeMount{
   175  						Name:      volumeName,
   176  						MountPath: mountPath,
   177  						SubPath:   "",
   178  					},
   179  					)
   180  				}
   181  			}
   182  		}
   183  		for i := range initContainers {
   184  			if strings.HasPrefix(initContainers[i].Name, containerName) {
   185  				for _, mountPath := range mountPaths {
   186  					initContainers[i].VolumeMounts = append(initContainers[i].VolumeMounts, corev1.VolumeMount{
   187  						Name:      volumeName,
   188  						MountPath: mountPath,
   189  						SubPath:   "",
   190  					},
   191  					)
   192  				}
   193  			}
   194  		}
   195  	}

   207  // HandleEphemeralStorage creates or deletes the ephemeral volume based on the preference setting
   208  func HandleEphemeralStorage(client kclient.ClientInterface, storageClient storage.Client, componentName string, isEphemeral bool) error {
   209  	selector := odolabels.Builder().WithComponentName(componentName).WithSourcePVC(storage.OdoSourceVolume).Selector()
   210  	pvcs, err := client.ListPVCs(selector)
   211  	if err != nil && !kerrors.IsNotFound(err) {
   212  		return err
   213  	}
   214  
   215  	if !isEphemeral {
   216  		if len(pvcs) == 0 {
   217  			err := storageClient.Create(storage.Storage{
   218  				ObjectMeta: metav1.ObjectMeta{
   219  					Name: storage.OdoSourceVolume,
   220  				},
   221  				Spec: storage.StorageSpec{
   222  					Size: storage.OdoSourceVolumeSize,
   223  				},
   224  			})
   225  
   226  			if err != nil {
   227  				return err
   228  			}
   229  		} else if len(pvcs) > 1 {
   230  			return fmt.Errorf("number of source volumes shouldn't be greater than 1")
   231  		}
   232  	} else {
   233  		if len(pvcs) > 0 {
   234  			for _, pvc := range pvcs {
   235  				err := client.DeletePVC(pvc.Name)
   236  				if err != nil {
   237  					return err
   238  				}
   239  			}
   240  		}
   241  	}
   242  	return nil
   243  }
