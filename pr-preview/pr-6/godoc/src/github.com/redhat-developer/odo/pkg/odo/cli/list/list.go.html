    72  // Complete ...
    73  func (lo *ListOptions) Complete(cmdline cmdline.Cmdline, args []string) (err error) {
    74  
    75  	// Check to see if KUBECONFIG exists, and if not, error the user that we would not be able to get cluster information
    76  	// Do this before anything else, or else we will just error out with the:
    77  	// invalid configuration: no configuration has been provided, try setting KUBERNETES_MASTER environment variable
    78  	// instead
    79  	if !dfutil.CheckKubeConfigExist() {
    80  		return errors.New("KUBECONFIG not found. Unable to retrieve cluster information. Please set your Kubernetes configuration via KUBECONFIG env variable or ~/.kube/config")
    81  	}
    82  
    83  	// Create the local context and initial Kubernetes client configuration
    84  	lo.Context, err = genericclioptions.New(genericclioptions.NewCreateParameters(cmdline))
    85  	if err != nil {
    86  		return err
    87  	}
    88  
    89  	// Check for the Devfile and then retrieve all information regarding the local Devfile
    90  	lo.devfilePath = location.DevfileLocation("")
    91  	if util.CheckPathExists(lo.devfilePath) {
    92  
    93  		// Set the project / namespace based on the devfile context
    94  		lo.project = lo.Context.GetProject()
    95  
    96  		// Parse the devfile
    97  		devObj, parseErr := devfile.ParseAndValidateFromFile(lo.devfilePath)
    98  		if parseErr != nil {
    99  			return parseErr
   100  		}
   101  
   102  		// Create a local component from the parse devfile
   103  		localComponent := api.ComponentAbstract{
   104  			Name:      devObj.Data.GetMetadata().Name,
   105  			ManagedBy: "",
   106  			RunningIn: []api.RunningMode{},
   107  			Type:      component.GetComponentTypeFromDevfileMetadata(devObj.Data.GetMetadata()),
   108  		}
   109  
   110  		lo.localComponent = localComponent
   111  
   112  	}
   113  
   114  	// If the context is "" (devfile.yaml not found..), we get the active one from KUBECONFIG.
   115  	if lo.project == "" {
   116  		lo.project = lo.clientset.KubernetesClient.GetCurrentNamespace()
   117  	}
   118  
   119  	// If the namespace flag has been passed, we will search there.
   120  	// if it hasn't, we will search from the default project / namespace.
   121  	if lo.namespaceFlag != "" {
   122  		lo.namespaceFilter = lo.namespaceFlag
   123  	} else {
   124  		lo.namespaceFilter = lo.project
   125  	}
   126  
   127  	return
   128  }

   154  func (lo *ListOptions) run(cts context.Context) (api.ResourcesList, error) {
   155  	listSpinner := log.Spinnerf("Listing components from namespace '%s'", lo.namespaceFilter)
   156  	defer listSpinner.End(false)
   157  
   158  	// Step 1.
   159  	// Retrieve all related components from the Kubernetes cluster, from the given namespace
   160  	devfileComponents, err := component.ListAllClusterComponents(lo.clientset.KubernetesClient, lo.namespaceFilter)
   161  	if err != nil {
   162  		return api.ResourcesList{}, err
   163  	}
   164  	listSpinner.End(true)
   165  
   166  	// Step 2.
   167  	// If we have a local component, let's add it to the list of Devfiles
   168  	// This checks lo.localComponent.Name. If it's empty, we didn't parse one in the Complete() function, so there is no local devfile.
   169  	// We will only append the local component to the devfile if it doesn't exist in the list.
   170  	componentInDevfile := ""
   171  	if lo.localComponent.Name != "" {
   172  		if !component.Contains(lo.localComponent, devfileComponents) {
   173  			devfileComponents = append(devfileComponents, lo.localComponent)
   174  		}
   175  		componentInDevfile = lo.localComponent.Name
   176  	}
   177  	return api.ResourcesList{
   178  		ComponentInDevfile: componentInDevfile,
   179  		Components:         devfileComponents,
   180  	}, nil
   181  }

   213  func humanReadableOutput(list api.ResourcesList) {
   214  	components := list.Components
   215  	if len(components) == 0 {
   216  		log.Error("There are no components deployed.")
   217  		return
   218  	}
   219  
   220  	t := ui.NewTable()
   221  
   222  	// Create the header and then sort accordingly
   223  	t.AppendHeader(table.Row{"NAME", "PROJECT TYPE", "RUNNING IN", "MANAGED"})
   224  	t.SortBy([]table.SortBy{
   225  		{Name: "MANAGED", Mode: table.Asc},
   226  		{Name: "NAME", Mode: table.Dsc},
   227  	})
   228  
   229  	// Go through each component and add it to the table
   230  	for _, comp := range components {
   231  
   232  		// Mark the name as yellow in the index to it's easier to see.
   233  		name := text.Colors{text.FgHiYellow}.Sprint(comp.Name)
   234  
   235  		// Get the managed by label
   236  		managedBy := comp.ManagedBy
   237  		if managedBy == "" {
   238  			managedBy = api.TypeUnknown
   239  		}
   240  
   241  		// Get the mode (dev or deploy)
   242  		mode := comp.RunningIn.String()
   243  
   244  		// Get the type of the component
   245  		componentType := comp.Type
   246  		if componentType == "" {
   247  			componentType = api.TypeUnknown
   248  		}
   249  
   250  		// If we find our local unpushed component, let's change the output appropriately.
   251  		if list.ComponentInDevfile == comp.Name {
   252  			name = fmt.Sprintf("* %s", name)
   253  
   254  			if comp.ManagedBy == "" {
   255  				managedBy = "odo"
   256  			}
   257  		}
   258  
   259  		// If we are managing that component, output it as blue (our logo colour) to indicate it's used by odo
   260  		if managedBy == "odo" {
   261  			managedBy = text.Colors{text.FgBlue}.Sprintf("odo (%s)", comp.ManagedByVersion)
   262  		} else if managedBy != "" && comp.ManagedByVersion != "" {
   263  			// this is done to maintain the color of the output
   264  			managedBy += fmt.Sprintf("(%s)", comp.ManagedByVersion)
   265  		}
   266  
   267  		t.AppendRow(table.Row{name, componentType, mode, managedBy})
   268  	}
   269  	t.Render()
   270  
   271  }
