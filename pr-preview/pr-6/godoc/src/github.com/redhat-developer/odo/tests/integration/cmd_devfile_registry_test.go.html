    35  	It("Should list all default registries", func() {
    36  		output := helper.Cmd("odo", "preference", "view").ShouldPass().Out()
    37  		helper.MatchAllInOutput(output, []string{"DefaultDevfileRegistry"})
    38  	})

    45  	It("Should list detailed information regarding nodejs", func() {
    46  		args := []string{"registry", "--details", "--devfile", "nodejs", "--devfile-registry", "DefaultDevfileRegistry"}
    47  
    48  		By("using human readable output", func() {
    49  			output := helper.Cmd("odo", args...).ShouldPass().Out()
    50  			helper.MatchAllInOutput(output, []string{"nodejs-starter", "javascript", "Node.js Runtime", "Dev: Y"})
    51  		})
    52  		By("using JSON output", func() {
    53  			args = append(args, "-o", "json")
    54  			res := helper.Cmd("odo", args...).ShouldPass()
    55  			stdout, stderr := res.Out(), res.Err()
    56  			Expect(stderr).To(BeEmpty())
    57  			Expect(helper.IsJSON(stdout)).To(BeTrue())
    58  			helper.JsonPathContentIs(stdout, "0.name", "nodejs")
    59  			helper.JsonPathContentContain(stdout, "0.displayName", "Node")
    60  			helper.JsonPathContentContain(stdout, "0.description", "Node")
    61  			helper.JsonPathContentContain(stdout, "0.language", "javascript")
    62  			helper.JsonPathContentContain(stdout, "0.projectType", "nodejs")
    63  			helper.JsonPathContentContain(stdout, "0.starterProjects.0", "nodejs-starter")
    64  			helper.JsonPathContentContain(stdout, "0.devfileData.devfile.metadata.name", "nodejs")
    65  			helper.JsonPathContentContain(stdout, "0.devfileData.supportedOdoFeatures.dev", "true")
    66  		})
    67  	})
    68

    74  		})
