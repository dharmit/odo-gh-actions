    62  func (kubectl KubectlRunner) CheckCmdOpInRemoteDevfilePod(podName string, containerName string, prjName string, cmd []string, checkOp func(cmdOp string, err error) bool) bool {
    63  	var execOptions []string
    64  	execOptions = []string{"exec", podName, "--namespace", prjName, "--"}
    65  	if containerName != "" {
    66  		execOptions = []string{"exec", podName, "-c", containerName, "--namespace", prjName, "--"}
    67  	}
    68  	args := append(execOptions, cmd...)
    69  	session := CmdRunner(kubectl.path, args...)
    70  	stdOut := string(session.Wait().Out.Contents())
    71  	stdErr := string(session.Wait().Err.Contents())
    72  	if stdErr != "" && session.ExitCode() != 0 {
    73  		return checkOp(stdOut, fmt.Errorf("cmd %s failed with error %s on pod %s", cmd, stdErr, podName))
    74  	}
    75  	return checkOp(stdOut, nil)
    76  }

   289  // WaitForRunnerCmdOut runs "kubectl" command until it gets
   290  // the expected output.
   291  // It accepts 4 arguments
   292  // args (arguments to the program)
   293  // timeout (the time to wait for the output)
   294  // errOnFail (flag to set if test should fail if command fails)
   295  // check (function with output check logic)
   296  // It times out if the command doesn't fetch the
   297  // expected output  within the timeout period.
   298  func (kubectl KubectlRunner) WaitForRunnerCmdOut(args []string, timeout int, errOnFail bool, check func(output string) bool, includeStdErr ...bool) bool {
   299  	pingTimeout := time.After(time.Duration(timeout) * time.Minute)
   300  	// this is a test package so time.Tick() is acceptable
   301  	// nolint
   302  	tick := time.Tick(time.Second)
   303  	for {
   304  		select {
   305  		case <-pingTimeout:
   306  			Fail(fmt.Sprintf("Timeout after %v minutes", timeout))
   307  
   308  		case <-tick:
   309  			session := CmdRunner(kubectl.path, args...)
   310  			if errOnFail {
   311  				Eventually(session).Should(gexec.Exit(0), runningCmd(session.Command))
   312  			} else {
   313  				Eventually(session).Should(gexec.Exit(), runningCmd(session.Command))
   314  			}
   315  			session.Wait()
   316  			output := string(session.Out.Contents())
   317  
   318  			if len(includeStdErr) > 0 && includeStdErr[0] {
   319  				output += "\n"
   320  				output += string(session.Err.Contents())
   321  			}
   322  			if check(strings.TrimSpace(output)) {
   323  				return true
   324  			}
   325  		}
   326  	}
   327  }
