    29  	checkIfDevEnvIsUp := func(url, assertString string) {
    30  		Eventually(func() string {
    31  			resp, err := http.Get(fmt.Sprintf("http://%s", url))
    32  			Expect(err).ToNot(HaveOccurred())
    33  			defer resp.Body.Close()
    34  
    35  			body, _ := io.ReadAll(resp.Body)
    36  			return string(body)
    37  		}, 120*time.Second, 15*time.Second).Should(Equal(assertString))

    40  	Context("starting with empty Directory", func() {

    47  		It("should verify developer workflow from empty Directory", func() {
    48  			deploymentName := "my-component"
    49  			serviceName := "my-cs"

    51  			getSVCArgs := []string{"get", "svc", "-n", commonVar.Project}
    52  
    53  			command := []string{"odo", "init"}
    54  			_, err := helper.RunInteractive(command, nil, func(ctx helper.InteractiveContext) {
    55

    57  				helper.SendLine(ctx, "javascript")
    58

    60  				helper.SendLine(ctx, "Node.js\n")
    61

    63  				helper.SendLine(ctx, "nodejs-starter\n")
    64  
    65  				helper.ExpectString(ctx, "Enter component name")
    66  				helper.SendLine(ctx, componentName)
    67  
    68  				helper.ExpectString(ctx, "Your new component '"+componentName+"' is ready in the current directory")
    69  
    70  			})
    71  			Expect(err).To(BeNil())
    72  			Expect(helper.ListFilesInDir(commonVar.Context)).To(ContainElement("devfile.yaml"))
    73  			Expect(helper.ListFilesInDir(commonVar.Context)).To(ContainElement("server.js"))
    74  
    75  			// "execute odo dev and add changes to application"
    76  			var devSession helper.DevSession
    77  			var ports map[string]string
    78  
    79  			devSession, _, _, ports, err = helper.StartDevMode(nil)
    80  			helper.ReplaceString(filepath.Join(commonVar.Context, "server.js"), "from Node.js", "from updated Node.js")
    81  			Expect(err).ToNot(HaveOccurred())
    82  			_, _, _, err = devSession.WaitSync()
    83  			Expect(err).ToNot(HaveOccurred())
    84  			// "should update the changes"
    85  			checkIfDevEnvIsUp(ports["3000"], "Hello from updated Node.js Starter Application!")
    86  
    87  			// "changes are made to the applications"

    89  			_, _, _, err = devSession.WaitSync()
    90  			Expect(err).ToNot(HaveOccurred())
    91  			// "should deploy new changes"
    92  			checkIfDevEnvIsUp(ports["3000"], "Hello from Node.js app v2 Starter Application!")
    93  
    94  			// "running odo list"
    95  			stdout := helper.Cmd("odo", "list").ShouldPass().Out()
    96  			helper.MatchAllInOutput(stdout, []string{componentName, "nodejs", "Dev"})
    97  
    98  			// "exit dev mode and run odo deploy"
    99  			devSession.Stop()
   100  			devSession.WaitEnd()
   101  
   102  			// all resources should be deleted from the namespace

   116

   121  			// start dev mode again

   134  			helper.MatchAllInOutput(stdout, []string{componentName, "nodejs", "Dev", "Deploy"})
