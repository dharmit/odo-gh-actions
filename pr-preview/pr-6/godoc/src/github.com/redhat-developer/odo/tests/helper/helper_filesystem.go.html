    26  // DeleteDir deletes the specified path; due to Windows behaviour (for example https://github.com/redhat-developer/odo/issues/3371)
    27  // where Windows temporarily holds a lock on files and folders, we keep trying to delete until the operation passes (or it expires)
    28  func DeleteDir(dir string) {
    29  	attempts := 0
    30  
    31  	errorReportedAtLeastOnce := false
    32  
    33  	err := RunWithExponentialBackoff(func() error {
    34  		attempts++
    35  
    36  		fmt.Fprintf(GinkgoWriter, "Deleting dir: %s\n", dir)
    37  		err := os.RemoveAll(dir)
    38  		if err == nil {
    39  			return nil
    40  		}
    41  
    42  		errorReportedAtLeastOnce = true
    43  		fmt.Fprintf(GinkgoWriter, "Unable to delete %s on attempt #%d, trying again...\n", dir, attempts)
    44  
    45  		return err
    46  	}, 16, 2*time.Minute)
    47  	Expect(err).NotTo(HaveOccurred())
    48  
    49  	if errorReportedAtLeastOnce {
    50  		fmt.Fprintf(GinkgoWriter, "Successfully deleted %s after #%d attempts\n", dir, attempts)
    51  	}
    52  }

    54  // RunWithExponentialBackoff keeps trying to run 'fxn' until it no longer returns an error; if the function never succeeded,
    55  // then the most recent error is returned.
    56  func RunWithExponentialBackoff(fxn func() error, maxDelayInSeconds int, expireDuration time.Duration) error {
    57  	expireTime := time.Now().Add(expireDuration)
    58  	delayInSeconds := 1
    59  
    60  	var err error
    61  
    62  	for {
    63  
    64  		err = fxn()
    65  
    66  		if err == nil || time.Now().After(expireTime) {
    67  			break
    68  		}
    69  
    70  		delayInSeconds *= 2 // exponential backoff
    71  		if delayInSeconds > maxDelayInSeconds {
    72  			delayInSeconds = maxDelayInSeconds
    73  		}
    74  		time.Sleep(time.Duration(delayInSeconds) * time.Second)
    75  
    76  	}
    77  	return err
    78  
    79  }

   188  // copyDir copy one directory to the other
   189  // this function is called recursively info should start as os.Stat(src)
   190  func copyDir(src string, dst string, info os.FileInfo) error {
   191  
   192  	if info.IsDir() {
   193  		files, err := ioutil.ReadDir(src)
   194  		if err != nil {
   195  			return err
   196  		}
   197  
   198  		for _, file := range files {
   199  			dsrt := filepath.Join(src, file.Name())
   200  			ddst := filepath.Join(dst, file.Name())
   201  			if err := copyDir(dsrt, ddst, file); err != nil {
   202  				return err
   203  			}
   204  		}
   205  		return nil
   206  	}
   207  
   208  	if err := os.MkdirAll(filepath.Dir(dst), os.ModePerm); err != nil {
   209  		return err
   210  	}
   211  
   212  	return dfutil.CopyFile(src, dst, info)
   213  }
