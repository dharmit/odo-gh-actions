    61  func HumanReadableOutput(preferenceList preference.PreferenceList, registryList *[]preference.Registry) {
    62  	preferenceT := ui.NewTable()
    63  	preferenceT.AppendHeader(table.Row{"PARAMETER", "VALUE"})
    64  	preferenceT.SortBy([]table.SortBy{{Name: "PARAMETER", Mode: table.Asc}})
    65  	for _, pref := range preferenceList.Items {
    66  		value := showBlankIfNil(pref.Value)
    67  		if reflect.DeepEqual(value, pref.Default) {
    68  			value = fmt.Sprintf("%v (default)", value)
    69  		}
    70  		preferenceT.AppendRow(table.Row{pref.Name, value})
    71  	}
    72  	registryT := ui.NewTable()
    73  	registryT.AppendHeader(table.Row{"NAME", "URL", "SECURE"})
    74  
    75  	var regList []preference.Registry
    76  	if registryList != nil {
    77  		regList = *registryList
    78  	}
    79  	// Loop backwards here to ensure the registry display order is correct (display latest newly added registry firstly)
    80  	for i := len(regList) - 1; i >= 0; i-- {
    81  		registry := regList[i]
    82  		secure := "No"
    83  		if registry.Secure {
    84  			secure = "Yes"
    85  		}
    86  		registryT.AppendRow(table.Row{registry.Name, registry.URL, secure})
    87  	}
    88  
    89  	log.Info("Preference parameters:")
    90  	preferenceT.Render()
    91  	log.Info("\nDevfile registries:")
    92  	if registryList == nil || len(*registryList) == 0 {
    93  		log.Warning("No devfile registries added to the configuration. Refer to `odo preference add registry -h` to add one")
    94  		return
    95  	}
    96  	registryT.Render()
    97  }
    98  func showBlankIfNil(intf interface{}) interface{} {
    99  	imm := reflect.ValueOf(intf)
   100  
   101  	// if the value is nil then we should return a blank string
   102  	if imm.IsNil() {
   103  		return ""
   104  	}
   105  
   106  	// if its a pointer then we should de-ref it because we cant de-ref an interface{}
   107  	if imm.Kind() == reflect.Ptr {
   108  		return imm.Elem().Interface()
   109  	}
   110  
   111  	return intf
   112  }
