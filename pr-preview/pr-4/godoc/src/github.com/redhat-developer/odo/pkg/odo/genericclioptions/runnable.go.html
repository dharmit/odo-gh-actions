    66  func GenericRun(o Runnable, cmd *cobra.Command, args []string) {
    67  	var err error
    68  	startTime := time.Now()
    69  	cfg, _ := preference.NewClient()
    70  	disableTelemetry, _ := strconv.ParseBool(os.Getenv(segment.DisableTelemetryEnv))
    71  	debugTelemetry := segment.GetDebugTelemetryFile()
    72  
    73  	// Prompt the user to consent for telemetry if a value is not set already
    74  	// Skip prompting if the preference command is called
    75  	// This prompt has been placed here so that it does not prompt the user when they call --help
    76  	if !cfg.IsSet(preference.ConsentTelemetrySetting) && cmd.Parent().Name() != "preference" {
    77  		if !segment.RunningInTerminal() {
    78  			klog.V(4).Infof("Skipping telemetry question because there is no terminal (tty)\n")
    79  		} else if disableTelemetry {
    80  			klog.V(4).Infof("Skipping telemetry question due to %s=%t\n", segment.DisableTelemetryEnv, disableTelemetry)
    81  		} else {
    82  			var consentTelemetry bool
    83  			prompt := &survey.Confirm{Message: "Help odo improve by allowing it to collect usage data. Read about our privacy statement: https://developers.redhat.com/article/tool-data-collection. You can change your preference later by changing the ConsentTelemetry preference.", Default: true}
    84  			err = survey.AskOne(prompt, &consentTelemetry, nil)
    85  			ui.HandleError(err)
    86  			if err == nil {
    87  				if err1 := cfg.SetConfiguration(preference.ConsentTelemetrySetting, strconv.FormatBool(consentTelemetry)); err1 != nil {
    88  					klog.V(4).Info(err1.Error())
    89  				}
    90  			}
    91  		}
    92  	}
    93  	if len(debugTelemetry) > 0 {
    94  		klog.V(4).Infof("WARNING: debug telemetry, if enabled, will be logged in %s", debugTelemetry)
    95  	}
    96  
    97  	// set value for telemetry status in context so that we do not need to call IsTelemetryEnabled every time to check its status
    98  	scontext.SetTelemetryStatus(cmd.Context(), segment.IsTelemetryEnabled(cfg))
    99  
   100  	// Send data to telemetry in case of user interrupt
   101  	captureSignals := []os.Signal{syscall.SIGHUP, syscall.SIGTERM, syscall.SIGQUIT, os.Interrupt}
   102  	go commonutil.StartSignalWatcher(captureSignals, func(receivedSignal os.Signal) {
   103  		err = fmt.Errorf("user interrupted the command execution: %w", terminal.InterruptErr)
   104  		if handler, ok := o.(SignalHandler); ok {
   105  			err = handler.HandleSignal()
   106  			if err != nil {
   107  				log.Errorf("error handling interrupt signal : %v", err)
   108  			}
   109  		}
   110  		scontext.SetSignal(cmd.Context(), receivedSignal)
   111  		startTelemetry(cmd, err, startTime)
   112  	})
   113  
   114  	// CheckMachineReadableOutput
   115  	// fixes / checks all related machine readable output functions
   116  	util.LogErrorAndExit(CheckMachineReadableOutputCommand(cmd), "")
   117  
   118  	deps, err := clientset.Fetch(cmd)
   119  	if err != nil {
   120  		util.LogErrorAndExit(err, "")
   121  	}
   122  	o.SetClientset(deps)
   123  
   124  	cmdLineObj := cmdline.NewCobra(cmd)
   125  	// Run completion, validation and run.
   126  	// Only upload data to segment for completion and validation if a non-nil error is returned.
   127  	err = o.Complete(cmdLineObj, args)
   128  	if err != nil {
   129  		startTelemetry(cmd, err, startTime)
   130  	}
   131  	util.LogErrorAndExit(err, "")
   132  
   133  	err = o.Validate()
   134  	if err != nil {
   135  		startTelemetry(cmd, err, startTime)
   136  	}
   137  	util.LogErrorAndExit(err, "")
   138  
   139  	if jsonOutputter, ok := o.(JsonOutputter); ok && log.IsJSON() {
   140  		var out interface{}
   141  		out, err = jsonOutputter.RunForJsonOutput(cmdLineObj.Context())
   142  		if err == nil {
   143  			machineoutput.OutputSuccess(out)
   144  		}
   145  	} else {
   146  		err = o.Run(cmdLineObj.Context())
   147  	}
   148  	startTelemetry(cmd, err, startTime)
   149  	util.LogError(err, "")
   150  	if cleanuper, ok := o.(Cleanuper); ok {
   151  		cleanuper.Cleanup(err)
   152  	}
   153  	if err != nil {
   154  		os.Exit(1)
   155  	}
   156  }

   158  // startTelemetry uploads the data to segment if user has consented to usage data collection and the command is not telemetry
   159  // TODO: move this function to a more suitable place, preferably pkg/segment
   160  func startTelemetry(cmd *cobra.Command, err error, startTime time.Time) {
   161  	if scontext.GetTelemetryStatus(cmd.Context()) && !strings.Contains(cmd.CommandPath(), "telemetry") {
   162  		uploadData := &segment.TelemetryData{
   163  			Event: cmd.CommandPath(),
   164  			Properties: segment.TelemetryProperties{
   165  				Duration:      time.Since(startTime).Milliseconds(),
   166  				Success:       err == nil,
   167  				Tty:           segment.RunningInTerminal(),
   168  				Version:       fmt.Sprintf("odo %v (%v)", version.VERSION, version.GITCOMMIT),
   169  				CmdProperties: scontext.GetContextProperties(cmd.Context()),
   170  			},
   171  		}
   172  		if err != nil {
   173  			uploadData.Properties.Error = segment.SetError(err)
   174  			uploadData.Properties.ErrorType = segment.ErrorType(err)
   175  		}
   176  		data, err1 := json.Marshal(uploadData)
   177  		if err1 != nil {
   178  			klog.V(4).Infof("Failed to marshall telemetry data. %q", err1.Error())
   179  		}
   180  		command := exec.Command(os.Args[0], "telemetry", string(data))
   181  		if err1 = command.Start(); err1 != nil {
   182  			klog.V(4).Infof("Failed to start the telemetry process. Error: %q", err1.Error())
   183  			return
   184  		}
   185  		if err1 = command.Process.Release(); err1 != nil {
   186  			klog.V(4).Infof("Failed to release the process. %q", err1.Error())
   187  			return
   188  		}
   189  	}
   190  }

   194  func CheckMachineReadableOutputCommand(cmd *cobra.Command) error {
   195  
   196  	// Get the needed values
   197  	outputFlag := pflag.Lookup("o")
   198  	hasFlagChanged := outputFlag != nil && outputFlag.Changed
   199  	machineOutput := cmd.Annotations["machineoutput"]
   200  
   201  	// Check the valid output
   202  	if hasFlagChanged && outputFlag.Value.String() != "json" {
   203  		//revive:disable:error-strings This is a top-level error message displayed as is to the end user
   204  		return errors.New("Please input a valid output format for -o, available format: json")
   205  		//revive:enable:error-strings
   206  	}
   207  
   208  	// Check that if -o json has been passed, that the command actually USES json.. if not, error out.
   209  	if hasFlagChanged && outputFlag.Value.String() == "json" && machineOutput == "" {
   210  
   211  		// By default we "disable" logging, so activate it so that the below error can be shown.
   212  		_ = flag.Set("o", "")
   213  
   214  		// Return the error
   215  		//revive:disable:error-strings This is a top-level error message displayed as is to the end user
   216  		return errors.New("Machine readable output is not yet implemented for this command")
   217  		//revive:enable:error-strings
   218  	}
   219  
   220  	// Before running anything, we will make sure that no verbose output is made
   221  	// This is a HACK to manually override `-v 4` to `-v 0` (in which we have no klog.V(0) in our code...
   222  	// in order to have NO verbose output when combining both `-o json` and `-v 4` so json output
   223  	// is not malformed / mixed in with normal logging
   224  	if log.IsJSON() {
   225  		_ = flag.Set("v", "0")
   226  	} else {
   227  		// Override the logging level by the value (if set) by the ODO_LOG_LEVEL env
   228  		// The "-v" flag set on command line will take precedence over ODO_LOG_LEVEL env
   229  		v := flag.CommandLine.Lookup("v").Value.String()
   230  		if level, ok := os.LookupEnv("ODO_LOG_LEVEL"); ok && v == "0" {
   231  			_ = flag.CommandLine.Set("v", level)
   232  		}
   233  	}
   234  	return nil
   235  }
