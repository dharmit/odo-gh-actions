   222  // populateWorkingDir populates the working directory with .odo and devfile.yaml, and returns envinfo
   223  func populateWorkingDir(fs filesystem.Filesystem, workingDir, compName, projectName, appName string) *envinfo.EnvSpecificInfo {
   224  	_ = fs.MkdirAll(filepath.Join(workingDir, ".odo", "env"), 0755)
   225  	_ = fs.WriteFile(filepath.Join(workingDir, ".odo", "env", "env.yaml"), []byte{}, 0644)
   226  	env, err := envinfo.NewEnvSpecificInfo(workingDir)
   227  	if err != nil {
   228  		return env
   229  	}
   230  	_ = env.SetComponentSettings(envinfo.ComponentSettings{
   231  		Name:    compName,
   232  		Project: projectName,
   233  		AppName: appName,
   234  	})
   235  	devfileObj := testingutil.GetTestDevfileObjFromFile("devfile-deploy.yaml")
   236  	devfileYAML, err := yaml.Marshal(devfileObj.Data)
   237  	if err != nil {
   238  		return env
   239  	}
   240  	_ = fs.WriteFile(filepath.Join(workingDir, "devfile.yaml"), devfileYAML, 0644)
   241  	env.SetDevfileObj(devfileObj)
   242  	return env
   243  }

   258  // prepareContext prepares the mock genericclioptions.Context and returns it
   259  func prepareContext(ctrl *gomock.Controller, kubeClient kclient.ClientInterface, info *envinfo.EnvSpecificInfo, workingDir string) *genericclioptions.Context {
   260  	cmdline := cmdline.NewMockCmdline(ctrl)
   261  	cmdline.EXPECT().GetWorkingDirectory().Return(workingDir, nil).AnyTimes()
   262  	cmdline.EXPECT().CheckIfConfigurationNeeded().Return(true, nil).AnyTimes()
   263  	cmdline.EXPECT().FlagValueIfSet("project").Return("").AnyTimes()
   264  	cmdline.EXPECT().FlagValueIfSet("app").Return("").AnyTimes()
   265  	cmdline.EXPECT().FlagValueIfSet("component").Return("").AnyTimes()
   266  	cmdline.EXPECT().FlagValueIfSet("o").Return("").AnyTimes()
   267  	cmdline.EXPECT().GetKubeClient().Return(kubeClient, nil).AnyTimes()
   268  	createParameters := genericclioptions.NewCreateParameters(cmdline).NeedDevfile(workingDir)
   269  	context, err := genericclioptions.New(createParameters)
   270  	if err != nil {
   271  		return nil
   272  	}
   273  	context.EnvSpecificInfo = info
   274  	return context
   275  }
