     9  // GetKubernetesComponentsToPush returns the list of Kubernetes components to push,
    10  // by getting the list of Kubernetes components and removing the ones
    11  // referenced from a command in the devfile
    12  func GetKubernetesComponentsToPush(devfileObj parser.DevfileObj) ([]devfilev1.Component, error) {
    13  	k8sComponents, err := devfileObj.Data.GetComponents(parsercommon.DevfileOptions{
    14  		ComponentOptions: parsercommon.ComponentOptions{ComponentType: devfilev1.KubernetesComponentType},
    15  	})
    16  	if err != nil {
    17  		return nil, err
    18  	}
    19  
    20  	componentsMap := map[string]devfilev1.Component{}
    21  	for _, component := range k8sComponents {
    22  		componentsMap[component.Name] = component
    23  	}
    24  
    25  	commands, err := devfileObj.Data.GetCommands(parsercommon.DevfileOptions{})
    26  	if err != nil {
    27  		return nil, err
    28  	}
    29  
    30  	for _, command := range commands {
    31  		componentName := ""
    32  		if command.Exec != nil {
    33  			componentName = command.Exec.Component
    34  		} else if command.Apply != nil {
    35  			componentName = command.Apply.Component
    36  		}
    37  		if componentName == "" {
    38  			continue
    39  		}
    40  		delete(componentsMap, componentName)
    41  	}
    42  
    43  	k8sComponents = make([]devfilev1.Component, len(componentsMap))
    44  	i := 0
    45  	for _, v := range componentsMap {
    46  		k8sComponents[i] = v
    47  		i++
    48  	}
    49  
    50  	return k8sComponents, err
    51  }
