    32  func (o *DockerCompatibleBackend) Build(fs filesystem.Filesystem, image *devfile.ImageComponent, devfilePath string) error {
    33  
    34  	dockerfile, isTemp, err := resolveAndDownloadDockerfile(fs, image.Dockerfile.Uri)
    35  	if isTemp {
    36  		defer func(path string) {
    37  			if e := fs.Remove(path); e != nil {
    38  				klog.V(3).Infof("could not remove temporary Dockerfile at path %q: %v", path, err)
    39  			}
    40  		}(dockerfile)
    41  	}
    42  	if err != nil {
    43  		return err
    44  	}
    45  
    46  	// We use a "No Spin" since we are outputting to stdout / stderr
    47  	buildSpinner := log.SpinnerNoSpin("Building image locally")
    48  	defer buildSpinner.End(false)
    49  
    50  	err = os.Setenv("PROJECTS_ROOT", devfilePath)
    51  	if err != nil {
    52  		return err
    53  	}
    54  
    55  	err = os.Setenv("PROJECT_SOURCE", devfilePath)
    56  	if err != nil {
    57  		return err
    58  	}
    59  
    60  	shellCmd := getShellCommand(o.name, image, devfilePath, dockerfile)
    61  	klog.V(4).Infof("Running command: %v", shellCmd)
    62  	for i, cmd := range shellCmd {
    63  		shellCmd[i] = os.ExpandEnv(cmd)
    64  	}
    65  	cmd := exec.Command(shellCmd[0], shellCmd[1:]...)
    66  	cmdEnv := []string{
    67  		"PROJECTS_ROOT=" + devfilePath,
    68  		"PROJECT_SOURCE=" + devfilePath,
    69  	}
    70  	cmd.Env = append(os.Environ(), cmdEnv...)
    71  	cmd.Stdout = log.GetStdout()
    72  	cmd.Stderr = log.GetStderr()
    73  
    74  	// Set all output as italic when doing a push, then return to normal at the end
    75  	color.Set(color.Italic)
    76  	defer color.Unset()
    77  	err = cmd.Run()
    78  	if err != nil {
    79  		return fmt.Errorf("error running %s command: %w", o.name, err)
    80  	}
    81  
    82  	buildSpinner.End(true)
    83  	return nil

    86  // resolveAndDownloadDockerfile resolves and downloads (if needed) the specified Dockerfile URI.
    87  // For now, it only supports resolving HTTP(S) URIs, in which case it downloads the remote file
    88  // to a temporary file. The path to that temporary file is then returned.
    89  //
    90  // In all other cases, the specified URI path is returned as is.
    91  // This means that non-HTTP(S) URIs will *not* get resolved, but will be returned as is.
    92  //
    93  // In addition to the path, a boolean and a potential error are returned. The boolean indicates whether
    94  // the returned path is a temporary one; in such case, it is the caller's responsibility to delete this file
    95  // once it is done working with it.
    96  func resolveAndDownloadDockerfile(fs filesystem.Filesystem, uri string) (string, bool, error) {
    97  	uriLower := strings.ToLower(uri)
    98  	if strings.HasPrefix(uriLower, "http://") || strings.HasPrefix(uriLower, "https://") {
    99  		s := log.Spinner("Downloading Dockerfile")
   100  		defer s.End(false)
   101  		tempFile, err := fs.TempFile("", "odo_*.dockerfile")
   102  		if err != nil {
   103  			return "", false, err
   104  		}
   105  		dockerfile := tempFile.Name()
   106  		err = dfutil.DownloadFile(dfutil.DownloadParams{
   107  			Request: dfutil.HTTPRequestParams{
   108  				URL: uri,
   109  			},
   110  			Filepath: dockerfile,
   111  		})
   112  		s.End(err == nil)
   113  		return dockerfile, true, err
   114  	}
   115  	return uri, false, nil
   116  }

   118  //getShellCommand creates the docker compatible build command from detected backend,
   119  //container image and devfile path
   120  func getShellCommand(cmdName string, image *devfile.ImageComponent, devfilePath string, dockerfilePath string) []string {
   121  	var shellCmd []string
   122  	imageName := image.ImageName
   123  	dockerfile := dockerfilePath
   124  	if !filepath.IsAbs(dockerfile) {
   125  		dockerfile = filepath.Join(devfilePath, dockerfilePath)
   126  	}
   127  	buildpath := image.Dockerfile.BuildContext
   128  	if buildpath == "" {
   129  		buildpath = devfilePath
   130  	}
   131  	args := image.Dockerfile.Args
   132  	shellCmd = []string{
   133  		cmdName,
   134  		"build",
   135  		"-t",
   136  		imageName,
   137  		"-f",
   138  		dockerfile,
   139  		buildpath,
   140  	}
   141  	if len(args) > 0 {
   142  		shellCmd = append(shellCmd, args...)
   143  	}
   144  	return shellCmd
   145  }
