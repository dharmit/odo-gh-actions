    17  func (ei *EnvInfo) ListStorage() ([]localConfigProvider.LocalStorage, error) {
    18  	var storageList []localConfigProvider.LocalStorage
    19  
    20  	volumeMap := make(map[string]devfilev1.Volume)
    21  	components, err := ei.devfileObj.Data.GetComponents(common.DevfileOptions{})
    22  	if err != nil {
    23  		return storageList, err
    24  	}
    25  
    26  	for _, component := range components {
    27  		if component.Volume == nil {
    28  			continue
    29  		}
    30  		if component.Volume.Size == "" {
    31  			component.Volume.Size = DefaultVolumeSize
    32  		}
    33  		volumeMap[component.Name] = component.Volume.Volume
    34  	}
    35  
    36  	for _, component := range components {
    37  		if component.Container == nil {
    38  			continue
    39  		}
    40  		for _, volumeMount := range component.Container.VolumeMounts {
    41  			vol, ok := volumeMap[volumeMount.Name]
    42  			if ok {
    43  				storageList = append(storageList, localConfigProvider.LocalStorage{
    44  					Name:      volumeMount.Name,
    45  					Size:      vol.Size,
    46  					Ephemeral: vol.Ephemeral,
    47  					Path:      generator.GetVolumeMountPath(volumeMount),
    48  					Container: component.Name,
    49  				})
    50  			}
    51  		}
    52  	}
    53  
    54  	return storageList, nil
    55  }
