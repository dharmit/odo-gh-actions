    57  // GenerateSelfSignedCertificate creates a self-signed SSl certificate
    58  func GenerateSelfSignedCertificate(host string) (SelfSignedCertificate, error) {
    59  
    60  	privateKey, err := rsa.GenerateKey(rand.Reader, 2048)
    61  	if err != nil {
    62  		return SelfSignedCertificate{}, fmt.Errorf("unable to generate rsa key: %w", err)
    63  	}
    64  	template := x509.Certificate{
    65  		SerialNumber: big.NewInt(time.Now().Unix()),
    66  		Subject: pkix.Name{
    67  			CommonName:   "Odo self-signed certificate",
    68  			Organization: []string{"Odo"},
    69  		},
    70  		NotBefore:             time.Now(),
    71  		NotAfter:              time.Now().Add(time.Hour * 24 * 365 * 10),
    72  		KeyUsage:              x509.KeyUsageCertSign | x509.KeyUsageKeyEncipherment | x509.KeyUsageDigitalSignature,
    73  		BasicConstraintsValid: true,
    74  		DNSNames:              []string{"*." + host},
    75  	}
    76  
    77  	certificateDerEncoding, err := x509.CreateCertificate(rand.Reader, &template, &template, &privateKey.PublicKey, privateKey)
    78  	if err != nil {
    79  		return SelfSignedCertificate{}, fmt.Errorf("unable to create certificate: %w", err)
    80  	}
    81  	out := &bytes.Buffer{}
    82  	err = pem.Encode(out, &pem.Block{Type: "CERTIFICATE", Bytes: certificateDerEncoding})
    83  	if err != nil {
    84  		return SelfSignedCertificate{}, fmt.Errorf("unable to encode certificate: %w", err)
    85  	}
    86  	certPemEncode := out.String()
    87  	certPemByteArr := []byte(certPemEncode)
    88  
    89  	tlsPrivKeyEncoding := x509.MarshalPKCS1PrivateKey(privateKey)
    90  	err = pem.Encode(out, &pem.Block{Type: "RSA PRIVATE KEY", Bytes: tlsPrivKeyEncoding})
    91  	if err != nil {
    92  		return SelfSignedCertificate{}, fmt.Errorf("unable to encode rsa private key: %w", err)
    93  	}
    94  	keyPemEncode := out.String()
    95  	keyPemByteArr := []byte(keyPemEncode)
    96  
    97  	return SelfSignedCertificate{CertPem: certPemByteArr, KeyPem: keyPemByteArr}, nil
    98  }

   144  // CreateSecrets creates a secret for each port, containing the host and port of the component
   145  // This is done so other components can later inject the secret into the environment
   146  // and have the "coordinates" to communicate with this component
   147  func (c *Client) CreateSecrets(componentName string, commonObjectMeta metav1.ObjectMeta, svc *corev1.Service, ownerReference metav1.OwnerReference) error {
   148  	originalName := commonObjectMeta.Name
   149  	for _, svcPort := range svc.Spec.Ports {
   150  		portAsString := fmt.Sprintf("%v", svcPort.Port)
   151  
   152  		// we need to create multiple secrets, so each one has to contain the port in it's name
   153  		// so we change the name of each secret by adding the port number
   154  		commonObjectMeta.Name = fmt.Sprintf("%v-%v", originalName, portAsString)
   155  
   156  		// we also add the port as an annotation to the secret
   157  		// this comes in handy when we need to "query" for the appropriate secret
   158  		// of a component based on the port
   159  		commonObjectMeta.Annotations[ComponentPortAnnotationName] = portAsString
   160  
   161  		err := c.CreateSecret(
   162  			commonObjectMeta,
   163  			map[string]string{
   164  				secretKeyName(componentName, "host"): svc.Name,
   165  				secretKeyName(componentName, "port"): portAsString,
   166  			},
   167  			ownerReference)
   168  
   169  		if err != nil {
   170  			return fmt.Errorf("unable to create Secret for %s: %w", commonObjectMeta.Name, err)
   171  		}
   172  	}
   173  
   174  	// restore the original values of the fields we changed
   175  	commonObjectMeta.Name = originalName
   176  	delete(commonObjectMeta.Annotations, ComponentPortAnnotationName)
   177  
   178  	return nil
   179  }

   198  // WaitAndGetSecret blocks and waits until the secret is available
   199  func (c *Client) WaitAndGetSecret(name string, namespace string) (*corev1.Secret, error) {
   200  	klog.V(3).Infof("Waiting for secret %s to become available", name)
   201  
   202  	w, err := c.KubeClient.CoreV1().Secrets(namespace).Watch(context.TODO(), metav1.ListOptions{
   203  		FieldSelector: fields.Set{"metadata.name": name}.AsSelector().String(),
   204  	})
   205  	if err != nil {
   206  		return nil, fmt.Errorf("unable to watch secret: %w", err)
   207  	}
   208  	defer w.Stop()
   209  	for {
   210  		val, ok := <-w.ResultChan()
   211  		if !ok {
   212  			break
   213  		}
   214  		if e, ok := val.Object.(*corev1.Secret); ok {
   215  			klog.V(3).Infof("Secret %s now exists", e.Name)
   216  			return e, nil
   217  		}
   218  	}
   219  	return nil, fmt.Errorf("unknown error while waiting for secret '%s'", name)
   220  }
