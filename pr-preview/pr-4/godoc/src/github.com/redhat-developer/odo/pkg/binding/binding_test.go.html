   428  func getServiceBindingInlinedContent(ns string, bindAsFiles bool, namingStrategy string) string {
   429  	h := fmt.Sprintf(`apiVersion: binding.operators.coreos.com/v1alpha1
   430  kind: ServiceBinding
   431  metadata:
   432    creationTimestamp: null
   433    name: my-nodejs-app-cluster-sample
   434  spec:
   435    application:
   436      group: apps
   437      kind: Deployment
   438      name: my-nodejs-app-app
   439      version: v1
   440    bindAsFiles: %v
   441    detectBindingResources: true`, bindAsFiles)
   442  
   443  	fNoNamespace := `
   444    services:
   445    - group: postgresql.k8s.enterprisedb.io
   446      id: my-nodejs-app-cluster-sample
   447      kind: Cluster
   448      name: cluster-sample
   449      resource: clusters
   450      version: v1
   451  status:
   452    secret: ""
   453  `
   454  
   455  	fNamespace := fmt.Sprintf(`
   456    services:
   457    - group: postgresql.k8s.enterprisedb.io
   458      id: my-nodejs-app-cluster-sample
   459      kind: Cluster
   460      name: cluster-sample
   461      namespace: %s
   462      resource: clusters
   463      version: v1
   464  status:
   465    secret: ""
   466  `, ns)
   467  
   468  	if namingStrategy != "" {
   469  		if ns != "" {
   470  			return fmt.Sprintf(`%s
   471    namingStrategy: %s%s`, h, namingStrategy, fNamespace)
   472  		}
   473  		return fmt.Sprintf(`%s
   474    namingStrategy: %s%s`, h, namingStrategy, fNoNamespace)
   475  	}
   476  
   477  	if ns != "" {
   478  		return h + fNamespace
   479  	}
   480  
   481  	return h + fNoNamespace
   482  }
