   160  func TestDeleteEnvDirIfEmpty(t *testing.T) {
   161  	// create a fake fs in memory
   162  	fs := filesystem.NewFakeFs()
   163  	// create a odo config directory on fake fs
   164  	configDir, err := fs.TempDir(os.TempDir(), "odo")
   165  	if err != nil {
   166  		t.Error(err)
   167  	}
   168  	// create a mock env info from above fake fs & dir
   169  	esi, err := mockEnvSpecificInfo(configDir, fs)
   170  	if err != nil {
   171  		t.Error(err)
   172  	}
   173  
   174  	envDir := filepath.Join(configDir, ".odo", "env")
   175  	if _, err = fs.Stat(envDir); os.IsNotExist(err) {
   176  		t.Error("config directory doesn't exist")
   177  	}
   178  
   179  	tests := []struct {
   180  		name string
   181  		// create indicates if a file is supposed to be created in the odo config dir
   182  		create     bool
   183  		setupEnv   func(create bool, fs filesystem.Filesystem, envDir string) error
   184  		wantOdoDir bool
   185  		wantErr    bool
   186  	}{
   187  		{
   188  			name:       "Case 1: Empty config dir",
   189  			create:     false,
   190  			setupEnv:   createDirectoryAndFile,
   191  			wantOdoDir: false,
   192  		},
   193  		{
   194  			name:       "Case 2: Config dir with test file",
   195  			create:     true,
   196  			setupEnv:   createDirectoryAndFile,
   197  			wantOdoDir: true,
   198  		},
   199  	}
   200  
   201  	for _, tt := range tests {
   202  
   203  		err := tt.setupEnv(tt.create, fs, envDir)
   204  		if err != nil {
   205  			t.Error(err)
   206  		}
   207  
   208  		err = esi.DeleteEnvDirIfEmpty()
   209  		if err != nil {
   210  			t.Error(err)
   211  		}
   212  
   213  		file, err := fs.Stat(envDir)
   214  		if !tt.wantOdoDir && !os.IsNotExist(err) {
   215  			// we don't want odo dir but odo dir exists
   216  			fmt.Println(file.Size())
   217  			t.Error("odo env directory exists even after deleting it")
   218  			t.Errorf("Error in test %q", tt.name)
   219  		} else if tt.wantOdoDir && os.IsNotExist(err) {
   220  			// we want odo dir to exist after odo delete --all but it does not exist
   221  			t.Error("wanted odo directory to exist after odo delete --all")
   222  			t.Errorf("Error in test %q", tt.name)
   223  		}
   224  	}
   225  }
