   114  // StartDevMode starts a dev session with `odo dev`
   115  // It returns a session structure, the contents of the standard and error outputs
   116  // and the redirections endpoints to access ports opened by component
   117  // when the dev mode is completely started
   118  func StartDevMode(envvars []string, opts ...string) (DevSession, []byte, []byte, map[string]string, error) {
   119  	args := []string{"dev", "--random-ports"}
   120  	args = append(args, opts...)
   121  	session := Cmd("odo", args...).AddEnv(envvars...).Runner().session
   122  	WaitForOutputToContain("Press Ctrl+c to exit `odo dev` and delete resources from the cluster", 360, 10, session)
   123  	result := DevSession{
   124  		session: session,
   125  	}
   126  	outContents := session.Out.Contents()
   127  	errContents := session.Err.Contents()
   128  	err := session.Out.Clear()
   129  	if err != nil {
   130  		return DevSession{}, nil, nil, nil, err
   131  	}
   132  	err = session.Err.Clear()
   133  	if err != nil {
   134  		return DevSession{}, nil, nil, nil, err
   135  	}
   136  	return result, outContents, errContents, getPorts(string(outContents)), nil
   137  
   138  }

   145  // Stop a Dev session cleanly (equivalent as hitting Ctrl-c)
   146  func (o *DevSession) Stop() {
   147  	if o.stopped {
   148  		return
   149  	}
   150  	err := terminateProc(o.session)
   151  	gomega.Expect(err).NotTo(gomega.HaveOccurred())
   152  	o.stopped = true
   153  }

   159  // WaitSync waits for the synchronization of files to be finished
   160  // It returns the contents of the standard and error outputs
   161  // and the list of forwarded ports
   162  // since the end of the dev mode or the last time WaitSync/GetInfo has been called
   163  func (o DevSession) WaitSync() ([]byte, []byte, map[string]string, error) {
   164  	WaitForOutputToContainOne([]string{"Pushing files...", "Updating Component..."}, 180, 10, o.session)
   165  	WaitForOutputToContain("Watching for changes in the current directory", 240, 10, o.session)
   166  	return o.GetInfo()
   167  }

   192  // RunDevMode runs a dev session and executes the `inside` code when the dev mode is completely started
   193  // The inside handler is passed the internal session pointer, the contents of the standard and error outputs,
   194  // and a slice of strings - ports - giving the redirections in the form localhost:<port_number> to access ports opened by component
   195  func RunDevMode(additionalOpts []string, envvars []string, inside func(session *gexec.Session, outContents []byte, errContents []byte, ports map[string]string)) error {
   196  	session, outContents, errContents, urls, err := StartDevMode(envvars, additionalOpts...)
   197  	if err != nil {
   198  		return err
   199  	}
   200  	defer func() {
   201  		session.Stop()
   202  		session.WaitEnd()
   203  	}()
   204  	inside(session.session, outContents, errContents, urls)
   205  	return nil
   206  }

   208  // DevModeShouldFail runs `odo dev` with an intention to fail, and checks for a given substring
   209  // `odo dev` runs in an infinite reconciliation loop, and hence running it with Cmd will not work for a lot of failing cases,
   210  // this function is helpful in such cases.
   211  // TODO(pvala): Modify StartDevMode to take substring arg into account, and replace this method with it.
   212  func DevModeShouldFail(envvars []string, substring string, opts ...string) (DevSession, []byte, []byte, error) {
   213  	args := []string{"dev", "--random-ports"}
   214  	args = append(args, opts...)
   215  	session := Cmd("odo", args...).AddEnv(envvars...).Runner().session
   216  	WaitForOutputToContain(substring, 360, 10, session)
   217  	result := DevSession{
   218  		session: session,
   219  	}
   220  	defer func() {
   221  		result.Stop()
   222  		result.WaitEnd()
   223  	}()
   224  	outContents := session.Out.Contents()
   225  	errContents := session.Err.Contents()
   226  	err := session.Out.Clear()
   227  	if err != nil {
   228  		return DevSession{}, nil, nil, err
   229  	}
   230  	err = session.Err.Clear()
   231  	if err != nil {
   232  		return DevSession{}, nil, nil, err
   233  	}
   234  	return result, outContents, errContents, nil
   235  }

   237  // getPorts returns a map of ports redirected depending on the information in s
   238  //  `- Forwarding from 127.0.0.1:40001 -> 3000` will return { "3000": "127.0.0.1:40001" }
   239  func getPorts(s string) map[string]string {
   240  	result := map[string]string{}
   241  	re := regexp.MustCompile("(127.0.0.1:[0-9]+) -> ([0-9]+)")
   242  	matches := re.FindAllStringSubmatch(s, -1)
   243  	for _, match := range matches {
   244  		result[match[2]] = match[1]
   245  	}
   246  	return result
   247  }
