    33  // WaitForCmdOut runs a command until it gets
    34  // the expected output.
    35  // It accepts 5 arguments, program (program to be run)
    36  // args (arguments to the program)
    37  // timeoutInMinutes (the time to wait for the output)
    38  // errOnFail (flag to set if test should fail if command fails)
    39  // check (function with output check logic)
    40  // It times out if the command doesn't fetch the
    41  // expected output  within the timeout period.
    42  func WaitForCmdOut(program string, args []string, timeoutInMinutes int, errOnFail bool, check func(output string) bool, includeStdErr ...bool) bool {
    43  	pingTimeout := time.After(time.Duration(timeoutInMinutes) * time.Minute)
    44  	// this is a test package so time.Tick() is acceptable
    45  	// nolint
    46  	tick := time.Tick(time.Second)
    47  	for {
    48  		select {
    49  		case <-pingTimeout:
    50  			Fail(fmt.Sprintf("Timeout after %v minutes", timeoutInMinutes))
    51  
    52  		case <-tick:
    53  			session := CmdRunner(program, args...)
    54  			if errOnFail {
    55  				Eventually(session).Should(gexec.Exit(0), runningCmd(session.Command))
    56  			} else {
    57  				Eventually(session).Should(gexec.Exit(), runningCmd(session.Command))
    58  			}
    59  			session.Wait()
    60  			output := string(session.Out.Contents())
    61  
    62  			if len(includeStdErr) > 0 && includeStdErr[0] {
    63  				output += "\n"
    64  				output += string(session.Err.Contents())
    65  			}
    66  			if check(strings.TrimSpace(output)) {
    67  				return true
    68  			}
    69  		}
    70  	}
    71  }

   187  // CommonBeforeEach is common function runs before every test Spec (It)
   188  // returns CommonVar values that are used within the test script
   189  func CommonBeforeEach(setupCluster bool) CommonVar {
   190  	SetDefaultEventuallyTimeout(10 * time.Minute)
   191  	SetDefaultConsistentlyDuration(30 * time.Second)
   192  
   193  	commonVar := CommonVar{}
   194  	commonVar.Context = CreateNewContext()
   195  	commonVar.ConfigDir = CreateNewContext()
   196  	commonVar.OriginalKubeconfig = os.Getenv("KUBECONFIG")
   197  	commonVar.CliRunner = GetCliRunner()
   198  	LocalKubeconfigSet(commonVar.ConfigDir)
   199  	if setupCluster {
   200  		commonVar.Project = commonVar.CliRunner.CreateAndSetRandNamespaceProject()
   201  	}
   202  	commonVar.OriginalWorkingDirectory = Getwd()
   203  	os.Setenv("GLOBALODOCONFIG", filepath.Join(commonVar.ConfigDir, "preference.yaml"))
   204  	// Set ConsentTelemetry to false so that it does not prompt to set a preference value
   205  	cfg, _ := preference.NewClient()
   206  	err := cfg.SetConfiguration(preference.ConsentTelemetrySetting, "false")
   207  	Expect(err).To(BeNil())
   208  	// Use ephemeral volumes (emptyDir) in tests to make test faster
   209  	err = cfg.SetConfiguration(preference.EphemeralSetting, "true")
   210  	Expect(err).To(BeNil())
   211  	SetDefaultDevfileRegistryAsStaging()
   212  	return commonVar
   213  }

   215  // CommonAfterEach is common function that cleans up after every test Spec (It)
   216  func CommonAfterEach(commonVar CommonVar) {
   217  	// Get details, including test filename, test case name, test result, and test duration for each test spec and adds it to local testResults.txt file
   218  	// Ginkgo test related variables
   219  	commonVar.testFileName = CurrentSpecReport().ContainerHierarchyLocations[0].FileName
   220  	commonVar.testCase = CurrentSpecReport().FullText()
   221  	commonVar.testFailed = CurrentSpecReport().Failed()
   222  	commonVar.testDuration = CurrentSpecReport().RunTime.Seconds()
   223  
   224  	var prNum string
   225  	var resultsRow string
   226  	prNum = os.Getenv("GIT_PR_NUMBER")
   227  	passedOrFailed := "PASSED"
   228  	if commonVar.testFailed {
   229  		passedOrFailed = "FAILED"
   230  	}
   231  	clusterType := "OCP"
   232  	if IsKubernetesCluster() {
   233  		clusterType = "KUBERNETES"
   234  	}
   235  	testDate := strings.Split(time.Now().Format(time.RFC3339), "T")[0]
   236  	resultsRow = prNum + "," + testDate + "," + clusterType + "," + commonVar.testFileName + "," + commonVar.testCase + "," + passedOrFailed + "," + strconv.FormatFloat(commonVar.testDuration, 'E', -1, 64) + "\n"
   237  	testResultsFile := filepath.Join("/", "tmp", "testResults.txt")
   238  	if runtime.GOOS == "windows" {
   239  		testResultsFile = filepath.Join(os.Getenv("TEMP"), "testResults.txt")
   240  	}
   241  	f, err := os.OpenFile(testResultsFile, os.O_APPEND|os.O_WRONLY|os.O_CREATE, 0600)
   242  	if err != nil {
   243  		fmt.Println("Error when opening file: ", err)
   244  	} else {
   245  		_, err = f.WriteString(resultsRow)
   246  		if err != nil {
   247  			fmt.Println("Error when writing to file: ", err)
   248  		}
   249  		if err = f.Close(); err != nil {
   250  			fmt.Println("Error when closing file: ", err)
   251  		}
   252  	}
   253  
   254  	if commonVar.Project != "" {
   255  		// delete the random project/namespace created in CommonBeforeEach
   256  		commonVar.CliRunner.DeleteNamespaceProject(commonVar.Project, false)
   257  	}
   258  	// restores the original kubeconfig and working directory
   259  	Chdir(commonVar.OriginalWorkingDirectory)
   260  	err = os.Setenv("KUBECONFIG", commonVar.OriginalKubeconfig)
   261  	Expect(err).NotTo(HaveOccurred())
   262  
   263  	// delete the temporary context directory
   264  	DeleteDir(commonVar.Context)
   265  	DeleteDir(commonVar.ConfigDir)
   266  
   267  	os.Unsetenv("GLOBALODOCONFIG")
   268  }
